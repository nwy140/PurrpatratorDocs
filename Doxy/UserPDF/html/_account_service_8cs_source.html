71 public bool RegisterByEmail(string email, List<ServiceTypes> serviceTypes, Action<AccountServiceResponse> callback = null, Action<string> errorCallback = null, string origin = null)
72 {
73 if (this.RequestPendingResult)
74 {
75 Debug.LogError("Registration request pending result. Not sending another.");
76 return false;
77 }
78
79 if (!IsValidEmail(email))
80 {
81 Debug.LogErrorFormat("Email \"{0}\" is not valid", email);
82 return false;
83 }
84
85 string serviceTypeString = GetServiceTypesFromList(serviceTypes);
86 if (string.IsNullOrEmpty(serviceTypeString))
87 {
88 Debug.LogError("serviceTypes string is null or empty");
89 return false;
90 }
91
92 string fullUrl = GetUrlWithQueryStringEscaped(email, serviceTypeString, origin);
93
94 RequestHeaders["x-functions-key"] = string.IsNullOrEmpty(CustomToken) ? DefaultToken : CustomToken;
95
96
97 this.RequestPendingResult = true;
98
99 PhotonEditorUtils.StartCoroutine(
100 PhotonEditorUtils.HttpPost(fullUrl,
101 RequestHeaders,
102 null,
103 s =>
104 {
105 this.RequestPendingResult = false;
106 //Debug.LogWarningFormat("received response {0}", s);
107 if (string.IsNullOrEmpty(s))
108 {
109 if (errorCallback != null)
110 {
111 errorCallback("Server's response was empty. Please register through account website during this service interruption.");
112 }
113 }
114 else
115 {
116 AccountServiceResponse ase = this.ParseResult(s);
117 if (ase == null)
118 {
119 if (errorCallback != null)
120 {
121 errorCallback("Error parsing registration response. Please try registering from account website");
122 }
123 }
124 else if (callback != null)
125 {
126 callback(ase);
127 }
128 }
129 },
130 e =>
131 {
132 this.RequestPendingResult = false;
133 if (errorCallback != null)
134 {
135 errorCallback(e);
136 }
137 })
138 );
139 return true;
140 }

143 private string GetUrlWithQueryStringEscaped(string email, string serviceTypes, string originAv)
144 {
145 string emailEscaped = UnityEngine.Networking.UnityWebRequest.EscapeURL(email);
146 string st = UnityEngine.Networking.UnityWebRequest.EscapeURL(serviceTypes);

148 string serviceUrl = string.Format(ServiceUrl, string.IsNullOrEmpty(CustomContext) ? DefaultContext : CustomContext );

157 private AccountServiceResponse ParseResult(string result)
158 {
159 try
160 {
161 AccountServiceResponse res = JsonUtility.FromJson<AccountServiceResponse>(result);
162 // Unity's JsonUtility does not support deserializing Dictionary, we manually parse it, dirty & ugly af, better then using a 3rd party lib
163 if (res.ReturnCode == AccountServiceReturnCodes.Success)
164 {
165 string[] parts = result.Split(new[] { "\"ApplicationIds\":{" }, StringSplitOptions.RemoveEmptyEntries);
166 parts = parts[1].Split('}');
167 string applicationIds = parts[0];
168 if (!string.IsNullOrEmpty(applicationIds))
169 {
170 parts = applicationIds.Split(new[] { ',', '"', ':' }, StringSplitOptions.RemoveEmptyEntries);
171 res.ApplicationIds = new Dictionary<string, string>(parts.Length / 2);
172 for (int i = 0; i < parts.Length; i = i + 2)
173 {
174 res.ApplicationIds.Add(parts[i], parts[i + 1]);
175 }
176 }
177 else
178 {
179 Debug.LogError("The server did not return any AppId, ApplicationIds was empty in the response.");
180 return null;
181 }
182 }
183 return res;
184 }
185 catch (Exception ex) // probably JSON parsing exception, check if returned string is valid JSON
186 {
187 Debug.LogException(ex);
188 return null;
189 }
190 }

197 private static string GetServiceTypesFromList(List<ServiceTypes> appTypes)
198 {
199 if (appTypes == null || appTypes.Count <= 0)
200 {
201 return null;
202 }
203
204 string serviceTypes = ((int)appTypes[0]).ToString();
205 for (int i = 1; i < appTypes.Count; i++)
206 {
207 int appType = (int)appTypes[i];
208 serviceTypes = string.Format("{0},{1}", serviceTypes, appType);
209 }
210
211 return serviceTypes;
212 }

217 RegexOptions.CultureInvariant | RegexOptions.IgnoreCase);
218 public static bool IsValidEmail(string mailAddress)
219 {
220 if (string.IsNullOrEmpty(mailAddress))
221 {
222 return false;
223 }
224 var result = reg.Match(mailAddress);
225 return result.Success;
