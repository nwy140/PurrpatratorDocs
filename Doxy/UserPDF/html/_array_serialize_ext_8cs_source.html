274 public static void Append(this byte[] buffer, ulong value, ref int bitposition, int bits)
275 {
276 if (bits == 0)
277 return;
278
279 const int MAXBITS = 8;
280 const int MODULUS = MAXBITS - 1;
281 int offset = (bitposition & MODULUS); // this is just a modulus
282 int index = bitposition >> 3;
283
284 ulong offsetmask = ((1UL << offset) - 1);
285
286 ulong comp = buffer[index] & offsetmask;
287 ulong result = comp | (value << offset);
288
289 buffer[index] = (byte)result;
290
291 offset = MAXBITS - offset;
292 while (offset < bits)
293 {
294 index++;
295 buffer[index] = (byte)((value >> offset));
296 offset = offset + MAXBITS;
297 }
298 bitposition += bits;
299 }

383 public static void Write(this byte[] buffer, ulong value, ref int bitposition, int bits)
384 {
385 if (bits == 0)
386 return;
387
388 const int MAXBITS = 8;
389 const int MODULUS = MAXBITS - 1;
390 int offset = bitposition & MODULUS;
391 int index = bitposition >> 3;
392 int totalpush = offset + bits;
393
394 ulong mask = ulong.MaxValue >> (64 - bits);
395 ulong offsetmask = mask << offset;
396 ulong offsetcomp = value << offset;
397
398 buffer[index] = (byte)((buffer[index] & ~offsetmask) | (offsetcomp & offsetmask));
399
400 offset = MAXBITS - offset;
401 totalpush = totalpush - MAXBITS;
402
403 // These are complete overwrites of the array element, so no masking is required
404 while (totalpush > MAXBITS)
405 {
406 index++;
407 offsetcomp = value >> offset;
408 buffer[index] = (byte)offsetcomp;
409 offset += MAXBITS;
410 totalpush = totalpush - MAXBITS;
411 }
412
413 // remaning partial write needs masking
414 if (totalpush > 0)
415 {
416 index++;
417
418 offsetmask = mask >> offset;
419 offsetcomp = value >> offset;
420 buffer[index] = (byte)((buffer[index] & ~offsetmask) | (offsetcomp & offsetmask));
421 }
422 bitposition += bits;
423 }

517 public static ulong Read(this byte[] buffer, ref int bitposition, int bits)
518 {
519 if (bits == 0)
520 return 0;
521
522 const int MAXBITS = 8;
523 const int MODULUS = MAXBITS - 1;
524 int offset = bitposition & MODULUS; // this is just a modulus
525 int index = bitposition >> 3;
526
527#if UNITY_EDITOR || DEVELOPEMENT_BUILD
528 if ((bitposition + bits) > (buffer.Length << 3))
529 UnityEngine.Debug.LogError(bufferOverrunMsg);
530#endif
531
532 ulong mask = ulong.MaxValue >> (64 - bits);
533 ulong value = (ulong)buffer[index] >> offset;
534 offset = MAXBITS - offset;
535 while (offset < bits)
536 {
537 index++;
538 value |= (ulong)buffer[index] << offset;
539 offset += MAXBITS;
540 }
541
542 bitposition += bits;
543 return value & mask;
