186 public unsafe static void Append(ulong* uPtr, ulong value, ref int bitposition, int bits)
187 {
188 if (bits <= 0)
189 return;
190
191 const int MAXBITS = 64;
192 const int MODULUS = MAXBITS - 1;
193 int offset = bitposition & MODULUS;
194 int index = bitposition >> 6;
195
196 ulong offsetmask = ((1UL << offset) - 1);
197
198 ulong result = (uPtr[index] & offsetmask) | (value << offset);
199 uPtr[index] = result;
200 uPtr[index + 1] = result >> (MAXBITS - offset);
201
202 bitposition += bits;
203 }

241 public unsafe static void Write(ulong* uPtr, ulong value, ref int bitposition, int bits)
242 {
243 if (bits <= 0)
244 return;
245
246 const int MAXBITS = 64;
247 const int MODULUS = MAXBITS - 1;
248 int offset = bitposition & MODULUS;
249 int index = bitposition >> 6;
250
251 ulong mask = ulong.MaxValue >> (64 - bits);
252
253 ulong offsetmask = mask << offset;
254 ulong offsetcomp = value << offset;
255
256 uPtr[index] = (uPtr[index] & ~offsetmask) | (offsetcomp & offsetmask);
257
258 offset = MAXBITS - offset;
259
260 if (offset < bits)
261 {
262 offsetmask = mask >> offset;
263 offsetcomp = value >> offset;
264 index++;
265
266 uPtr[index] = (uPtr[index] & ~offsetmask) | (offsetcomp & offsetmask);
267 }
268 bitposition += bits;
269 }

279 public unsafe static ulong Read(ulong* uPtr, ref int bitposition, int bits)
280 {
281 if (bits == 0)
282 return 0;
283
284 const int MAXBITS = 64;

286 int offset = bitposition & MODULUS; // this is just a modulus

290
291 ulong value = uPtr[index] >> offset;
292 value |= uPtr[index + 1] << (MAXBITS - offset);
293
294 bitposition += bits;

567 public unsafe static void ReadOutUnsafe(ulong* sourcePtr, int sourcePos, ulong* targetPtr, ref int targetPos, int bits)
568 {
569 if (bits == 0)
570 return;
571
572 int readpos = sourcePos;
573 int remaining = bits;
574
575 while (remaining > 0)
576 {
577 int cnt = remaining > 64 ? 64 : remaining;
578 ulong val = Read(sourcePtr, ref readpos, cnt);
579 Write(targetPtr, val, ref targetPos, cnt);
580
581 remaining -= cnt;
582 }
583 }

594 public unsafe static void ReadOutUnsafe(this ulong[] source, int sourcePos, byte[] target, ref int targetPos, int bits)
595 {
596 if (bits == 0)
597 return;
598 int readpos = sourcePos;
599 int remaining = bits;
600
601 System.Diagnostics.Debug.Assert((targetPos + bits) <= (target.Length << 3), bufferOverrunMsg);
602 System.Diagnostics.Debug.Assert((sourcePos + bits) <= (source.Length << 3), bufferOverrunMsg);
603
604 fixed (ulong* sPtr = source)

606 {

612 ulong val = Read(sPtr, ref readpos, cnt);
