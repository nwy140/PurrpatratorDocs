95 private void DestroyAsteroidGlobally()
96 {
97 isDestroyed = true;
98
99 if (isLargeAsteroid)
100 {
101 int numberToSpawn = Random.Range(3, 6);
102
103 for (int counter = 0; counter < numberToSpawn; ++counter)
104 {
105 Vector3 force = Quaternion.Euler(0, counter * 360.0f / numberToSpawn, 0) * Vector3.forward * Random.Range(0.5f, 1.5f) * 300.0f;
106 Vector3 torque = Random.insideUnitSphere * Random.Range(500.0f, 1500.0f);
107 object[] instantiationData = {force, torque, false, PhotonNetwork.Time};
108
109 PhotonNetwork.InstantiateRoomObject("SmallAsteroid", transform.position + force.normalized * 10.0f, Quaternion.Euler(0, Random.value * 180.0f, 0), 0, instantiationData);
110 }
111 }
112
113 PhotonNetwork.Destroy(gameObject);
114 }
