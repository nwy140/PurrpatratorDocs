64 private IEnumerator SpawnAsteroid()
65 {
66 while (true)
67 {
68 yield return new WaitForSeconds(Random.Range(AsteroidsGame.ASTEROIDS_MIN_SPAWN_TIME, AsteroidsGame.ASTEROIDS_MAX_SPAWN_TIME));
69
70 Vector2 direction = Random.insideUnitCircle;
71 Vector3 position = Vector3.zero;
72
73 if (Mathf.Abs(direction.x) > Mathf.Abs(direction.y))
74 {
75 // Make it appear on the left/right side
76 position = new Vector3(Mathf.Sign(direction.x) * Camera.main.orthographicSize * Camera.main.aspect, 0, direction.y * Camera.main.orthographicSize);
77 }
78 else
79 {
80 // Make it appear on the top/bottom
81 position = new Vector3(direction.x * Camera.main.orthographicSize * Camera.main.aspect, 0, Mathf.Sign(direction.y) * Camera.main.orthographicSize);
82 }
83
84 // Offset slightly so we are not out of screen at creation time (as it would destroy the asteroid right away)
85 position -= position.normalized * 0.1f;
86
87
88 Vector3 force = -position.normalized * 1000.0f;
89 Vector3 torque = Random.insideUnitSphere * Random.Range(500.0f, 1500.0f);
90 object[] instantiationData = {force, torque, true};
91
92 PhotonNetwork.InstantiateRoomObject("BigAsteroid", position, Quaternion.Euler(Random.value * 360.0f, Random.value * 360.0f, Random.value * 360.0f), 0, instantiationData);
93 }
94 }

96 private IEnumerator EndOfGame(string winner, int score)
97 {
98 float timer = 5.0f;
99
100 while (timer > 0.0f)
101 {
102 InfoText.text = string.Format("Player {0} won with {1} points.\n\n\nReturning to login screen in {2} seconds.", winner, score, timer.ToString("n2"));
103
104 yield return new WaitForEndOfFrame();
105
106 timer -= Time.deltaTime;
107 }
108
109 PhotonNetwork.LeaveRoom();
110 }

139 public override void OnPlayerPropertiesUpdate(Player targetPlayer, Hashtable changedProps)
140 {
141 if (changedProps.ContainsKey(AsteroidsGame.PLAYER_LIVES))
142 {
143 CheckEndOfGame();
144 return;
145 }
146
147 if (!PhotonNetwork.IsMasterClient)
148 {
149 return;
150 }
151
152
153 // if there was no countdown yet, the master client (this one) waits until everyone loaded the level and sets a timer start
154 int startTimestamp;
155 bool startTimeIsSet = CountdownTimer.TryGetStartTime(out startTimestamp);
156
157 if (changedProps.ContainsKey(AsteroidsGame.PLAYER_LOADED_LEVEL))
158 {
159 if (CheckAllPlayerLoadedLevel())
160 {
161 if (!startTimeIsSet)
162 {
163 CountdownTimer.SetStartTime();
164 }
165 }
166 else

168 // not all players loaded yet. wait:
169 Debug.Log("setting text waiting for players! ",this.InfoText);

179 // called by OnCountdownTimerIsExpired() when the timer ended
180 private void StartGame()
181 {
182 Debug.Log("StartGame!");
183
184 // on rejoin, we have to figure out if the spaceship exists or not
185 // if this is a rejoin (the ship is already network instantiated and will be setup via event) we don't need to call PN.Instantiate
186
187
188 float angularStart = (360.0f / PhotonNetwork.CurrentRoom.PlayerCount) * PhotonNetwork.LocalPlayer.GetPlayerNumber();
189 float x = 20.0f * Mathf.Sin(angularStart * Mathf.Deg2Rad);
190 float z = 20.0f * Mathf.Cos(angularStart * Mathf.Deg2Rad);
191 Vector3 position = new Vector3(x, 0.0f, z);
192 Quaternion rotation = Quaternion.Euler(0.0f, angularStart, 0.0f);
193
194 PhotonNetwork.Instantiate("Spaceship", position, rotation, 0); // avoid this call on rejoin (ship was network instantiated before)
195
196 if (PhotonNetwork.IsMasterClient)
197 {
198 StartCoroutine(SpawnAsteroid());
199 }
200 }
201
202 private bool CheckAllPlayerLoadedLevel()
203 {
204 foreach (Player p in PhotonNetwork.PlayerList)
205 {
206 object playerLoadedLevel;
207
208 if (p.CustomProperties.TryGetValue(AsteroidsGame.PLAYER_LOADED_LEVEL, out playerLoadedLevel))
209 {
210 if ((bool) playerLoadedLevel)
211 {
212 continue;
213 }
214 }
215
216 return false;
217 }
218
219 return true;
220 }
221
222 private void CheckEndOfGame()
223 {
224 bool allDestroyed = true;
225
226 foreach (Player p in PhotonNetwork.PlayerList)
227 {
228 object lives;
229 if (p.CustomProperties.TryGetValue(AsteroidsGame.PLAYER_LIVES, out lives))
230 {
231 if ((int) lives > 0)
232 {
233 allDestroyed = false;
234 break;
235 }
236 }
237 }
238
239 if (allDestroyed)
240 {
241 if (PhotonNetwork.IsMasterClient)
242 {
243 StopAllCoroutines();
244 }
245
246 string winner = "";
247 int score = -1;
248
249 foreach (Player p in PhotonNetwork.PlayerList)
250 {
251 if (p.GetScore() > score)
252 {
253 winner = p.NickName;
254 score = p.GetScore();
255 }
256 }
257
258 StartCoroutine(EndOfGame(winner, score));
259 }
260 }
