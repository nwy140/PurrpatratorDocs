131 private void EnforceMode(int index)
132 {
133 int modeIndex = (index + 1) / 3;
134 BezierControlPointMode mode = modes[modeIndex];
135 if (mode == BezierControlPointMode.Free || !loop && (modeIndex == 0 || modeIndex == modes.Length - 1))
136 {
137 return;
138 }
139
140 int middleIndex = modeIndex * 3;
141 int fixedIndex, enforcedIndex;
142 if (index <= middleIndex)
143 {
144 fixedIndex = middleIndex - 1;
145 if (fixedIndex < 0)
146 {
147 fixedIndex = points.Length - 2;
148 }
149 enforcedIndex = middleIndex + 1;
150 if (enforcedIndex >= points.Length)
151 {
152 enforcedIndex = 1;
153 }
154 }else
155 {
156 fixedIndex = middleIndex + 1;
157 if (fixedIndex >= points.Length)
158 {
159 fixedIndex = 1;
160 }
161 enforcedIndex = middleIndex - 1;
162 if (enforcedIndex < 0)
163 {
164 enforcedIndex = points.Length - 2;
165 }
166 }
167
168 Vector3 middle = points[middleIndex];
169 Vector3 enforcedTangent = middle - points[fixedIndex];
170 if (mode == BezierControlPointMode.Aligned)
171 {
172 enforcedTangent = enforcedTangent.normalized * Vector3.Distance(middle, points[enforcedIndex]);
173 }
174 points[enforcedIndex] = middle + enforcedTangent;

262 public void ComputeLengths()
263 {
264 int subDivisions = 100;
265
266 int totalSamples = points.Length * subDivisions;
267
268 // lets create lengths for each control point.
269 this.lengths = new float[totalSamples];
270 this.lengthsTime = new float[totalSamples];
271
272 float totalDistance = 0;
273 float CurrentTime = 0f;
274
275 Vector3 pos;
276 Vector3 lastPos = this.GetPoint (0f);
277 // go from the first, to the second to last
278 for (var i = 0; i < totalSamples - 1; i++)
279 {
280 CurrentTime = (1f * i) / totalSamples;
281 pos = this.GetPoint (CurrentTime);
282
283 float _delta = (pos - lastPos).magnitude;
284
285 totalDistance += _delta ;
286 this.lengths [i] = totalDistance;
287
288 this.lengthsTime [i] = CurrentTime;
289 lastPos = pos;
290 }
291
292 this.TotalLength = totalDistance;
293
294 }

296 public Vector3 GetPositionAtDistance(float distance,bool reverse = false)
297 {
298 if (reverse)
299 {
300 distance = this.TotalLength - distance;
301 }
302
303 distance = Mathf.Repeat (distance, this.TotalLength);
304
305 // make sure that we are within the total distance of the points
306 if(distance <= 0) return points[0];
307 if(distance >= this.TotalLength) return points[points.Length - 1];
308
309 // lets find the first point that is below the distance
310 // but, who's next point is above the distance
311 var index = 0;
312 while (index < lengths.Length -1 && lengths[index] < distance)
313 index++;
