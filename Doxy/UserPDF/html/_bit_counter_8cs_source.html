202 public static int CountTrueBits(this int val, int range = 32)
203 {
204
205 int truecount = 0;
206 for (int i = 0; i < range; ++i)
207 {
208 int tmp = val >> i;
209
210 if (tmp == 0)
211 {
212 return truecount;
213 }
214
215 if ((tmp & 1) != 0)
216 {
217 truecount++;
218 }
219 }
220
221 return truecount;
222 }
