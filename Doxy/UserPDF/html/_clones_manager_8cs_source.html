72 public static Project CreateCloneFromPath(string sourceProjectPath)
73 {
74 Project sourceProject = new Project(sourceProjectPath);
75
76 string cloneProjectPath = null;
77
78 //Find available clone suffix id
79 for (int i = 0; i < MaxCloneProjectCount; i++)
80 {
81 string originalProjectPath = ClonesManager.GetCurrentProject().projectPath;
82 string possibleCloneProjectPath = originalProjectPath + ClonesManager.CloneNameSuffix + "_" + i;
83
84 if (!Directory.Exists(possibleCloneProjectPath))
85 {
86 cloneProjectPath = possibleCloneProjectPath;
87 break;
88 }
89 }
90
91 if (string.IsNullOrEmpty(cloneProjectPath))
92 {
93 Debug.LogError("The number of cloned projects has reach its limit. Limit: " + MaxCloneProjectCount);
94 return null;
95 }
96
97 Project cloneProject = new Project(cloneProjectPath);
98
99 Debug.Log("Start cloning project, original project: " + sourceProject + ", clone project: " + cloneProject);
100
101 ClonesManager.CreateProjectFolder(cloneProject);
102
103 //Copy Folders
104 Debug.Log("Library copy: " + cloneProject.libraryPath);
105 ClonesManager.CopyDirectoryWithProgressBar(sourceProject.libraryPath, cloneProject.libraryPath,
106 "Cloning Project Library '" + sourceProject.name + "'. ");
107 Debug.Log("Packages copy: " + cloneProject.libraryPath);
108 ClonesManager.CopyDirectoryWithProgressBar(sourceProject.packagesPath, cloneProject.packagesPath,
109 "Cloning Project Packages '" + sourceProject.name + "'. ");
110
111
112 //Link Folders
113 ClonesManager.LinkFolders(sourceProject.assetPath, cloneProject.assetPath);
114 ClonesManager.LinkFolders(sourceProject.projectSettingsPath, cloneProject.projectSettingsPath);
115 ClonesManager.LinkFolders(sourceProject.autoBuildPath, cloneProject.autoBuildPath);
116 ClonesManager.LinkFolders(sourceProject.localPackages, cloneProject.localPackages);
117
118 ClonesManager.RegisterClone(cloneProject);
119
120 return cloneProject;

190 public static bool IsCloneProjectRunning(string projectPath)
191 {
192
193 //Determine whether it is opened in another instance by checking the UnityLockFile
194 string UnityLockFilePath = new string[] { projectPath, "Temp", "UnityLockfile" }
195 .Aggregate(Path.Combine);
196
197 switch (Application.platform)
198 {
199 case (RuntimePlatform.WindowsEditor):
200 //Windows editor will lock "UnityLockfile" file when project is being opened.
201 //Sometime, for instance: windows editor crash, the "UnityLockfile" will not be deleted even the project
202 //isn't being opened, so a check to the "UnityLockfile" lock status may be necessary.
203 if (Preferences.AlsoCheckUnityLockFileStaPref.Value)
204 return File.Exists(UnityLockFilePath) && FileUtilities.IsFileLocked(UnityLockFilePath);
205 else
206 return File.Exists(UnityLockFilePath);
207 case (RuntimePlatform.OSXEditor):
208 //Mac editor won't lock "UnityLockfile" file when project is being opened
209 return File.Exists(UnityLockFilePath);
210 case (RuntimePlatform.LinuxEditor):
211 return File.Exists(UnityLockFilePath);
212 default:
213 throw new System.NotImplementedException("IsCloneProjectRunning: Unsupport Platfrom: " + Application.platform);
214 }

220 public static void DeleteClone(string cloneProjectPath)
221 {
223 if (ClonesManager.IsClone()) return;
224
226 if (cloneProjectPath == string.Empty) return;
227 if (cloneProjectPath == ClonesManager.GetOriginalProjectPath()) return;
228
229 //Check what OS is
230 string identifierFile;
231 string args;
232 switch (Application.platform)
233 {
234 case (RuntimePlatform.WindowsEditor):
235 Debug.Log("Attempting to delete folder \"" + cloneProjectPath + "\"");
236
237 //The argument file will be deleted first at the beginning of the project deletion process
238 //to prevent any further reading and writing to it(There's a File.Exist() check at the (file)editor windows.)
239 //If there's any file in the directory being write/read during the deletion process, the directory can't be fully removed.
240 identifierFile = Path.Combine(cloneProjectPath, ClonesManager.ArgumentFileName);
241 File.Delete(identifierFile);
242
243 args = "/c " + @"rmdir /s/q " + string.Format("\"{0}\"", cloneProjectPath);
244 StartHiddenConsoleProcess("cmd.exe", args);
245
246 break;
247 case (RuntimePlatform.OSXEditor):
248 Debug.Log("Attempting to delete folder \"" + cloneProjectPath + "\"");
249
250 //The argument file will be deleted first at the beginning of the project deletion process
251 //to prevent any further reading and writing to it(There's a File.Exist() check at the (file)editor windows.)
252 //If there's any file in the directory being write/read during the deletion process, the directory can't be fully removed.
253 identifierFile = Path.Combine(cloneProjectPath, ClonesManager.ArgumentFileName);
254 File.Delete(identifierFile);
255
256 FileUtil.DeleteFileOrDirectory(cloneProjectPath);
257
258 break;
259 case (RuntimePlatform.LinuxEditor):
260 Debug.Log("Attempting to delete folder \"" + cloneProjectPath + "\"");
261 identifierFile = Path.Combine(cloneProjectPath, ClonesManager.ArgumentFileName);
262 File.Delete(identifierFile);
263
264 FileUtil.DeleteFileOrDirectory(cloneProjectPath);
265
266 break;
267 default:
268 Debug.LogWarning("Not in a known editor. Where are you!?");
269 break;
270 }

533 private static void CopyDirectoryWithProgressBarRecursive(DirectoryInfo source, DirectoryInfo destination,
534 ref long totalBytes, ref long copiedBytes, string progressBarPrefix = "")
535 {
537 if (source.FullName.ToLower() == destination.FullName.ToLower())
538 {
539 Debug.LogError("Cannot copy directory into itself.");
540 return;
541 }
542
544 if (totalBytes == 0)
545 {
546 totalBytes = ClonesManager.GetDirectorySize(source, true, progressBarPrefix);
547 }
548
550 if (!Directory.Exists(destination.FullName))
551 {
552 Directory.CreateDirectory(destination.FullName);
553 }
554
556 foreach (FileInfo file in source.GetFiles())
557 {
558 try
559 {
560 file.CopyTo(Path.Combine(destination.ToString(), file.Name), true);
561 }
562 catch (IOException)
563 {
566 }
567
569 copiedBytes += file.Length;
570
572 float progress = (float)copiedBytes / (float)totalBytes;
573 bool cancelCopy = EditorUtility.DisplayCancelableProgressBar(
574 progressBarPrefix + "Copying '" + source.FullName + "' to '" + destination.FullName + "'...",
575 "(" + (progress * 100f).ToString("F2") + "%) Copying file '" + file.Name + "'...",
576 progress);
577 if (cancelCopy) return;
578 }
579
581 foreach (DirectoryInfo sourceNestedDir in source.GetDirectories())
582 {
583 DirectoryInfo nextDestingationNestedDir = destination.CreateSubdirectory(sourceNestedDir.Name);
584 ClonesManager.CopyDirectoryWithProgressBarRecursive(sourceNestedDir, nextDestingationNestedDir,
585 ref totalBytes, ref copiedBytes, progressBarPrefix);
586 }
587 }

596 private static long GetDirectorySize(DirectoryInfo directory, bool includeNested = false,
597 string progressBarPrefix = "")
598 {
599 EditorUtility.DisplayProgressBar(progressBarPrefix + "Calculating size of directories...",
600 "Scanning '" + directory.FullName + "'...", 0f);
601
603 long filesSize = directory.GetFiles().Sum((FileInfo file) => file.Length);
604
606 long directoriesSize = 0;
607 if (includeNested)
608 {
609 IEnumerable<DirectoryInfo> nestedDirectories = directory.GetDirectories();
610 foreach (DirectoryInfo nestedDir in nestedDirectories)
611 {
612 directoriesSize += ClonesManager.GetDirectorySize(nestedDir, true, progressBarPrefix);
613 }
614 }
615
616 return filesSize + directoriesSize;
617 }
