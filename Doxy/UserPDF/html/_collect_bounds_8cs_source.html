34 public static Bounds CollectMyBounds(this GameObject go, BoundsType factorIn, out int numOfBoundsFound, bool includeChildren = true, bool includeInactive = false)
35 {
36 // if we are ignoring inactive, an inactive parent is already a null. Quit here.
37 if (!go.activeInHierarchy && !!includeInactive)
38 {
39 numOfBoundsFound = 0;
40 return new Bounds();
41 }
42
43 bool bothtype = factorIn == BoundsType.Both;
44 bool rendtype = bothtype || factorIn == BoundsType.MeshRenderer;
45 bool colltype = bothtype || factorIn == BoundsType.Collider;
46
47 // Clear the reusables so they have counts of zero
48 meshFilters.Clear();
49 meshRenderers.Clear();
50 colliders.Clear();
51 spriteRenderers.Clear();
52 validColliders.Clear();
53#if !DISABLE_PHYSICS_2D
54 validColliders2D.Clear();
55#endif
56 int myBoundsCount = 0;
57
58 // Find all of the MeshRenderers and Colliders (as specified)
59 if (rendtype)
60 {
61 if (go.activeInHierarchy)
62 {
63 if (includeChildren)
64 {
65 go.GetComponentsInChildren(includeInactive, meshRenderers);
66 go.GetComponentsInChildren(includeInactive, meshFilters);
67 go.GetComponentsInChildren(includeInactive, spriteRenderers);
68 }
69 else
70 {
71 go.GetComponents(meshRenderers);
72 go.GetComponents(meshFilters);
73 go.GetComponents(spriteRenderers);
74 }
75 }
76 }
77
78 if (colltype)
79 {
80 if (go.activeInHierarchy)
81 {
82 if (includeChildren)
83 {
84 go.GetComponentsInChildren(includeInactive, colliders);
85#if !DISABLE_PHYSICS_2D
86 go.GetComponentsInChildren(includeInactive, colliders2D);
87#endif
88
89 }
90 else
91 {
92 go.GetComponents(colliders);
93#if !DISABLE_PHYSICS_2D
94 go.GetComponents(colliders2D);
95#endif
96 }
97 }
98 }
99
100 // Add any MeshRenderer attached to the found MeshFilters to their own list.
101 // We want the MeshRenderer for its bounds, but only if there is a MeshFilter, otherwise there is a risk of a 0,0,0
102 for (int i = 0; i < meshFilters.Count; i++)
103 {
104 Renderer mr = meshFilters[i].GetComponent<Renderer>();
105
106 if (mr && (mr.enabled || includeInactive))
107 {
108 if (!meshRenderers.Contains(mr))
109 meshRenderers.Add(mr);
110 }
111 }
112
113 // Collect only the valid colliders (ignore inactive if not includeInactive)
114 for (int i = 0; i < colliders.Count; i++)
115 {
116 if (colliders[i].enabled || includeInactive)
117 if (colliders[i])
118 validColliders.Add(colliders[i]);
119 }
120
121#if !DISABLE_PHYSICS_2D
122
123 // Collect only the valid colliders (ignore inactive if not includeInactive)
124 for (int i = 0; i < colliders2D.Count; i++)
125 {
126 if (colliders2D[i] && colliders2D[i].enabled || includeInactive)
127 // 2d colliders arrive as null but present in scene changes, test for null
128 if (colliders2D[i])
129 validColliders2D.Add(colliders2D[i]);
130 }
131#endif
132 // Make sure we found some bounds objects, or we need to quit.
133 numOfBoundsFound =
134 meshRenderers.Count +
135 spriteRenderers.Count +
136 validColliders.Count
137#if !DISABLE_PHYSICS_2D
138 + validColliders2D.Count
139#endif
140 ;
141 // No values means no bounds will be found, and this will break things if we try to use it.
142 if (numOfBoundsFound == 0)
143 {
144 return new Bounds();
145 }
146
147 // Get a starting bounds. We need this because the default of centered 0,0,0 will break things if the map is
148 // offset and doesn't encapsulte the world origin.
149 Bounds compositeBounds;
150
151 if (meshRenderers.Count > 0)
152 compositeBounds = meshRenderers[0].bounds;
153
154 else if (validColliders.Count > 0)
155 compositeBounds = validColliders[0].bounds;
156
157#if !DISABLE_PHYSICS_2D
158 else if (validColliders2D.Count > 0 && validColliders2D[0])
159 compositeBounds = validColliders2D[0].bounds;
160#endif
161 else if (spriteRenderers.Count > 0)
162 compositeBounds = spriteRenderers[0].bounds;
164 else
165 return new Bounds();
166
167 for (int i = 0; i < spriteRenderers.Count; i++)
168 {
169 myBoundsCount++;
170 compositeBounds.Encapsulate(spriteRenderers[i].bounds);
171 }
172
173 // Encapsulate all outer found bounds into that. We will be adding the root to itself, but no biggy, this only runs once.
174 for (int i = 0; i < meshRenderers.Count; i++)
175 {
176 myBoundsCount++;
177 compositeBounds.Encapsulate(meshRenderers[i].bounds);
178 }
179
180 for (int i = 0; i < validColliders.Count; i++)
181 {
182 myBoundsCount++;
183 compositeBounds.Encapsulate(validColliders[i].bounds);
184 }
185#if !DISABLE_PHYSICS_2D
186 for (int i = 0; i < validColliders2D.Count; i++)
187 {
188 myBoundsCount++;
189 if (validColliders2D[i])
190 compositeBounds.Encapsulate(validColliders2D[i].bounds);
191 }
192#endif
193 return compositeBounds;
194
195 }
