116 private void CreateCellHierarchy()
117 {
118 if (!this.IsCellCountAllowed())
119 {
120 if (Debug.isDebugBuild)
121 {
122 Debug.LogError("There are too many cells created by your subdivision options. Maximum allowed number of cells is " + (MAX_NUMBER_OF_ALLOWED_CELLS - this.FIRST_GROUP_ID) +
123 ". Current number of cells is " + this.CellCount + ".");
124 return;
125 }
126 else
127 {
128 Application.Quit();
129 }
130 }
131
132 CellTreeNode rootNode = new CellTreeNode(this.idCounter++, CellTreeNode.ENodeType.Root, null);
133
134 if (this.YIsUpAxis)
135 {
136 this.Center = new Vector2(transform.position.x, transform.position.y);
137 this.Size = new Vector2(transform.localScale.x, transform.localScale.y);
138
139 rootNode.Center = new Vector3(this.Center.x, this.Center.y, 0.0f);
140 rootNode.Size = new Vector3(this.Size.x, this.Size.y, 0.0f);
141 rootNode.TopLeft = new Vector3((this.Center.x - (this.Size.x / 2.0f)), (this.Center.y - (this.Size.y / 2.0f)), 0.0f);
142 rootNode.BottomRight = new Vector3((this.Center.x + (this.Size.x / 2.0f)), (this.Center.y + (this.Size.y / 2.0f)), 0.0f);
143 }
144 else
145 {
146 this.Center = new Vector2(transform.position.x, transform.position.z);
147 this.Size = new Vector2(transform.localScale.x, transform.localScale.z);
148
149 rootNode.Center = new Vector3(this.Center.x, 0.0f, this.Center.y);
150 rootNode.Size = new Vector3(this.Size.x, 0.0f, this.Size.y);
151 rootNode.TopLeft = new Vector3((this.Center.x - (this.Size.x / 2.0f)), 0.0f, (this.Center.y - (this.Size.y / 2.0f)));
152 rootNode.BottomRight = new Vector3((this.Center.x + (this.Size.x / 2.0f)), 0.0f, (this.Center.y + (this.Size.y / 2.0f)));
153 }
154
155 this.CreateChildCells(rootNode, 1);
156
157 this.CellTree = new CellTree(rootNode);

159 this.RecreateCellHierarchy = false;

167 private void CreateChildCells(CellTreeNode parent, int cellLevelInHierarchy)
168 {
169 if (cellLevelInHierarchy > this.NumberOfSubdivisions)
170 {
171 return;
172 }
173
174 int rowCount = (int)this.Subdivisions[(cellLevelInHierarchy - 1)].x;
175 int columnCount = (int)this.Subdivisions[(cellLevelInHierarchy - 1)].y;
176
177 float startX = parent.Center.x - (parent.Size.x / 2.0f);
178 float width = parent.Size.x / rowCount;
179
180 for (int row = 0; row < rowCount; ++row)
181 {
182 for (int column = 0; column < columnCount; ++column)
183 {
184 float xPos = startX + (row * width) + (width / 2.0f);
185
186 CellTreeNode node = new CellTreeNode(this.idCounter++, (this.NumberOfSubdivisions == cellLevelInHierarchy) ? CellTreeNode.ENodeType.Leaf : CellTreeNode.ENodeType.Node, parent);
187
188 if (this.YIsUpAxis)
189 {
190 float startY = parent.Center.y - (parent.Size.y / 2.0f);
191 float height = parent.Size.y / columnCount;
192 float yPos = startY + (column * height) + (height / 2.0f);
193
194 node.Center = new Vector3(xPos, yPos, 0.0f);
195 node.Size = new Vector3(width, height, 0.0f);
196 node.TopLeft = new Vector3(xPos - (width / 2.0f), yPos - (height / 2.0f), 0.0f);
197 node.BottomRight = new Vector3(xPos + (width / 2.0f), yPos + (height / 2.0f), 0.0f);
198 }
199 else
200 {
201 float startZ = parent.Center.z - (parent.Size.z / 2.0f);
202 float depth = parent.Size.z / columnCount;
203 float zPos = startZ + (column * depth) + (depth / 2.0f);
204
205 node.Center = new Vector3(xPos, 0.0f, zPos);
206 node.Size = new Vector3(width, 0.0f, depth);
207 node.TopLeft = new Vector3(xPos - (width / 2.0f), 0.0f, zPos - (depth / 2.0f));
208 node.BottomRight = new Vector3(xPos + (width / 2.0f), 0.0f, zPos + (depth / 2.0f));
209 }
210
211 parent.AddChild(node);
212
213 this.CreateChildCells(node, (cellLevelInHierarchy + 1));
214 }
215 }

237 private bool IsCellCountAllowed()
238 {
239 int horizontalCells = 1;
240 int verticalCells = 1;
241
242 foreach (Vector2 v in this.Subdivisions)
243 {
244 horizontalCells *= (int)v.x;
245 verticalCells *= (int)v.y;
246 }
247
248 this.CellCount = horizontalCells * verticalCells;
249
250 return (this.CellCount <= (MAX_NUMBER_OF_ALLOWED_CELLS - this.FIRST_GROUP_ID));
251 }

258 public List<byte> GetActiveCells(Vector3 position)
259 {
260 List<byte> activeCells = new List<byte>(0);
261 this.CellTree.RootNode.GetActiveCells(activeCells, this.YIsUpAxis, position);
262
263 // it makes sense to sort the "nearby" cells. those are in the list in positions after the subdivisions the point is inside. 2 subdivisions result in 3 areas the point is in.
264 int cellsActive = this.NumberOfSubdivisions + 1;
265 int cellsNearby = activeCells.Count - cellsActive;
266 if (cellsNearby > 0)
267 {
268 activeCells.Sort(cellsActive, cellsNearby, new ByteComparer());
269 }
270 return activeCells;
271 }

411 public void GetActiveCells(List<byte> activeCells, bool yIsUpAxis, Vector3 position)
412 {
413 if (this.NodeType != ENodeType.Leaf)
414 {
415 foreach (CellTreeNode node in this.Childs)
416 {
417 node.GetActiveCells(activeCells, yIsUpAxis, position);
418 }
419 }
420 else
421 {
422 if (this.IsPointNearCell(yIsUpAxis, position))
423 {
424 if (this.IsPointInsideCell(yIsUpAxis, position))
425 {
426 activeCells.Insert(0, this.Id);
427
428 CellTreeNode p = this.Parent;
429 while (p != null)
430 {
431 activeCells.Insert(0, p.Id);
432
433 p = p.Parent;
434 }
435 }
436 else
437 {
438 activeCells.Add(this.Id);
439 }
440 }
441 }

450 public bool IsPointInsideCell(bool yIsUpAxis, Vector3 point)
451 {
452 if ((point.x < this.TopLeft.x) || (point.x > this.BottomRight.x))
453 {
454 return false;
455 }
456
457 if (yIsUpAxis)
458 {
459 if ((point.y >= this.TopLeft.y) && (point.y <= this.BottomRight.y))
460 {
461 return true;
462 }
463 }
464 else
465 {
466 if ((point.z >= this.TopLeft.z) && (point.z <= this.BottomRight.z))
467 {
468 return true;
469 }
470 }
471
472 return false;
473 }

481 public bool IsPointNearCell(bool yIsUpAxis, Vector3 point)
482 {
483 if (this.maxDistance == 0.0f)
484 {
485 this.maxDistance = (this.Size.x + this.Size.y + this.Size.z) / 2.0f;
486 }
487
488 return ((point - this.Center).sqrMagnitude <= (this.maxDistance * this.maxDistance));
489 }
