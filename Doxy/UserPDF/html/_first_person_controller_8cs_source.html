133 private void Move()
134 {
135 // set target speed based on move speed, sprint speed and if sprint is pressed
136 float targetSpeed = _input.sprint ? SprintSpeed : MoveSpeed;
137
138 // a simplistic acceleration and deceleration designed to be easy to remove, replace, or iterate upon
139
140 // note: Vector2's == operator uses approximation so is not floating point error prone, and is cheaper than magnitude
141 // if there is no input, set the target speed to 0
142 if (_input.move == Vector2.zero) targetSpeed = 0.0f;
143
144 // a reference to the players current horizontal velocity
145 float currentHorizontalSpeed = new Vector3(_controller.velocity.x, 0.0f, _controller.velocity.z).magnitude;
146
147 float speedOffset = 0.1f;
148 float inputMagnitude = _input.analogMovement ? _input.move.magnitude : 1f;
149
150 // accelerate or decelerate to target speed
151 if (currentHorizontalSpeed < targetSpeed - speedOffset || currentHorizontalSpeed > targetSpeed + speedOffset)
152 {
153 // creates curved result rather than a linear one giving a more organic speed change
154 // note T in Lerp is clamped, so we don't need to clamp our speed
155 _speed = Mathf.Lerp(currentHorizontalSpeed, targetSpeed * inputMagnitude, Time.deltaTime * SpeedChangeRate);
156
157 // round speed to 3 decimal places
158 _speed = Mathf.Round(_speed * 1000f) / 1000f;
159 }
160 else
161 {
162 _speed = targetSpeed;
163 }
164
165 // normalise input direction
166 Vector3 inputDirection = new Vector3(_input.move.x, 0.0f, _input.move.y).normalized;
167
168 // note: Vector2's != operator uses approximation so is not floating point error prone, and is cheaper than magnitude
169 // if there is a move input rotate player when the player is moving
170 if (_input.move != Vector2.zero)
171 {
172 // move
173 inputDirection = transform.right * _input.move.x + transform.forward * _input.move.y;
174 }
175
176 // move the player
177 _controller.Move(inputDirection.normalized * (_speed * Time.deltaTime) + new Vector3(0.0f, _verticalVelocity, 0.0f) * Time.deltaTime);
178 }

180 private void JumpAndGravity()
181 {
182 if (Grounded)
183 {
184 // reset the fall timeout timer
185 _fallTimeoutDelta = FallTimeout;
186
187 // stop our velocity dropping infinitely when grounded
188 if (_verticalVelocity < 0.0f)
189 {
190 _verticalVelocity = -2f;
191 }
192
193 // Jump
194 if (_input.jump && _jumpTimeoutDelta <= 0.0f)
195 {
196 // the square root of H * -2 * G = how much velocity needed to reach desired height
197 _verticalVelocity = Mathf.Sqrt(JumpHeight * -2f * Gravity);
198 }
199
200 // jump timeout
201 if (_jumpTimeoutDelta >= 0.0f)
202 {
203 _jumpTimeoutDelta -= Time.deltaTime;
204 }
205 }
206 else
207 {
208 // reset the jump timeout timer
209 _jumpTimeoutDelta = JumpTimeout;
210
211 // fall timeout
212 if (_fallTimeoutDelta >= 0.0f)
213 {
214 _fallTimeoutDelta -= Time.deltaTime;
215 }
216
217 // if we are not grounded, do not jump
218 _input.jump = false;
219 }
220
221 // apply gravity over time if under terminal (multiply by delta time twice to linearly speed up over time)
222 if (_verticalVelocity < _terminalVelocity)
223 {
224 _verticalVelocity += Gravity * Time.deltaTime;
225 }
226 }
