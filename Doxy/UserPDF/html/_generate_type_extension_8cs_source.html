440 private static void GenerateFields(Type pObjType)
441 {
442 sbFrame.Length = 0;
443 sbFactory.Length = 0;
444 sbSnapshot.Length = 0;
445 sbCopyToFrame.Length = 0;
446 sbCapture.Length = 0;
447 sbApplyToObj.Length = 0;
448 sbCopyFrameToFrame.Length = 0;
449 sbInterpToObj.Length = 0;
450 sbInterpToFrame.Length = 0;
451 sbDelegates.Length = 0;
452 sbInit.Length = 0;
453 sbPack.Length = 0;
454 sbUnpack.Length = 0;
455
456 hasSnapCallbacks = false;
457
458 var pObjAttr = (pObjType.GetCustomAttributes(typeof(PackObjectAttribute), false)[0] as PackObjectAttribute);
459
460 var defaultInclusion = pObjAttr.defaultInclusion;
461
462 var fields = pObjType.GetFields(BindingFlags.Public | BindingFlags.Instance | BindingFlags.NonPublic);
463
464 for (int i = 0, cnt = fields.Length; i < cnt; ++i)
465 {
466 var fieldinfo = fields[i];
467
468 string fname = fieldinfo.Name;
469 string objNamespace = pObjType.Namespace;
470 string objTypeName = (objNamespace != null && objNamespace != "") ? (pObjType.Namespace + "." + pObjType.Name) : pObjType.Name;
471
472 Type ftype = fieldinfo.FieldType;
473 string ftypename = fieldinfo.FieldType.Name;
474
475 string ffulltype;
476 string genericarg;
477 string primitive;
478
479 if (ftype.IsGenericType)
480 {
481 primitive = ftype.GetGenericArguments()[0].ToString();
482 ffulltype = ftype.FullName.ToString().Replace("`1[", "<").Replace("]", ">").Replace("+", ".");
483 genericarg = "<" + primitive + ">";
484 ftypename = fieldinfo.FieldType.Name.Replace("`1", "");
485 }
486 else
487 {
488 ffulltype = ftype.FullName.Replace("+",".");
489 ftypename = fieldinfo.FieldType.Name;
490 primitive = ffulltype;
491 genericarg = "";
492 }
493
494
496 var packAttrs = fieldinfo.GetCustomAttributes(typeof(SyncVarBaseAttribute), false);
497
498 bool isExplicitInclusion = defaultInclusion == DefaultPackInclusion.Explicit;
499
500 if (isExplicitInclusion && packAttrs.Length == 0)
501 continue;
502
503 SyncVarBaseAttribute packAttr; // = (objType.GetCustomAttributes(typeof(SyncVarBaseAttribute), false)[0] as SyncVarBaseAttribute);
504
505 if (ftype.IsValueType && !ftype.IsUnManaged())
506 {
507 Debug.LogWarning(ftypename + " is a PackObject, but cannot be packed because it is a managed type. Structs cannot contain references or any managed types to be packable.");
508 continue;
509 }
510
511 bool isPublic = fieldinfo.IsPublic;
512
513 bool generatePackAttribute;
515 if (packAttrs.Length == 0)
516 {
517 // Only add attributes automatically if field is public. Otherwise including all public fields
518 // Will find all kinds of MonoB internals.
519 if (!isPublic)
520 continue;
521
522 packAttr = new SyncVarAttribute();
523 generatePackAttribute = true;
524 }
525 else
526 {
527 // TODO: make this a loop in case other attributes are also on this field.
528 packAttr = packAttrs[0] as SyncVarBaseAttribute;
529 generatePackAttribute = false;
530 }
531
532 bool typeIsSupported = TypeIsHandled(ftype, packAttr);
533
534 string attrname = packAttr.GetType().Name;
535
536 if (typeIsSupported && !isPublic)
537 {
538 Debug.LogWarning("<b>[" + attrname.Replace("Attribute", "") + "]</b> on field <b>" + ftypename + " " + fname + "</b> of <b>" + objTypeName + "</b> supports type, but field is not public. Change field to public to include in packing.");
539 continue;
540 }
541
542 var nestedPackObjAttrs = ftype.GetCustomAttributes(typeof(PackObjectAttribute), false);
543 PackObjectAttribute nestedPackObjAttr = nestedPackObjAttrs.Length == 0 ? null : (nestedPackObjAttrs[0] as PackObjectAttribute);
544 bool isNestedPackObj = nestedPackObjAttr != null;
545
546#if UNITY_2018_1_OR_NEWER
547
548 if (isNestedPackObj && !UnityEditor.PlayerSettings.allowUnsafeCode && ftype.IsValueType)
549 {
550 Debug.LogWarning("Cannot use " + typeof(PackObjectAttribute).Name + " on struct/value types unless PlayerSettings.allowUnsafeCode == true.");
551 continue;
552
553 }
554#endif
555
557 if (!typeIsSupported)
558 {
559
561 if (!isNestedPackObj)
562 {
564 if (!generatePackAttribute)
565 {
566 Debug.LogWarning("<b>[" + attrname.Replace("Attribute", "") + "]</b> on field <b>" + ftypename + " " + fname + "</b> of <b>" + objTypeName + fname + "</b> does not support type <b>"
567 + ftypename + "</b>. Will be ignored.");
568 }
569
570 continue;
571 }
572
573
575 if (CheckForNestedLoop(pObjType, ftype))
576 {
577 Debug.LogWarning("Field <b>" + ftypename + " " + fname + "</b> of <b>" + objTypeName + "</b> contains an infinite nested loop with field <b>" + "</b>. Will not be included in serialization.");
578 continue;
579 }
580
581 isNestedPackObj = true;
582 }
583
584 bool isEnum = fieldinfo.FieldType.IsEnum;
585 string enumUnderType = null;
586 string enumType = null; ;
587 if (isEnum)
588 {
589 enumUnderType = "(" + System.Enum.GetUnderlyingType(fieldinfo.FieldType).Name + ")";
590 enumType = "(" + ffulltype + ")";
591 }
592
593 Generate_FieldFrame(fieldinfo, sbFrame, fname, ftypename, ffulltype, isNestedPackObj, packAttr);
594 Generate_Factory(fieldinfo, sbFactory, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
595 Generate_Snapshot(fieldinfo, sbSnapshot, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
596 Generate_Capture(fieldinfo, sbCapture, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, pObjAttr, packAttr);
597 Generate_CopyTo(fieldinfo, sbCopyToFrame, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
598 Generate_ApplyToObj(fieldinfo, sbApplyToObj, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, pObjAttr, packAttr);
599 Generate_CopyFrameToFrame(fieldinfo, sbCopyFrameToFrame, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
600 Generate_InterpToObj(fieldinfo, sbInterpToObj, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
601 Generate_InterpToFrame(fieldinfo, sbInterpToFrame, objTypeName, fname, ftypename, ffulltype, isNestedPackObj, pObjType, packAttr);
602 Generate_FieldDelegates(fieldinfo, sbDelegates, fname, ftypename, genericarg, isNestedPackObj);
603 Generate_FieldInit(fieldinfo, sbInit, fname, ftypename, ffulltype, genericarg, primitive, isNestedPackObj, generatePackAttribute, packAttr);
604 Generate_FieldPack(fieldinfo, sbPack, fname, ftypename, ffulltype, isNestedPackObj, enumUnderType, enumType);
605 Generate_FieldUnpack(fieldinfo, sbUnpack, fname, ftypename, ffulltype, isNestedPackObj, enumUnderType, enumType, packAttr);
606 }
607
608 return;
609 }
