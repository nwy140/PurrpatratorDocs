49 public static Transform GetRandomSpawnPoint(bool avoidCollision = true)
50 {
51 if (spawns.Count == 0)
52 return null;
53
54 int startindex = Random.Range(0, spawns.Count - 1);
55
56 // Try to find a spawn that doesn't conflict
57 if (avoidCollision)
58 for (int i = 0; i < spawns.Count; i++)
59 if (!spawns[(i + startindex) % spawns.Count].IsBlocked)
60 return spawns[(i + startindex) % spawns.Count].transform;
61
62 // No collision free spawn found
63 return spawns[startindex].transform;
64 }
65
66 public static Transform GetNextSpawnPoint(bool avoidCollision = true)
67 {
68 if (spawns.Count == 0)
69 return null;
70
71 lastPicked = (lastPicked + 1) % spawns.Count;
72
73 // Try to find a spawn that doesn't conflict
74 if (avoidCollision)
75 for (int i = 0; i < spawns.Count; i++)
76 {
77 int next = (i + lastPicked) % spawns.Count;
78 if (!spawns[next].IsBlocked)
79 {
80 lastPicked = next;
81 break;
82 }
83 }
84
85 return spawns[lastPicked].transform;
86 }
87
88 public static Transform GetSpawnPointFromValue(int value)
89 {
90 if (spawns.Count == 0)
91 return null;
92
93 int spawnId = (value + 1) % spawns.Count;
94
95 return spawns[spawnId].transform;
