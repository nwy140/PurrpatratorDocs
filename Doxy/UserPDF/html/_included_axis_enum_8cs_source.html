53 public static Vector3 Lerp(this GameObject go, Vector3 start, Vector3 end, IncludedAxes ia, float t, bool localPosition = false)
54 {
55 // the lerped position, not accounting for any axis that are not included
56 Vector3 rawLerpedPos = Vector3.Lerp(start, end, t);
57
58 // for non-included axis, use the current postion
59 return new Vector3(
60 (((int)ia & 1) != 0) ? rawLerpedPos[0] : ((localPosition) ? go.transform.localPosition[0] : go.transform.position[0]),
61 (((int)ia & 2) != 0) ? rawLerpedPos[1] : ((localPosition) ? go.transform.localPosition[1] : go.transform.position[1]),
62 (((int)ia & 4) != 0) ? rawLerpedPos[2] : ((localPosition) ? go.transform.localPosition[2] : go.transform.position[2]));
63 }

68 public static void SetPosition(this GameObject go, Vector3 pos, IncludedAxes ia, bool localPosition = false)
69 {
70 Vector3 newpos = new Vector3(
71 (((int)ia & 1) != 0) ? pos[0] : (localPosition) ? go.transform.localPosition[0] : go.transform.position[0],
72 (((int)ia & 2) != 0) ? pos[1] : (localPosition) ? go.transform.localPosition[1] : go.transform.position[1],
73 (((int)ia & 4) != 0) ? pos[2] : (localPosition) ? go.transform.localPosition[2] : go.transform.position[2]);
74
75 if (!localPosition)
76 go.transform.position = newpos;
77 else
78 go.transform.localPosition = newpos;
79 }
