23 public static T GetActualObjectForSerializedProperty<T>(this FieldInfo fieldInfo, SerializedProperty property) where T : class
24 {
25 var obj = fieldInfo.GetValue(property.serializedObject.targetObject);
26
27 if (obj == null) { return null; }
28
29 T actualObject = null;
30 if (obj.GetType().IsArray)
31 {
32 var index = Convert.ToInt32(new string(property.propertyPath.Where(c => char.IsDigit(c)).ToArray()));
33 actualObject = ((T[])obj)[index];
34 }
35 else
36 {
37 actualObject = obj as T;
38 }
39 return actualObject;
40 }

42 public static object GetParent(SerializedProperty prop)
43 {
44 var path = prop.propertyPath.Replace(".Array.data[", "[");
45 object obj = prop.serializedObject.targetObject;
46 var elements = path.Split('.');
47 foreach (var element in elements.Take(elements.Length - 1))
48 {
49 if (element.Contains("["))
50 {
51 var elementName = element.Substring(0, element.IndexOf("["));
52 var index = Convert.ToInt32(element.Substring(element.IndexOf("[")).Replace("[", "").Replace("]", ""));
53 obj = GetValue(obj, elementName, index);
54 }
55 else
56 {
57 obj = GetValue(obj, element);
58 }
59 }
60 return obj;
61 }

68 public static int GetIndexOfDrawerObject(SerializedProperty property, bool reportError = true)
69 {
70 string path = property.propertyPath;
71
72 int start = path.IndexOf("[") + 1;
73 int len = path.IndexOf("]") - start;
74
75 if (len < 1)
76 return -1;
77
78 int index = -1;
79 if ((len > 0 && Int32.TryParse(path.Substring(path.IndexOf("[") + 1, len), out index)) == false && reportError)
80 {
81 Debug.Log("Attempted to find the index of a non-array serialized property.");
82 }
83 return index;
84 }

86 private static object GetValue(object source, string name)
87 {
88 if (source == null)
89 return null;
90 var type = source.GetType();
91 var f = type.GetField(name, BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance);
92 if (f == null)
93 {
94 var p = type.GetProperty(name, BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance | BindingFlags.IgnoreCase);
95 if (p == null)
96 return null;
97 return p.GetValue(source, null);
98 }
99 return f.GetValue(source);
100 }
101
102 private static object GetValue(object source, string name, int index)
103 {
104 var enumerable = GetValue(source, name) as IEnumerable;
105 var enm = enumerable.GetEnumerator();
106 while (index-- >= 0)
107 enm.MoveNext();
108 return enm.Current;
109 }
