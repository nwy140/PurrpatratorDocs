109 public static void Recalculate(LiteFloatCompressType compressType, float min, float max, bool accurateCenter, LiteFloatCrusher crusher)
110 {
111 int bits = (int)compressType;
112 crusher.bits = bits;
113
114 float range = max - min;
115 ulong maxcval = (bits == 64) ? ulong.MaxValue : (((ulong)1 << (int)bits) - 1);
116
117 if (accurateCenter && maxcval != 0)
118 maxcval--;
119
120 crusher.encoder = range == 0 ? 0 : maxcval / range;
121 crusher.decoder = maxcval == 0 ? 0 : range / maxcval;
122
123 crusher.maxCVal = maxcval;
124 }

144 public override ulong Encode(float val)
145 {
146 if (compressType == LiteFloatCompressType.Half16)
147 return HalfFloat.HalfUtilities.Pack(val);
148
149 else if (compressType == LiteFloatCompressType.Full32)
150 return ((ByteConverter)val).uint32;
151
154 float encval = ((val - min) * encoder + .5f);
155 if (encval < 0)
156 {
157 if (outOfBoundsHandling == LiteOutOfBoundsHandling.Clamp)
158 return 0;
159 else
160 return (ulong)((long)maxCVal + ((long)encval % 10));
161 }
162
163 if (encval > maxCVal)
164 {
165 if (outOfBoundsHandling == LiteOutOfBoundsHandling.Clamp)
166 return maxCVal;
167 else
168 return (ulong)(encval % maxCVal);
169 }
170
172 ulong cval = (ulong)encval;
173 return (cval > maxCVal) ? maxCVal : cval;
174 }

176 public override float Decode(uint cval)
177 {
178 if (compressType == LiteFloatCompressType.Half16)
179 return HalfFloat.HalfUtilities.Unpack((ushort)cval);
180
181 else if (compressType == LiteFloatCompressType.Full32)
182 return ((ByteConverter)cval).float32;
183
186 if (cval == 0)
187 return min;
188
189 if (cval == maxCVal)
190 return max;
191
192 return (cval * decoder) + min;
193 }
