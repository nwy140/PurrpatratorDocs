176 private void RoomOptionsToOpParameters(Dictionary<byte, object> op, RoomOptions roomOptions, bool usePropertiesKey = false)
177 {
178 if (roomOptions == null)
179 {
180 roomOptions = new RoomOptions();
181 }
182
183 Hashtable gameProperties = new Hashtable();
184 gameProperties[GamePropertyKey.IsOpen] = roomOptions.IsOpen;
185 gameProperties[GamePropertyKey.IsVisible] = roomOptions.IsVisible;
186 gameProperties[GamePropertyKey.PropsListedInLobby] = (roomOptions.CustomRoomPropertiesForLobby == null) ? new string[0] : roomOptions.CustomRoomPropertiesForLobby;
187 gameProperties.MergeStringKeys(roomOptions.CustomRoomProperties);
188 if (roomOptions.MaxPlayers > 0)
189 {
190 gameProperties[GamePropertyKey.MaxPlayers] = roomOptions.MaxPlayers;
191 }
192
193 if (!usePropertiesKey)
194 {
195 op[ParameterCode.GameProperties] = gameProperties; // typically, the key for game props is 248
196 }
197 else
198 {
199 op[ParameterCode.Properties] = gameProperties; // when an op uses 248 as filter, the "create room" props can be set as 251
200 }
201
202
203 int flags = 0; // a new way to send the room options as bitwise-flags
204
205 if (roomOptions.CleanupCacheOnLeave)
206 {
207 op[ParameterCode.CleanupCacheOnLeave] = true; // this defines the server's room settings and logic
208 flags = flags | (int)RoomOptionBit.DeleteCacheOnLeave; // this defines the server's room settings and logic (for servers that support flags)
209 }
210 else
211 {
212 op[ParameterCode.CleanupCacheOnLeave] = false; // this defines the server's room settings and logic
213 gameProperties[GamePropertyKey.CleanupCacheOnLeave] = false; // this is only informational for the clients which join
214 }
215
216 #if SERVERSDK
217 op[ParameterCode.CheckUserOnJoin] = roomOptions.CheckUserOnJoin;
218 if (roomOptions.CheckUserOnJoin)
219 {
220 flags = flags | (int) RoomOptionBit.CheckUserOnJoin;
221 }
222 #else
223 // in PUN v1.88 and PUN 2, CheckUserOnJoin is set by default:
224 flags = flags | (int) RoomOptionBit.CheckUserOnJoin;
225 op[ParameterCode.CheckUserOnJoin] = true;
226 #endif
227
228 if (roomOptions.PlayerTtl > 0 || roomOptions.PlayerTtl == -1)
229 {
230 op[ParameterCode.PlayerTTL] = roomOptions.PlayerTtl; // TURNBASED
231 }
232
233 if (roomOptions.EmptyRoomTtl > 0)
234 {
235 op[ParameterCode.EmptyRoomTTL] = roomOptions.EmptyRoomTtl; //TURNBASED
236 }
237
238 if (roomOptions.SuppressRoomEvents)
239 {
240 flags = flags | (int)RoomOptionBit.SuppressRoomEvents;
241 op[ParameterCode.SuppressRoomEvents] = true;
242 }
243 if (roomOptions.SuppressPlayerInfo)
244 {
245 flags = flags | (int)RoomOptionBit.SuppressPlayerInfo;
246 }
247
248 if (roomOptions.Plugins != null)
249 {
250 op[ParameterCode.Plugins] = roomOptions.Plugins;
251 }
252 if (roomOptions.PublishUserId)
253 {
254 flags = flags | (int)RoomOptionBit.PublishUserId;
255 op[ParameterCode.PublishUserId] = true;
256 }
257 if (roomOptions.DeleteNullProperties)
258 {
259 flags = flags | (int)RoomOptionBit.DeleteNullProps; // this is only settable as flag
260 }
261 if (roomOptions.BroadcastPropsChangeToAll)
262 {
263 flags = flags | (int)RoomOptionBit.BroadcastPropsChangeToAll; // this is only settable as flag
264 }
265
266 op[ParameterCode.RoomOptionFlags] = flags;
267 }
