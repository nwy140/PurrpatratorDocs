39 [RequireComponent(typeof(PhotonView))]
40 public class MoveByKeys : Photon.Pun.MonoBehaviourPun

51 public void Start()
52 {
53 //enabled = photonView.isMine;
54 this.isSprite = (GetComponent<SpriteRenderer>() != null);
55
56 this.body2d = GetComponent<Rigidbody2D>();
57 this.body = GetComponent<Rigidbody>();

61 // Update is called once per frame
62 public void FixedUpdate()
63 {
64 if (!photonView.IsMine)
65 {
66 return;
67 }
68
69 if ((Input.GetAxisRaw("Horizontal") < -0.1f) || (Input.GetAxisRaw("Horizontal") > 0.1f))
70 {
71 transform.position += Vector3.right * (Speed * Time.deltaTime) * Input.GetAxisRaw("Horizontal");
72 }
73
74 // jumping has a simple "cooldown" time but you could also jump in the air
75 if (this.jumpingTime <= 0.0f)
76 {
77 if (this.body != null || this.body2d != null)
78 {
79 // obj has a Rigidbody and can jump (AddForce)
80 if (Input.GetKey(KeyCode.Space))
81 {
82 this.jumpingTime = this.JumpTimeout;
83
84 Vector2 jump = Vector2.up * this.JumpForce;
85 if (this.body2d != null)
86 {
87 this.body2d.AddForce(jump);
88 }
89 else if (this.body != null)
90 {
91 this.body.AddForce(jump);
92 }
93 }
94 }
95 }
96 else
97 {
98 this.jumpingTime -= Time.deltaTime;
99 }
100
101 // 2d objects can't be moved in 3d "forward"
102 if (!this.isSprite)
103 {
104 if ((Input.GetAxisRaw("Vertical") < -0.1f) || (Input.GetAxisRaw("Vertical") > 0.1f))
105 {
106 transform.position += Vector3.forward * (Speed * Time.deltaTime) * Input.GetAxisRaw("Vertical");
107 }
108 }
109 }
110 }
