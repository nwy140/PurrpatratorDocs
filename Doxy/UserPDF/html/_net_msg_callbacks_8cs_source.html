40 public static void OnEvent(EventData photonEvent)
41 {
42 byte msgId = photonEvent.Code;
43
44 if (!callbacks.ContainsKey(msgId))
45 return;
46
47#if PUN_2_19_OR_NEWER
48 ByteArraySlice slice;
49 byte[] buffer;
50 var peer = PhotonNetwork.NetworkingClient.LoadBalancingPeer;
51 bool useSlices = peer.UseByteArraySlicePoolForEvents && peer.SerializationProtocolType != SerializationProtocol.GpBinaryV16;
52 if (useSlices)
53 {
54 slice = photonEvent.CustomData as ByteArraySlice;
55 buffer = slice.Buffer;
56 }
57 else
58 {
59 slice = null;
60 buffer = (photonEvent.CustomData as byte[]);
61 }
62
63#else
64 byte[] buffer = (photonEvent.CustomData as byte[]);
65#endif
66
67 var cbs = callbacks[msgId];
68 if (cbs.bufferCallbacks != null && cbs.bufferCallbacks.Count > 0)
69 {
70 foreach (var cb in cbs.bufferCallbacks)
71 cb(null, photonEvent.Sender, buffer);
72 }
73
74#if PUN_2_19_OR_NEWER
75 // Release our nonalloc slice to the pool.
76 if (useSlices)
77 slice.Release();
78#endif
79 }
