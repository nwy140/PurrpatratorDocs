181 // used to register for various events (post-load)
182 private static void OnDelayCall()
183 {
184 //Debug.Log("OnDelayCall()");
185
186 postInspectorUpdate = true;
187
188 EditorApplication.playModeStateChanged -= PlayModeStateChanged;
189 EditorApplication.playModeStateChanged += PlayModeStateChanged;
190
191 #if UNITY_2021_1_OR_NEWER
192 CompilationPipeline.compilationStarted -= OnCompileStarted21;
193 CompilationPipeline.compilationStarted += OnCompileStarted21;
194 #else
195 CompilationPipeline.assemblyCompilationStarted -= OnCompileStarted;
196 CompilationPipeline.assemblyCompilationStarted += OnCompileStarted;
197 #endif
198
199 #if (UNITY_2018 || UNITY_2018_1_OR_NEWER)
200 EditorApplication.projectChanged -= OnProjectChanged;
201 EditorApplication.projectChanged += OnProjectChanged;
202 #else
203 EditorApplication.projectWindowChanged -= OnProjectChanged;
204 EditorApplication.projectWindowChanged += OnProjectChanged;
205 #endif
206
207
208 if (!EditorApplication.isPlaying && !EditorApplication.isPlayingOrWillChangePlaymode)
209 {
210 OnProjectChanged(); // call this initially from here, as the project change events happened earlier (on start of the Editor)
211 PhotonEditor.UpdateRpcList();
212 }
213 }

217 // called in editor, opens wizard for initial setup, keeps scene PhotonViews up to date and closes connections when compiling (to avoid issues)
218 private static void OnProjectChanged()
219 {
220 PhotonEditorUtils.ProjectChangedWasCalled = true;
221
222
223 // Prevent issues with Unity Cloud Builds where ServerSettings are not found.
224 // Also, within the context of a Unity Cloud Build, ServerSettings is already present anyway.
225 #if UNITY_CLOUD_BUILD
226 return;
227 #endif
228
229 if (PhotonNetwork.PhotonServerSettings == null || PhotonNetwork.PhotonServerSettings.AppSettings == null || string.IsNullOrEmpty(PhotonNetwork.PhotonServerSettings.AppSettings.AppIdRealtime))
230 {
231 PhotonNetwork.LoadOrCreateSettings(true);
232 }
233
234 if (PhotonNetwork.PhotonServerSettings == null)
235 {
236 // the PhotonServerSettings are loaded or created. If both fails, the Editor should probably not run (anymore).
237 return;
238 }
239
240 PunSceneSettings.SanitizeSceneSettings();
241
242
243 // serverSetting is null when the file gets deleted. otherwise, the wizard should only run once and only if hosting option is not (yet) set
244 if (!PhotonNetwork.PhotonServerSettings.DisableAutoOpenWizard)
245 {
246 ShowRegistrationWizard();
247 PhotonNetwork.PhotonServerSettings.DisableAutoOpenWizard = true;
248 PhotonEditor.SaveSettings();
249 }
250 }

260 private static void OnCompileStarted(string obj)
261 {
262 if (PhotonNetwork.IsConnected)
263 {
264 // log warning, unless there was one recently
265 if (EditorApplication.timeSinceStartup - lastWarning > 3)
266 {
267 Debug.LogWarning(CurrentLang.WarningPhotonDisconnect);
268 lastWarning = EditorApplication.timeSinceStartup;
269 }
270
271 PhotonNetwork.Disconnect();
272 PhotonNetwork.NetworkingClient.LoadBalancingPeer.DispatchIncomingCommands();
273 #if UNITY_2019_4_OR_NEWER && UNITY_EDITOR
274 EditorApplication.ExitPlaymode();
275 #endif
276 }

617 private AccountService serviceClient;
618 protected virtual void RegisterWithEmail(string email)
619 {
620 List<ServiceTypes> types = new List<ServiceTypes>();
621 types.Add(ServiceTypes.Pun);
622 if (PhotonEditorUtils.HasChat)
623 {
624 types.Add(ServiceTypes.Chat);
625 }
626 if (PhotonEditorUtils.HasVoice)
627 {
628 types.Add(ServiceTypes.Voice);
629 }
630
631
632 if (this.serviceClient == null)
633 {
634 this.serviceClient = new AccountService();
635 this.serviceClient.CustomToken = CustomToken;
636 this.serviceClient.CustomContext = CustomContext;
637 }
638 else
639 {
640 // while RegisterByEmail will check RequestPendingResult below, it would also display an error message. no needed in this case
641 if (this.serviceClient.RequestPendingResult)
642 {
643 Debug.LogWarning("Registration request is pending a response. Please wait.");
644 return;
645 }
646 }
647
648 this.emailSentToAccount = email;
649 this.emailSentToAccountIsRegistered = false;
650
651 if (this.serviceClient.RegisterByEmail(email, types, RegisterWithEmailSuccessCallback, RegisterWithEmailErrorCallback, "PUN"+PhotonNetwork.PunVersion))
652 {
653 this.photonSetupState = PhotonSetupStates.EmailRegistrationPending;
654 EditorUtility.DisplayProgressBar(CurrentLang.ConnectionTitle, CurrentLang.ConnectionInfo, 0.5f);
655 }
656 else
657 {
658 this.DisplayErrorMessage("Email registration request could not be sent. Retry again or check error logs and contact support.");
659 }
660 }
661
662 private void RegisterWithEmailSuccessCallback(AccountServiceResponse res)
663 {
664 EditorUtility.ClearProgressBar();
665 this.emailSentToAccountIsRegistered = true; // email is either registered now, or was already
666
667 if (res.ReturnCode == AccountServiceReturnCodes.Success)
668 {
669 string key = ((int) ServiceTypes.Pun).ToString();
670 string appId;
671 if (res.ApplicationIds.TryGetValue(key, out appId))
672 {
673 this.mailOrAppId = appId;
674 PhotonNetwork.PhotonServerSettings.UseCloud(this.mailOrAppId, null);
675 key = ((int) ServiceTypes.Chat).ToString();
676 if (res.ApplicationIds.TryGetValue(key, out appId))
677 {
678 PhotonNetwork.PhotonServerSettings.AppSettings.AppIdChat = appId;
679 }
680 else if (PhotonEditorUtils.HasChat)
681 {
682 Debug.LogWarning("Registration successful but no Chat AppId returned");
683 }
684 key = ((int) ServiceTypes.Voice).ToString();
685 if (res.ApplicationIds.TryGetValue(key, out appId))
686 {
687 PhotonNetwork.PhotonServerSettings.AppSettings.AppIdVoice = appId;
688 }
689 else if (PhotonEditorUtils.HasVoice)
690 {
691 Debug.LogWarning("Registration successful but no Voice AppId returned");
692 }
693 PhotonEditor.SaveSettings();
694 this.photonSetupState = PhotonSetupStates.GoEditPhotonServerSettings;
695 }
696 else
697 {
698 DisplayErrorMessage("Registration successful but no PUN AppId returned");
699 }
700 }
701 else
702 {
703 PhotonEditor.SaveSettings();
704
705 if (res.ReturnCode == AccountServiceReturnCodes.EmailAlreadyRegistered)
706 {
707 this.photonSetupState = PhotonSetupStates.EmailAlreadyRegistered;
708 }
709 else
710 {
711 DisplayErrorMessage(res.Message);
712 }
713 }
714 }

746 public static void UpdateRpcList()
747 {
748 //Debug.Log("UpdateRpcList()");
749
750 if (PhotonNetwork.PhotonServerSettings == null)
751 {
752 Debug.LogWarning("UpdateRpcList() wasn not able to access the PhotonServerSettings. Not updating the RPCs.");
753 return;
754 }
755
756
757 // check all "script assemblies" for methods with PunRPC attribute
758 List<string> additionalRpcs = new List<string>(); // not yet listed rpc-method names go here
759 List<string> allRpcs = new List<string>();
760
761
762 #if UNITY_2019_2_OR_NEWER
763
764 // we can make use of the new TypeCache to find methods with PunRPC attribute
765 var extractedMethods = TypeCache.GetMethodsWithAttribute<PunRPC>();
766 foreach (var methodInfo in extractedMethods)
767 {
768 allRpcs.Add(methodInfo.Name);
769 if (!PhotonNetwork.PhotonServerSettings.RpcList.Contains(methodInfo.Name) && !additionalRpcs.Contains(methodInfo.Name))
770 {
771 additionalRpcs.Add(methodInfo.Name);
772 }
773 }
774
775 #else
776
777 System.Reflection.Assembly[] assemblies = System.AppDomain.CurrentDomain.GetAssemblies().Where(a => !(a.ManifestModule is System.Reflection.Emit.ModuleBuilder)).ToArray();
778
779 foreach (var assembly in assemblies)
780 {
781 if (!assembly.Location.Contains("ScriptAssemblies") || assembly.FullName.StartsWith("Assembly-CSharp-Editor"))
782 {
783 continue;
784 }
785
786 var types = assembly.GetTypes().Where(t => t.IsSubclassOf(typeof(MonoBehaviour)));
787 var methodInfos = types.SelectMany(t => t.GetMethods(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance));
788 var methodNames = methodInfos.Where(m => m.IsDefined(typeof(PunRPC), false)).Select(mi => mi.Name).ToArray();
789 var additional = methodNames.Where(n => !PhotonNetwork.PhotonServerSettings.RpcList.Contains(n) && !additionalRpcs.Contains(n));
790
791 allRpcs.AddRange(methodNames);
792 additionalRpcs.AddRange(additional);
793 }
794
795 #endif
796
797
798 if (additionalRpcs.Count <= 0)
799 {
800 //Debug.Log("UpdateRPCs did not found new.");
801 return;
802 }
803
804
805 if (additionalRpcs.Count + PhotonNetwork.PhotonServerSettings.RpcList.Count >= byte.MaxValue)
806 {
807 if (allRpcs.Count <= byte.MaxValue)
808 {
809 bool clearList = EditorUtility.DisplayDialog(CurrentLang.IncorrectRPCListTitle, CurrentLang.IncorrectRPCListLabel, CurrentLang.RemoveOutdatedRPCsLabel, CurrentLang.CancelButton);
810 if (clearList)
811 {
812 PhotonNetwork.PhotonServerSettings.RpcList.Clear();
813 additionalRpcs = allRpcs.Distinct().ToList(); // we add all unique names
814 }
815 else
816 {
817 return;
818 }
819 }
820 else
821 {
822 EditorUtility.DisplayDialog(CurrentLang.FullRPCListTitle, CurrentLang.FullRPCListLabel, CurrentLang.SkipRPCListUpdateLabel);
823 return;
824 }
825 }
826
827
828 additionalRpcs.Sort();
829 Undo.RecordObject(PhotonNetwork.PhotonServerSettings, "RPC-list update of PUN.");
830 PhotonNetwork.PhotonServerSettings.RpcList.AddRange(additionalRpcs);
831 EditorUtility.SetDirty(PhotonNetwork.PhotonServerSettings);
832
833 //Debug.Log("Updated RPCs. Added: "+additionalRpcs.Count);
834 }
