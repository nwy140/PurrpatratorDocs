361 internal static void ExecuteRpc(Hashtable rpcData, Player sender)
362 {
363 if (rpcData == null || !rpcData.ContainsKey(keyByteZero))
364 {
365 Debug.LogError("Malformed RPC; this should never occur. Content: " + SupportClassPun.DictionaryToString(rpcData));
366 return;
367 }
368
369 // ts: updated with "flat" event data
370 int netViewID = (int)rpcData[keyByteZero]; // LIMITS PHOTONVIEWS&PLAYERS
371 int otherSidePrefix = 0; // by default, the prefix is 0 (and this is not being sent)
372 if (rpcData.ContainsKey(keyByteOne))
373 {
374 otherSidePrefix = (short)rpcData[keyByteOne];
375 }
376
377
378 string inMethodName;
379 if (rpcData.ContainsKey(keyByteFive))
380 {
381 int rpcIndex = (byte)rpcData[keyByteFive]; // LIMITS RPC COUNT
382 if (rpcIndex > PhotonNetwork.PhotonServerSettings.RpcList.Count - 1)
383 {
384 Debug.LogError("Could not find RPC with index: " + rpcIndex + ". Going to ignore! Check PhotonServerSettings.RpcList");
385 return;
386 }
387 else
388 {
389 inMethodName = PhotonNetwork.PhotonServerSettings.RpcList[rpcIndex];
390 }
391 }
392 else
393 {
394 inMethodName = (string)rpcData[keyByteThree];
395 }
396
397 object[] arguments = null;
398 if (rpcData.ContainsKey(keyByteFour))
399 {
400 arguments = (object[])rpcData[keyByteFour];
401 }
402
403 PhotonView photonNetview = GetPhotonView(netViewID);
404 if (photonNetview == null)
405 {
406 int viewOwnerId = netViewID / PhotonNetwork.MAX_VIEW_IDS;
407 bool owningPv = (viewOwnerId == NetworkingClient.LocalPlayer.ActorNumber);
408 bool ownerSent = sender != null && viewOwnerId == sender.ActorNumber;
409
410 if (owningPv)
411 {
412 Debug.LogWarning("Received RPC \"" + inMethodName + "\" for viewID " + netViewID + " but this PhotonView does not exist! View was/is ours." + (ownerSent ? " Owner called." : " Remote called.") + " By: " + sender);
413 }
414 else
415 {
416 Debug.LogWarning("Received RPC \"" + inMethodName + "\" for viewID " + netViewID + " but this PhotonView does not exist! Was remote PV." + (ownerSent ? " Owner called." : " Remote called.") + " By: " + sender + " Maybe GO was destroyed but RPC not cleaned up.");
417 }
418 return;
419 }
420
421 if (photonNetview.Prefix != otherSidePrefix)
422 {
423 Debug.LogError("Received RPC \"" + inMethodName + "\" on viewID " + netViewID + " with a prefix of " + otherSidePrefix + ", our prefix is " + photonNetview.Prefix + ". The RPC has been ignored.");
424 return;
425 }
426
427 // Get method name
428 if (string.IsNullOrEmpty(inMethodName))
429 {
430 Debug.LogError("Malformed RPC; this should never occur. Content: " + SupportClassPun.DictionaryToString(rpcData));
431 return;
432 }
433
434 if (PhotonNetwork.LogLevel >= PunLogLevel.Full)
435 {
436 Debug.Log("Received RPC: " + inMethodName);
437 }
438
439
440 // SetReceiving filtering
441 if (photonNetview.Group != 0 && !allowedReceivingGroups.Contains(photonNetview.Group))
442 {
443 return; // Ignore group
444 }
445
446 Type[] argumentsTypes = null;
447 if (arguments != null && arguments.Length > 0)
448 {
449 argumentsTypes = new Type[arguments.Length];
450 int i = 0;
451 for (int index = 0; index < arguments.Length; index++)
452 {
453 object objX = arguments[index];
454 if (objX == null)
455 {
456 argumentsTypes[i] = null;
457 }
458 else
459 {
460 argumentsTypes[i] = objX.GetType();
461 }
462
463 i++;
464 }
465 }
466
467
468 int receivers = 0;
469 int foundMethods = 0;
470 if (!PhotonNetwork.UseRpcMonoBehaviourCache || photonNetview.RpcMonoBehaviours == null || photonNetview.RpcMonoBehaviours.Length == 0)
471 {
472 photonNetview.RefreshRpcMonoBehaviourCache();
473 }
474
475 for (int componentsIndex = 0; componentsIndex < photonNetview.RpcMonoBehaviours.Length; componentsIndex++)
476 {
477 MonoBehaviour monob = photonNetview.RpcMonoBehaviours[componentsIndex];
478 if (monob == null)
479 {
480 Debug.LogError("ERROR You have missing MonoBehaviours on your gameobjects!");
481 continue;
482 }
483
484 Type type = monob.GetType();
485
486 // Get [PunRPC] methods from cache
487 List<MethodInfo> cachedRPCMethods = null;
488 bool methodsOfTypeInCache = monoRPCMethodsCache.TryGetValue(type, out cachedRPCMethods);
489
490 if (!methodsOfTypeInCache)
491 {
492 List<MethodInfo> entries = SupportClassPun.GetMethods(type, typePunRPC);
493
494 monoRPCMethodsCache[type] = entries;
495 cachedRPCMethods = entries;
496 }
497
498 if (cachedRPCMethods == null)
499 {
500 continue;
501 }
502
503 // Check cache for valid methodname+arguments
504 for (int index = 0; index < cachedRPCMethods.Count; index++)
505 {
506 MethodInfo mInfo = cachedRPCMethods[index];
507 if (!mInfo.Name.Equals(inMethodName))
508 {
509 continue;
510 }
511
512 ParameterInfo[] parameters = mInfo.GetCachedParemeters();
513 foundMethods++;
514
515
516 // if we got no arguments:
517 if (arguments == null)
518 {
519 if (parameters.Length == 0)
520 {
521 receivers++;
522 object o = mInfo.Invoke((object)monob, null);
523 if (PhotonNetwork.RunRpcCoroutines)
524 {
525 IEnumerator ie = null;//o as IEnumerator;
526 if ((ie = o as IEnumerator) != null)
527 {
528 PhotonHandler.Instance.StartCoroutine(ie);
529 }
530 }
531 }
532 else if (parameters.Length == 1 && parameters[0].ParameterType == typeof(PhotonMessageInfo))
533 {
534 int sendTime = (int)rpcData[keyByteTwo];
535
536 receivers++;
537 object o = mInfo.Invoke((object)monob, new object[] { new PhotonMessageInfo(sender, sendTime, photonNetview) });
538 if (PhotonNetwork.RunRpcCoroutines)
539 {
540 IEnumerator ie = null;//o as IEnumerator;
541 if ((ie = o as IEnumerator) != null)
542 {
543 PhotonHandler.Instance.StartCoroutine(ie);
544 }
545 }
546 }
547 continue;
548 }
549
550
551 // if there are any arguments (in the incoming call check if the method is compatible
552 if (parameters.Length == arguments.Length)
553 {
554 // Normal, PhotonNetworkMessage left out
555 if (CheckTypeMatch(parameters, argumentsTypes))
556 {
557 receivers++;
558 object o = mInfo.Invoke((object)monob, arguments);
559 if (PhotonNetwork.RunRpcCoroutines)
560 {
561 IEnumerator ie = null;//o as IEnumerator;
562 if ((ie = o as IEnumerator) != null)
563 {
564 PhotonHandler.Instance.StartCoroutine(ie);
565 }
566 }
567 }
568 continue;
569 }
570
571 if (parameters.Length == arguments.Length + 1)
572 {
573 // Check for PhotonNetworkMessage being the last
574 if (parameters[parameters.Length - 1].ParameterType == typeof(PhotonMessageInfo) && CheckTypeMatch(parameters, argumentsTypes))
575 {
576 int sendTime = (int)rpcData[keyByteTwo];
577 object[] argumentsWithInfo = new object[arguments.Length + 1];
578 arguments.CopyTo(argumentsWithInfo, 0);
579 argumentsWithInfo[argumentsWithInfo.Length - 1] = new PhotonMessageInfo(sender, sendTime, photonNetview);
580
581 receivers++;
582 object o = mInfo.Invoke((object)monob, argumentsWithInfo);
583 if (PhotonNetwork.RunRpcCoroutines)
584 {
585 IEnumerator ie = null;//o as IEnumerator;
586 if ((ie = o as IEnumerator) != null)
587 {
588 PhotonHandler.Instance.StartCoroutine(ie);
589 }
590 }
591 }
592 continue;
593 }
594
595 if (parameters.Length == 1 && parameters[0].ParameterType.IsArray)
596 {
597 receivers++;
598 object o = mInfo.Invoke((object)monob, new object[] { arguments });
599 if (PhotonNetwork.RunRpcCoroutines)
600 {
601 IEnumerator ie = null;//o as IEnumerator;
602 if ((ie = o as IEnumerator) != null)
603 {
604 PhotonHandler.Instance.StartCoroutine(ie);
605 }
606 }
607 continue;
608 }
609 }
610 }
611
612 // Error handling
613 if (receivers != 1)
614 {
615 string argsString = string.Empty;
616 int argsLength = 0;
617 if (argumentsTypes != null)
618 {
619 argsLength = argumentsTypes.Length;
620 for (int index = 0; index < argumentsTypes.Length; index++)
621 {
622 Type ty = argumentsTypes[index];
623 if (argsString != string.Empty)
624 {
625 argsString += ", ";
626 }
627
628 if (ty == null)
629 {
630 argsString += "null";
631 }
632 else
633 {
634 argsString += ty.Name;
635 }
636 }
637 }
638
639 GameObject context = photonNetview != null ? photonNetview.gameObject : null;
640 if (receivers == 0)
641 {
642 if (foundMethods == 0)
643 {
644 // found no method that matches
645 Debug.LogErrorFormat(context, "RPC method '{0}({2})' not found on object with PhotonView {1}. Implement as non-static. Apply [PunRPC]. Components on children are not found. " +
646 "Return type must be void or IEnumerator (if you enable RunRpcCoroutines). RPCs are a one-way message.", inMethodName, netViewID, argsString);
647 }
648 else
649 {
650 // found a method but not the right arguments
651 Debug.LogErrorFormat(context, "RPC method '{0}' found on object with PhotonView {1} but has wrong parameters. Implement as '{0}({2})'. PhotonMessageInfo is optional as final parameter." +
652 "Return type must be void or IEnumerator (if you enable RunRpcCoroutines).", inMethodName, netViewID, argsString);
653 }
654 }
655 else
656 {
657 // multiple components have the same method
658 Debug.LogErrorFormat(context, "RPC method '{0}({2})' found {3}x on object with PhotonView {1}. Only one component should implement it." +
659 "Return type must be void or IEnumerator (if you enable RunRpcCoroutines).", inMethodName, netViewID, argsString, foundMethods);
660 }
661 }

670 private static bool CheckTypeMatch(ParameterInfo[] methodParameters, Type[] callParameterTypes)
671 {
672 if (methodParameters.Length < callParameterTypes.Length)
673 {
674 return false;
675 }
676
677 for (int index = 0; index < callParameterTypes.Length; index++)
678 {
679#if NETFX_CORE
680 TypeInfo methodParamTI = methodParameters[index].ParameterType.GetTypeInfo();
681 TypeInfo callParamTI = callParameterTypes[index].GetTypeInfo();
682
683 if (callParameterTypes[index] != null && !methodParamTI.IsAssignableFrom(callParamTI) && !(callParamTI.IsEnum && System.Enum.GetUnderlyingType(methodParamTI.AsType()).GetTypeInfo().IsAssignableFrom(callParamTI)))
684 {
685 return false;
686 }
687#else
688 Type type = methodParameters[index].ParameterType;
689 if (callParameterTypes[index] != null && !type.IsAssignableFrom(callParameterTypes[index]) && !(type.IsEnum && System.Enum.GetUnderlyingType(type).IsAssignableFrom(callParameterTypes[index])))
690 {
691 return false;
692 }
693#endif
694 }
695
696 return true;
697 }

781 internal static void RemoveInstantiatedGO(GameObject go, bool localOnly)
782 {
783 // Avoid cleanup if we are quitting.
784 if (ConnectionHandler.AppQuits)
785 return;
786
787 if (go == null)
788 {
789 Debug.LogError("Failed to 'network-remove' GameObject because it's null.");
790 return;
791 }
792
793 // Don't remove the GO if it doesn't have any PhotonView
794 go.GetComponentsInChildren<PhotonView>(true, foundPVs);
795 if (foundPVs.Count <= 0)
796 {
797 Debug.LogError("Failed to 'network-remove' GameObject because has no PhotonView components: " + go);
798 return;
799 }
800
801 PhotonView viewZero = foundPVs[0];
802
803 // Don't remove GOs that are owned by others (unless this is the master and the remote player left)
804 if (!localOnly)
805 {
806 //Debug.LogWarning("Destroy " + instantiationId + " creator " + creatorId, go);
807 if (!viewZero.IsMine)
808 {
809 Debug.LogError("Failed to 'network-remove' GameObject. Client is neither owner nor MasterClient taking over for owner who left: " + viewZero);
810 foundPVs.Clear(); // as foundPVs is re-used, clean it to avoid lingering references
811 return;
812 }
813 }
814
815 // cleanup instantiation (event and local list)
816 if (!localOnly)
817 {
818 ServerCleanInstantiateAndDestroy(viewZero); // server cleaning
819 }
820
821 int creatorActorNr = viewZero.CreatorActorNr;
822
823 // cleanup PhotonViews and their RPCs events (if not localOnly)
824 for (int j = foundPVs.Count - 1; j >= 0; j--)
825 {
826 PhotonView view = foundPVs[j];
827 if (view == null)
828 {
829 continue;
830 }
831
832 // TODO: Probably should have a enum that defines when auto-detachment should occur.
833 // Check nested PVs for different creator. Detach if different, to avoid destroying reparanted objects.
834 if (j != 0)
835 {
836 // view does not belong to the same object as the root PV - unparent this nested PV to avoid destruction.
837 if (view.CreatorActorNr != creatorActorNr)
838 {
839 view.transform.SetParent(null, true);
840 continue;
841 }
842 }
843
844 // Notify all children PVs of impending destruction. Send the root PV (the actual object getting destroyed) to the callbacks.
845 view.OnPreNetDestroy(viewZero);
846
847 // we only destroy/clean PhotonViews that were created by PhotonNetwork.Instantiate (and those have an instantiationId!)
848 if (view.InstantiationId >= 1)
849 {
850 LocalCleanPhotonView(view);
851 }
852 if (!localOnly)
853 {
854 OpCleanRpcBuffer(view);
855 }
856 }
857
858 if (PhotonNetwork.LogLevel >= PunLogLevel.Full)
859 {
860 Debug.Log("Network destroy Instantiated GO: " + go.name);
861 }
862
863 foundPVs.Clear(); // as foundPVs is re-used, clean it to avoid lingering references
864
865 go.SetActive(false); // PUN 2 disables objects before the return to the pool
866 prefabPool.Destroy(go); // PUN 2 always uses a PrefabPool (even for the default implementation)
867 }

1606 internal static void RunViewUpdate()
1607 {
1608 if (PhotonNetwork.OfflineMode || CurrentRoom == null || CurrentRoom.Players == null)
1609 {
1610 return;
1611 }
1612
1613
1614 // no need to send OnSerialize messages while being alone (these are not buffered anyway)
1615#if !PHOTON_DEVELOP
1616 if (CurrentRoom.Players.Count <= 1)
1617 {
1618 return;
1619 }
1620#else
1621 serializeRaiseEvOptions.Receivers = (CurrentRoom.Players.Count == 1) ? ReceiverGroup.All : ReceiverGroup.Others;
1622#endif
1623
1624
1625
1626 /* Format of the event's data object[]:
1627 * [0] = PhotonNetwork.ServerTimestamp;
1628 * [1] = currentLevelPrefix; OPTIONAL!
1629 * [2] = object[] of PhotonView x
1630 * [3] = object[] of PhotonView y or NULL
1631 * [...]
1632 *
1633 * We only combine updates for XY objects into one RaiseEvent to avoid fragmentation.
1634 * The Reliability and Interest Group are only used for RaiseEvent and not contained in the event/data that reaches the other clients.
1635 * This is read in OnEvent().
1636 */
1637
1638
1639 var enumerator = photonViewList.GetEnumerator(); // replacing foreach (PhotonView view in this.photonViewList.Values) for memory allocation improvement
1640 while (enumerator.MoveNext())
1641 {
1642 PhotonView view = enumerator.Current.Value;
1643
1644 // a client only sends updates for active, synchronized PhotonViews that are under it's control (isMine)
1645 if (view.Synchronization == ViewSynchronization.Off || view.IsMine == false || view.isActiveAndEnabled == false)
1646 {
1647 continue;
1648 }
1649
1650 if (blockedSendingGroups.Contains(view.Group))
1651 {
1652 continue; // Block sending on this group
1653 }
1654
1655
1656 // call the PhotonView's serialize method(s)
1657 List<object> evData = OnSerializeWrite(view);
1658 if (evData == null)
1659 {
1660 continue;
1661 }
1662
1663 RaiseEventBatch eventBatch = new RaiseEventBatch();
1664 eventBatch.Reliable = view.Synchronization == ViewSynchronization.ReliableDeltaCompressed || view.mixedModeIsReliable;
1665 eventBatch.Group = view.Group;
1666
1667 SerializeViewBatch svBatch = null;
1668 bool found = serializeViewBatches.TryGetValue(eventBatch, out svBatch);
1669 if (!found)
1670 {
1671 svBatch = new SerializeViewBatch(eventBatch, 2); // NOTE: the 2 first entries are kept empty for timestamp and level prefix
1672 serializeViewBatches.Add(eventBatch, svBatch);
1673 }
1674
1675 svBatch.Add(evData);
1676 if (svBatch.ObjectUpdates.Count == svBatch.ObjectUpdates.Capacity)
1677 {
1678 SendSerializeViewBatch(svBatch);
1679 }

1881 private static List<object> DeltaCompressionWrite(List<object> previousContent, List<object> currentContent)
1882 {
1883 if (currentContent == null || previousContent == null || previousContent.Count != currentContent.Count)
1884 {
1885 return currentContent; // the current data needs to be sent (which might be null)
1886 }
1887
1888 if (currentContent.Count <= SyncFirstValue)
1889 {
1890 return null; // this send doesn't contain values (except the "headers"), so it's not being sent
1891 }
1892

1894 compressedContent[SyncCompressed] = false;
1895 int compressedValues = 0;
1896
1897 Queue<int> valuesThatAreChangedToNull = null;
1898 for (int index = SyncFirstValue; index < currentContent.Count; index++)
1899 {
1900 object newObj = currentContent[index];
1901 object oldObj = previousContent[index];
1902 if (AlmostEquals(newObj, oldObj))
1903 {
1904 // compress (by using null, instead of value, which is same as before)
1905 compressedValues++;
1906 compressedContent[index] = null;
1907 }
1908 else
1909 {
1910 compressedContent[index] = newObj;
1911
1912 // value changed, we don't replace it with null
1913 // new value is null (like a compressed value): we have to mark it so it STAYS null instead of being replaced with previous value
1914 if (newObj == null)
1915 {
1916 if (valuesThatAreChangedToNull == null)
1917 {
1918 valuesThatAreChangedToNull = new Queue<int>(currentContent.Count);
1919 }
1920 valuesThatAreChangedToNull.Enqueue(index);
1921 }
1922 }
1923 }
1924

1931 return null;
1932 }
1933

1936 {

1938 }
1939 }
