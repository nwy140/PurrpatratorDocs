17 [HelpURL("https://doc.photonengine.com/en-us/pun/v2/gameplay/synchronization-and-state")]

57 public void Update()
58 {
59 var tr = transform;
60
61 if (!this.photonView.IsMine)
62 {
63 if (m_UseLocal)
64
65 {
66 tr.localPosition = Vector3.MoveTowards(tr.localPosition, this.m_NetworkPosition, this.m_Distance * Time.deltaTime * PhotonNetwork.SerializationRate);
67 tr.localRotation = Quaternion.RotateTowards(tr.localRotation, this.m_NetworkRotation, this.m_Angle * Time.deltaTime * PhotonNetwork.SerializationRate);
68 }
69 else
70 {
71 tr.position = Vector3.MoveTowards(tr.position, this.m_NetworkPosition, this.m_Distance * Time.deltaTime * PhotonNetwork.SerializationRate);
72 tr.rotation = Quaternion.RotateTowards(tr.rotation, this.m_NetworkRotation, this.m_Angle * Time.deltaTime * PhotonNetwork.SerializationRate);
73 }
74 }

77 public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
78 {
79 var tr = transform;
80
81 // Write
82 if (stream.IsWriting)
83 {
84 if (this.m_SynchronizePosition)
85 {
86 if (m_UseLocal)
87 {
88 this.m_Direction = tr.localPosition - this.m_StoredPosition;
89 this.m_StoredPosition = tr.localPosition;
90 stream.SendNext(tr.localPosition);
91 stream.SendNext(this.m_Direction);
92 }
93 else
94 {
95 this.m_Direction = tr.position - this.m_StoredPosition;
96 this.m_StoredPosition = tr.position;
97 stream.SendNext(tr.position);
98 stream.SendNext(this.m_Direction);
99 }
100 }
101
102 if (this.m_SynchronizeRotation)
103 {
104 if (m_UseLocal)
105 {
106 stream.SendNext(tr.localRotation);
107 }
108 else
109 {
110 stream.SendNext(tr.rotation);
111 }
112 }
113
114 if (this.m_SynchronizeScale)
115 {
116 stream.SendNext(tr.localScale);
117 }
118 }
119 // Read
120 else
121 {
122 if (this.m_SynchronizePosition)
123 {
124 this.m_NetworkPosition = (Vector3)stream.ReceiveNext();
125 this.m_Direction = (Vector3)stream.ReceiveNext();
126
127 if (m_firstTake)
128 {
129 if (m_UseLocal)
130 tr.localPosition = this.m_NetworkPosition;
131 else
132 tr.position = this.m_NetworkPosition;
133
134 this.m_Distance = 0f;
135 }
136 else
137 {
138 float lag = Mathf.Abs((float)(PhotonNetwork.Time - info.SentServerTime));
139 this.m_NetworkPosition += this.m_Direction * lag;
140 if (m_UseLocal)
141 {
142 this.m_Distance = Vector3.Distance(tr.localPosition, this.m_NetworkPosition);
143 }
144 else
145 {
146 this.m_Distance = Vector3.Distance(tr.position, this.m_NetworkPosition);
147 }
148 }
149
150 }
151
152 if (this.m_SynchronizeRotation)
153 {
154 this.m_NetworkRotation = (Quaternion)stream.ReceiveNext();
155
156 if (m_firstTake)
157 {
158 this.m_Angle = 0f;
159
160 if (m_UseLocal)
161 {
162 tr.localRotation = this.m_NetworkRotation;
163 }
164 else
165 {
166 tr.rotation = this.m_NetworkRotation;
167 }
168 }
169 else
170 {
171 if (m_UseLocal)
172 {
173 this.m_Angle = Quaternion.Angle(tr.localRotation, this.m_NetworkRotation);
174 }
175 else
176 {
177 this.m_Angle = Quaternion.Angle(tr.rotation, this.m_NetworkRotation);
178 }
179 }
180 }
181
182 if (this.m_SynchronizeScale)
183 {
184 tr.localScale = (Vector3)stream.ReceiveNext();
185 }
186
187 if (m_firstTake)
188 {
189 m_firstTake = false;
190 }
191 }
192 }
193 }
