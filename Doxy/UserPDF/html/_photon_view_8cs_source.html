210 public int OwnerActorNr
211 {
212 get { return this.ownerActorNr; }
213 set
214 {
215 if (value != 0 && this.ownerActorNr == value)
216 {
217 return;
218 }
219
220 Player prevOwner = this.Owner;
221
222 this.Owner = PhotonNetwork.CurrentRoom == null ? null : PhotonNetwork.CurrentRoom.GetPlayer(value, true);
223 this.ownerActorNr = this.Owner != null ? this.Owner.ActorNumber : value;
224
225 this.AmOwner = PhotonNetwork.LocalPlayer != null && this.ownerActorNr == PhotonNetwork.LocalPlayer.ActorNumber;
226
227 this.UpdateCallbackLists();
228 if (!ReferenceEquals(this.OnOwnerChangeCallbacks, null))
229 {
230 for (int i = 0, cnt = this.OnOwnerChangeCallbacks.Count; i < cnt; ++i)
231 {
232 this.OnOwnerChangeCallbacks[i].OnOwnerChange(this.Owner, prevOwner);
233 }
234 }
235 }

244 get { return this.controllerActorNr; }

704 private struct CallbackTargetChange
705 {
706 public IPhotonViewCallback obj;
707 public Type type;
708 public bool add;
709
710 public CallbackTargetChange(IPhotonViewCallback obj, Type type, bool add)
711 {
712 this.obj = obj;
713 this.type = type;
714 this.add = add;
715 }
716 }
717
718 private Queue<CallbackTargetChange> CallbackChangeQueue = new Queue<CallbackTargetChange>();
719
720 private List<IOnPhotonViewPreNetDestroy> OnPreNetDestroyCallbacks;
721 private List<IOnPhotonViewOwnerChange> OnOwnerChangeCallbacks;
722 private List<IOnPhotonViewControllerChange> OnControllerChangeCallbacks;
723
728 public void AddCallbackTarget(IPhotonViewCallback obj)
729 {
730 CallbackChangeQueue.Enqueue(new CallbackTargetChange(obj, null, true));
731 }
732
737 public void RemoveCallbackTarget(IPhotonViewCallback obj)
738 {
739 CallbackChangeQueue.Enqueue(new CallbackTargetChange(obj, null, false));
740 }
741
747 public void AddCallback<T>(IPhotonViewCallback obj) where T : class, IPhotonViewCallback
748 {
749 CallbackChangeQueue.Enqueue(new CallbackTargetChange(obj, typeof(T), true));
750 }
751
757 public void RemoveCallback<T>(IPhotonViewCallback obj) where T : class, IPhotonViewCallback
758 {
759 CallbackChangeQueue.Enqueue(new CallbackTargetChange(obj, typeof(T), false));
760 }
761
765 private void UpdateCallbackLists()
766 {
767 while (CallbackChangeQueue.Count > 0)
768 {
769 var item = CallbackChangeQueue.Dequeue();
770 var obj = item.obj;
771 var type = item.type;
772 var add = item.add;
773
774 if (type == null)
775 {
776 TryRegisterCallback(obj, ref OnPreNetDestroyCallbacks, add);
777 TryRegisterCallback(obj, ref OnOwnerChangeCallbacks, add);
778 TryRegisterCallback(obj, ref OnControllerChangeCallbacks, add);
779 }
780 else if (type == typeof(IOnPhotonViewPreNetDestroy))
781 RegisterCallback(obj as IOnPhotonViewPreNetDestroy, ref OnPreNetDestroyCallbacks, add);
782
783 else if (type == typeof(IOnPhotonViewOwnerChange))
784 RegisterCallback(obj as IOnPhotonViewOwnerChange, ref OnOwnerChangeCallbacks, add);
785
786 else if (type == typeof(IOnPhotonViewControllerChange))
787 RegisterCallback(obj as IOnPhotonViewControllerChange, ref OnControllerChangeCallbacks, add);
788 }
789 }
790
791 private void TryRegisterCallback<T>(IPhotonViewCallback obj, ref List<T> list, bool add) where T : class, IPhotonViewCallback
792 {
793 T iobj = obj as T;
794 if (iobj != null)
795 {
796 RegisterCallback(iobj, ref list, add);
797 }
798 }
799
800 private void RegisterCallback<T>(T obj, ref List<T> list, bool add) where T : class, IPhotonViewCallback
801 {
802 if (ReferenceEquals(list, null))
803 list = new List<T>();
804
805 if (add)
806 {
807 if (!list.Contains(obj))
808 list.Add(obj);
809 }
810 else
811 {
812 if (list.Contains(obj))
813 list.Remove(obj);
814 }
815 }
