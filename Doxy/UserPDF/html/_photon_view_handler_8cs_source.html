35 internal static void OnHierarchyChanged()
36 {
37 // set prefabs to viewID 0 if needed
38 // organize resource PVs in a list per viewID
39
40 // process the lists: if more than one photonView is in a list, we have to resolve the clash
41 // check if only one view had the viewId earlier
42 // apply a new viewID to the others
43
44 // update the cached list of instances and their viewID
45
46
47 //Debug.LogWarning("OnHierarchyChanged(). isPlaying: " + Application.isPlaying);
48 if (Application.isPlaying)
49 {
50 return;
51 }
52
53
54 PhotonView[] photonViewResources = Resources.FindObjectsOfTypeAll<PhotonView>();
55 List<PhotonView> photonViewInstances = new List<PhotonView>();
56 Dictionary<int, List<PhotonView>> viewInstancesPerViewId = new Dictionary<int, List<PhotonView>>();
57 List<PhotonView> photonViewsToReassign = new List<PhotonView>();
58
59 foreach (PhotonView view in photonViewResources)
60 {
61 if (PhotonEditorUtils.IsPrefab(view.gameObject))
62 {
63 // prefabs should use 0 as ViewID and sceneViewId
64 if (view.ViewID != 0 || view.sceneViewId != 0)
65 {
66 view.ViewID = 0;
67 view.sceneViewId = 0;
68 EditorUtility.SetDirty(view);
69 }
70
71 continue; // skip prefabs in further processing
72 }
73
74 photonViewInstances.Add(view);
75
76
77 // assign a new viewID if the viewId is lower than the minimum for this scene
78 if (!IsViewIdOkForScene(view))
79 {
80 photonViewsToReassign.Add(view);
81 continue; // this view definitely gets cleaned up, so it does not count versus duplicates, checked below
82 }
83
84
85 // organize the viewInstances into lists per viewID, so we know duplicate usage
86 if (!viewInstancesPerViewId.ContainsKey(view.sceneViewId))
87 {
88 viewInstancesPerViewId[view.sceneViewId] = new List<PhotonView>();
89 }
90 viewInstancesPerViewId[view.sceneViewId].Add(view);
91 }
92
93 //Debug.Log("PreviousAssignments: "+PunSceneViews.Instance.Views.Count);
94
95 foreach (List<PhotonView> list in viewInstancesPerViewId.Values)
96 {
97 if (list.Count <= 1)
98 {
99 continue; // skip lists with just one entry (the viewID is unique)
100 }
101
102
103 PhotonView previousAssignment = null;
104 bool wasAssigned = PunSceneViews.Instance.Views.TryGetValue(list[0].sceneViewId, out previousAssignment);
105
106 foreach (PhotonView view in list)
107 {
108 if (wasAssigned && view.Equals(previousAssignment))
109 {
110 // previously, we cached the used viewID as assigned to the current view. we don't change this.
111 continue;
112 }
113
114 //Debug.LogWarning("View to reassign due to viewID: "+view, view.gameObject);
115 photonViewsToReassign.Add(view);
116 }
117 }
118
119 int i;
120 foreach (PhotonView view in photonViewsToReassign)
121 {
122 i = MinSceneViewId(view);
123 while (viewInstancesPerViewId.ContainsKey(i))
124 {
125 i++;
126 }
127 view.sceneViewId = i;
128 viewInstancesPerViewId.Add(i, null); // we don't need the lists anymore but we care about getting the viewIDs listed
129 EditorUtility.SetDirty(view);
130 }
131
132
133 // update the "semi persistent" list of viewIDs and their PhotonViews
134 PunSceneViews.Instance.Views.Clear();
135 foreach (PhotonView view in photonViewInstances)
136 {
137 if (PunSceneViews.Instance.Views.ContainsKey(view.sceneViewId))
138 {
139 Debug.LogError("ViewIDs should no longer have duplicates! "+view.sceneViewId, view);
140 continue;
141 }
142
143 PunSceneViews.Instance.Views[view.sceneViewId] = view;
144 }
145
146 //Debug.Log("photonViewsToReassign.Count: "+photonViewsToReassign.Count + " count of viewIDs in use: "+viewInstancesPerViewId.Values.Count);
147 //Debug.Log("PreviousAssignments now counts: "+PunSceneViews.Instance.Views.Count);
148 }

151 private static int MinSceneViewId(PhotonView view)
152 {
153 int result = PunSceneSettings.MinViewIdForScene(view.gameObject.scene.name);
154 return result;
155 }
156
157 private static bool IsViewIdOkForScene(PhotonView view)
158 {
159 return view.sceneViewId >= MinSceneViewId(view);
160 }
