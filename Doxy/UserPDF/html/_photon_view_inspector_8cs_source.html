195 private void EditorFindObservables()
196 {
197 Undo.RecordObject(serializedObject.targetObject, "Find Observables");
198 var property = serializedObject.FindProperty("ObservedComponents");
199
200 // Just doing a Find updates the Observables list, but Unity fails to save that change.
201 // Instead we do the find, and then iterate the found objects into the serialize property, then apply that.
202 property.ClearArray();
203 m_Target.FindObservables(true);
204 for(int i = 0; i < m_Target.ObservedComponents.Count; ++i)
205 {
206 property.InsertArrayElementAtIndex(i);
207 property.GetArrayElementAtIndex(i).objectReferenceValue = m_Target.ObservedComponents[i];
208 }
209 serializedObject.ApplyModifiedProperties();
210 }
