211 public Player GetNextFor(int currentPlayerId)
212 {
213 if (this.RoomReference == null || this.RoomReference.Players == null || this.RoomReference.Players.Count < 2)
214 {
215 return null;
216 }
217
218 Dictionary<int, Player> players = this.RoomReference.Players;
219 int nextHigherId = int.MaxValue; // we look for the next higher ID
220 int lowestId = currentPlayerId; // if we are the player with the highest ID, there is no higher and we return to the lowest player's id
221
222 foreach (int playerid in players.Keys)
223 {
224 if (playerid < lowestId)
225 {
226 lowestId = playerid; // less than any other ID (which must be at least less than this player's id).
227 }
228 else if (playerid > currentPlayerId && playerid < nextHigherId)
229 {
230 nextHigherId = playerid; // more than our ID and less than those found so far.
231 }
232 }
233
234 //UnityEngine.Debug.LogWarning("Debug. " + currentPlayerId + " lower: " + lowestId + " higher: " + nextHigherId + " ");
235 //UnityEngine.Debug.LogWarning(this.RoomReference.GetPlayer(currentPlayerId));
236 //UnityEngine.Debug.LogWarning(this.RoomReference.GetPlayer(lowestId));
237 //if (nextHigherId != int.MaxValue) UnityEngine.Debug.LogWarning(this.RoomReference.GetPlayer(nextHigherId));
238 return (nextHigherId != int.MaxValue) ? players[nextHigherId] : players[lowestId];
239 }

247 protected internal virtual void InternalCacheProperties(Hashtable properties)
248 {
249 if (properties == null || properties.Count == 0 || this.CustomProperties.Equals(properties))
250 {
251 return;
252 }
253
254 if (properties.ContainsKey(ActorProperties.PlayerName))
255 {
256 string nameInServersProperties = (string)properties[ActorProperties.PlayerName];
257 if (nameInServersProperties != null)
258 {
259 if (this.IsLocal)
260 {
261 // the local playername is different than in the properties coming from the server
262 // so the local nickName was changed and the server is outdated -> update server
263 // update property instead of using the outdated nickName coming from server
264 if (!nameInServersProperties.Equals(this.nickName))
265 {
266 this.SetPlayerNameProperty();
267 }
268 }
269 else
270 {
271 this.NickName = nameInServersProperties;
272 }
273 }
274 }
275 if (properties.ContainsKey(ActorProperties.UserId))
276 {
277 this.UserId = (string)properties[ActorProperties.UserId];
278 }
279 if (properties.ContainsKey(ActorProperties.IsInactive))
280 {
281 this.IsInactive = (bool)properties[ActorProperties.IsInactive]; //TURNBASED new well-known propery for players
282 }
283
284 this.CustomProperties.MergeStringKeys(properties);
285 this.CustomProperties.StripKeysWithNullValues();
286 }

390 public bool SetCustomProperties(Hashtable propertiesToSet, Hashtable expectedValues = null, WebFlags webFlags = null)
391 {
392 if (propertiesToSet == null || propertiesToSet.Count == 0)
393 {
394 return false;
395 }
396
397 Hashtable customProps = propertiesToSet.StripToStringKeys() as Hashtable;
398
399 if (this.RoomReference != null)
400 {
401 if (this.RoomReference.IsOffline)
402 {
403 if (customProps.Count == 0)
404 {
405 return false;
406 }
407 this.CustomProperties.Merge(customProps);
408 this.CustomProperties.StripKeysWithNullValues();
409 // invoking callbacks
410 this.RoomReference.LoadBalancingClient.InRoomCallbackTargets.OnPlayerPropertiesUpdate(this, customProps);
411 return true;
412 }
413 else
414 {
415 Hashtable customPropsToCheck = expectedValues.StripToStringKeys() as Hashtable;
416
417 // send (sync) these new values if in online room
418 return this.RoomReference.LoadBalancingClient.OpSetPropertiesOfActor(this.actorNumber, customProps, customPropsToCheck, webFlags);
419 }
420 }
421 if (this.IsLocal)
422 {
423 if (customProps.Count == 0)
424 {
425 return false;
426 }
427 if (expectedValues == null && webFlags == null)
428 {
429 this.CustomProperties.Merge(customProps);
430 this.CustomProperties.StripKeysWithNullValues();
431 return true;
432 }
433 }
434
435 return false;
436 }
