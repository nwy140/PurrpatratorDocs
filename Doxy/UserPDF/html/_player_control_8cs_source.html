82 void IPunObservable.OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
83 {
84 // currently there is no strategy to improve on bandwidth, just passing the current distance and speed is enough,
85 // Input could be passed and then used to better control speed value
86 // Data could be wrapped as a vector2 or vector3 to save a couple of bytes
87 if (stream.IsWriting)
88 {
89 stream.SendNext(this.CurrentDistance);
90 stream.SendNext(this.CurrentSpeed);
91 stream.SendNext(this.m_input);
92 }
93 else
94 {
95 if (this.m_firstTake)
96 {
97 this.m_firstTake = false;
98 }
99
100 this.CurrentDistance = (float) stream.ReceiveNext();
101 this.CurrentSpeed = (float) stream.ReceiveNext();
102 this.m_input = (float) stream.ReceiveNext();
103 }
104 }

115 private void SetupCarOnTrack(int gridStartIndex)
116 {
117 // Setup the SplineWalker to be on the right starting grid position.
118 this.SplineWalker.spline = SlotLanes.Instance.GridPositions[gridStartIndex].Spline;
119 this.SplineWalker.currentDistance = SlotLanes.Instance.GridPositions[gridStartIndex].currentDistance;
120 this.SplineWalker.ExecutePositioning();
121
122 // create a new car
123 this.CarInstance = (GameObject) Instantiate(this.CarPrefabs[gridStartIndex], this.transform.position, this.transform.rotation);
124
125 // We'll wait for the first serializatin to pass, else we'll have a glitch where the car is positioned at the wrong position.
126 if (!this.photonView.IsMine)
127 {
128 this.CarInstance.SetActive(false);
129 }
130
131 // depending on wether we control this instance locally, we force the car to become active ( because when you are alone in the room, serialization doesn't happen, but still we want to allow the user to race around)
132 if (PhotonNetwork.CurrentRoom.PlayerCount == 1)
133 {
134 this.m_firstTake = false;

137 this.CarInstance.transform.SetParent(this.transform);

176 // Update is called once per frame
177 private void Update()
178 {
179 if (this.SplineWalker == null || this.CarInstance == null)
180 {
181 return;
182 }
183
184 if (this.photonView.IsMine)
185 {
186 this.m_input = Input.GetAxis("Vertical");
187 if (this.m_input == 0f)
188 {
189 this.CurrentSpeed -= Time.deltaTime * this.Drag;
190 }
191 else
192 {
193 this.CurrentSpeed += this.m_input;
194 }
195 this.CurrentSpeed = Mathf.Clamp(this.CurrentSpeed, 0f, this.MaximumSpeed);
196 this.SplineWalker.Speed = this.CurrentSpeed;
197
198 this.CurrentDistance = this.SplineWalker.currentDistance;
199 }
200 else
201 {
202 if (this.m_input == 0f)
203 {
204 this.CurrentSpeed -= Time.deltaTime * this.Drag;
205 }
206
207 this.CurrentSpeed = Mathf.Clamp (this.CurrentSpeed, 0f, this.MaximumSpeed);
208 this.SplineWalker.Speed = this.CurrentSpeed;
209
210
211
212 if (this.CurrentDistance != 0 && this.SplineWalker.currentDistance != this.CurrentDistance)
213 {
214 //Debug.Log ("SplineWalker.currentDistance=" + SplineWalker.currentDistance + " CurrentDistance=" + CurrentDistance);
215 this.SplineWalker.Speed += (this.CurrentDistance - this.SplineWalker.currentDistance) * Time.deltaTime * 50f;
216 }
217
218 }
219
220 // Only activate the car if we are sure we have the proper positioning, else it will glitch visually during the initialisation process.
221 if (!this.m_firstTake && !this.CarInstance.activeSelf)
222 {
223 this.CarInstance.SetActive(true);
224 this.SplineWalker.Speed = this.CurrentSpeed;
225 this.SplineWalker.SetPositionOnSpline (this.CurrentDistance);
226
227 }
228 }
