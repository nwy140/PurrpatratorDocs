113 string ParseKey(object key)
114 {
115 if (key.GetType() == typeof(byte))
116 {
117 byte _byteKey = (byte)key;
118
119 switch (_byteKey)
120 {
121 case (byte)255:
122 return "PlayerName";
123 case (byte)254:
124 return "Inactive";
125 case (byte)253:
126 return "UserId";
127 }

129 }

156 public override void OnPlayerPropertiesUpdate(Player target, ExitGames.Client.Photon.Hashtable changedProps)
157 {
158 //Debug.Log("OnPlayerPropertiesUpdate " + target.ActorNumber + " " + target.ToStringFull() + " " + changedProps.ToStringFull());

160 //Debug.Log("_player.ID " + _player.ActorNumber);
