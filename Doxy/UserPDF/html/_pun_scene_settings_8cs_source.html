110 public static int MinViewIdForScene(string sceneName)
111 {
112 if (string.IsNullOrEmpty(sceneName))
113 {
114 return 1;
115 }
116
117 PunSceneSettings pss = Instance;
118 if (pss == null)
119 {
120 Debug.LogError("pss cant be null");
121 return 1;
122 }
123
124 foreach (SceneSetting setting in pss.MinViewIdPerScene)
125 {
126 if (setting.sceneName.Equals(sceneName))
127 {
128 return setting.minViewId;
129 }
130 }
131 return 1;
132 }
