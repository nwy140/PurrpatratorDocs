103 public static void SetPunDemoBuildSettings()
104 {
105 string _PunPath = string.Empty;
106
107 string _thisPath = PhotonNetwork.FindAssetPath ("PunStartUp");
108
109 _thisPath = Application.dataPath + _thisPath.Substring (6); // remove "Assets/"
110
111 //_PunPath = PhotonEditorUtils.GetParent(_thisPath,"Photon");
112
113 if (string.IsNullOrEmpty(_PunPath))
114 {
115 _PunPath = Application.dataPath+"/Photon";
116 }
117
118 // find path of pun guide
119
120 string[] tempPaths = Directory.GetDirectories(_PunPath, "Demos*", SearchOption.AllDirectories);
121 if (tempPaths == null)
122 {
123 return;
124 }
125
126 List<EditorBuildSettingsScene> sceneAr = new List<EditorBuildSettingsScene> ();
127
128 // find scenes of guide
129 foreach (string guidePath in tempPaths)
130 {
131 tempPaths = Directory.GetFiles (guidePath, "*.unity", SearchOption.AllDirectories);
132
133 if (tempPaths == null || tempPaths.Length == 0)
134 {
135 return;
136 }
137
138 // add found guide scenes to build settings
139 for (int i = 0; i < tempPaths.Length; i++)
140 {
141 //Debug.Log(tempPaths[i]);
142 string path = tempPaths [i].Substring (Application.dataPath.Length - "Assets".Length);
143 path = path.Replace ('\\', '/');
144 //Debug.Log(path);
145
146 if (path.Contains ("PUNGuide_M2H") || path.Contains("DemoLoadBalancing"))
147 {
148 continue;
149 }
150
151 // edited to avoid old scene to be included.
152 if (path.Contains ("DemoHub-Scene"))
153 {
154 sceneAr.Insert (0, new EditorBuildSettingsScene (path, true));
155 continue;
156 }
157
158 sceneAr.Add (new EditorBuildSettingsScene (path, true));
159 }
160 }
161
162 EditorBuildSettings.scenes = sceneAr.ToArray();
163 EditorSceneManager.OpenScene(sceneAr[0].path);
164 }
165 }
