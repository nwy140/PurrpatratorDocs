101 [Obsolete("do not call this.")]
102 public void UpdateTeams()
103 {
104 Array enumVals = Enum.GetValues(typeof(Team));
105 foreach (var enumVal in enumVals)
106 {
107 PlayersPerTeam[(Team)enumVal].Clear();
108 }
109
110 for (int i = 0; i < PhotonNetwork.PlayerList.Length; i++)
111 {
112 Player player = PhotonNetwork.PlayerList[i];
113 Team playerTeam = player.GetTeam();
114 PlayersPerTeam[playerTeam].Add(player);
115 }
116 }

124 [Obsolete("Use player.GetPhotonTeam")]
125 public static PunTeams.Team GetTeam(this Player player)
126 {
127 object teamId;
128 if (player.CustomProperties.TryGetValue(PunTeams.TeamPlayerProp, out teamId))
129 {
130 return (PunTeams.Team)teamId;
131 }
132
133 return PunTeams.Team.none;
134 }

140 [Obsolete("Use player.JoinTeam")]
141 public static void SetTeam(this Player player, PunTeams.Team team)
142 {
143 if (!PhotonNetwork.IsConnectedAndReady)
144 {
145 Debug.LogWarning("JoinTeam was called in state: " + PhotonNetwork.NetworkClientState + ". Not IsConnectedAndReady.");
146 return;
147 }
148
149 PunTeams.Team currentTeam = player.GetTeam();
150 if (currentTeam != team)
151 {
152 player.SetCustomProperties(new Hashtable() { { PunTeams.TeamPlayerProp, (byte)team } });
153 }
154 }
