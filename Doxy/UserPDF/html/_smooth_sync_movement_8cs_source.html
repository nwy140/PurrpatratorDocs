11using UnityEngine;

13using Photon.Pun;
14using Photon.Realtime;

16namespace Photon.Pun.UtilityScripts
17{
21 [RequireComponent(typeof(PhotonView))]
22 public class SmoothSyncMovement : Photon.Pun.MonoBehaviourPun, IPunObservable
23 {
24 public float SmoothingDelay = 5;
25 public void Awake()
26 {
27 bool observed = false;
28 foreach (Component observedComponent in this.photonView.ObservedComponents)
29 {
30 if (observedComponent == this)
31 {
32 observed = true;
33 break;
34 }
35 }
36 if (!observed)
37 {
38 Debug.LogWarning(this + " is not observed by this object's photonView! OnPhotonSerializeView() in this class won't be used.");
39 }
40 }
41
42 public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
43 {
44 if (stream.IsWriting)
45 {
46 //We own this player: send the others our data
47 stream.SendNext(transform.position);
48 stream.SendNext(transform.rotation);
49 }
50 else
51 {
52 //Network player, receive data
53 correctPlayerPos = (Vector3)stream.ReceiveNext();
54 correctPlayerRot = (Quaternion)stream.ReceiveNext();
55 }
56 }
57
58 private Vector3 correctPlayerPos = Vector3.zero; //We lerp towards this
59 private Quaternion correctPlayerRot = Quaternion.identity; //We lerp towards this
60
61 public void Update()
62 {
63 if (!photonView.IsMine)
64 {
65 //Update remote player (smooth this, this looks good, at the cost of some accuracy)
66 transform.position = Vector3.Lerp(transform.position, correctPlayerPos, Time.deltaTime * this.SmoothingDelay);
67 transform.rotation = Quaternion.Lerp(transform.rotation, correctPlayerRot, Time.deltaTime * this.SmoothingDelay);
68 }
69 }
70
71 }
72}
