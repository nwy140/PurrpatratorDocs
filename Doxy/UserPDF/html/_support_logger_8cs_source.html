165 private string GetFormattedTimestamp()
166 {
167 if (this.startStopwatch == null)
168 {
169 this.startStopwatch = new Stopwatch();
170 this.startStopwatch.Start();
171 }
172
173 TimeSpan span = this.startStopwatch.Elapsed;
174 if (span.Minutes > 0)
175 {
176 return string.Format("[{0}:{1}.{1}]", span.Minutes, span.Seconds, span.Milliseconds);
177 }
178
179 return string.Format("[{0}.{1}]", span.Seconds, span.Milliseconds);
180 }

185 {

187 {
188 int currentRtt = this.client.LoadBalancingPeer.RoundTripTime;
189 if (currentRtt > this.pingMax)
190 {
191 this.pingMax = currentRtt;
192 }
193 if (currentRtt < this.pingMin)
194 {
195 this.pingMin = currentRtt;
196 }
197 }

283 public void OnConnected()
284 {
285 Debug.Log(this.GetFormattedTimestamp() + " SupportLogger OnConnected().");
286 this.pingMax = 0;
287 this.pingMin = this.client.LoadBalancingPeer.RoundTripTime;
288 this.LogBasics();

293 this.client.LoadBalancingPeer.TrafficStatsEnabled = true;
294 this.StartLogStats();
295 }
296
297 this.StartTrackValues();
