255 public virtual SerializationFlags OnNetSerialize(int frameId, byte[] buffer, ref int bitposition, SerializationFlags writeFlags)
256 {
257
258 //if (!IsMine)
259 //{
260 // /// attached bool
261 // buffer.WriteBool(false, ref bitposition);
262 // return SerializationFlags.None;
263 //}
264
265 Frame frame = frames[frameId];
266
267 SerializationFlags flags;
268
269 var tes = frame.contactRecords;
270
272
273 if (frame.content == FrameContents.Complete)
274 {
275 for (int i = 0, cnt = tes.Count; i < cnt; ++i)
276 {
277 var te = tes[i];
278
279 //Debug.LogError("fid: " + frameId + " Send Pickup " + te.contactSystemViewID + " " + te.contactType + " " + te.contactSystemIndex);
280
282 buffer.Write(1, ref bitposition, 1);
283 buffer.WritePackedBytes((uint)te.contactSystemViewID, ref bitposition, 32);
284 // TODO: this requires a proper bitcount, not 8
285 buffer.WritePackedBits(te.contactSystemIndex, ref bitposition, 8);
286
287 // Convert the mask into an index, since only one bit should ever be true.
288 var contactType = te.contactType;
289 int maskindex = contactType == ContactType.Enter ? 0 : contactType == ContactType.Hitscan ? 3 : contactType == ContactType.Stay ? 1 : 2;
290 buffer.Write((uint)maskindex, ref bitposition, 2);
291 }
292 buffer.Write(0, ref bitposition, 1);
293 flags = SerializationFlags.HasContent | SerializationFlags.ForceReliable;
294 }
295 else
296 {
298 buffer.WriteBool(false, ref bitposition);
299 flags = SerializationFlags.None;
300 }
301
302 return flags;
303 }

305 public SerializationFlags OnNetDeserialize(int originFrameId, byte[] buffer, ref int bitposition, FrameArrival arrival)
306 {
307 Frame frame = frames[originFrameId];
308
309 // Get first content bool to see if we have any contacts at all.
310 bool hasContent = buffer.ReadBool(ref bitposition);
311 if (hasContent)
312 {
314 do
315 {
316 var tes = frame.contactRecords;
317
318 var te = new ContactRecord(
319 (int)buffer.ReadPackedBytes(ref bitposition, 32),
320 (byte)buffer.ReadPackedBits(ref bitposition, 8),
321 (ContactType)(1 << (int)buffer.Read(ref bitposition, 2))
322 );
323
324 //Debug.LogError(name + " DES: " + te);
325 tes.Add(te);
326
327 }
328 // Loop for all contact records.
329 while (buffer.ReadBool(ref bitposition));
330
331 frame.content = FrameContents.Complete;
332 return SerializationFlags.HasContent /*| SerializationFlags.ForceReliable*/;
333 }
334
335 // No contact records
336 frame.content = FrameContents.Empty;
337 return SerializationFlags.None;
338 }

342 protected override void ApplySnapshot(Frame snapframe, Frame targframe, bool snapIsValid, bool targIsValid)
343 {
344 base.ApplySnapshot(snapframe, targframe, snapIsValid, targIsValid);
345
346 if (snapframe.content == FrameContents.Complete)
347 {
348 List<ContactRecord> contacts = snapframe.contactRecords;
349
350 for (int i = 0, cnt = contacts.Count; i < cnt; ++i)
351 {
352 ContactRecord contact = contacts[i];
353 var pv = PhotonNetwork.GetPhotonView(contact.contactSystemViewID);
354
355 if (pv && pv.IsMine)
356 {
357 var cm = pv.GetComponent<ContactManager>();
358 if (cm)
359 {
360 var currentAttachedICS = cm.GetContacting(contact.contactSystemIndex);
361
362 // Retry the trigger and see if this will consume it, if so run pickup to apply it.
363 var contactevent = new ContactEvent(currentAttachedICS, contactTrigger, contact.contactType);
364
365 var consumed = Contact(contactevent);
366 if (consumed != Consumption.None)
367 Consume(snapframe, contactevent, consumed);
368 }
369 }

379 protected static int ConvertMaskToIndex(int mask)
380 {
381 int bits = 0;
382 if (mask > 32767)
383 {
384 mask >>= 16;
385 bits += 16;
386 }
387
388 if (mask > 127)
389 {
390 mask >>= 8;
391 bits += 8;
392 }
393
394 if (mask > 7)
395 {
396 mask >>= 4;
397 bits += 4;
398 }
399
400 if (mask > 1)
401 {
402 mask >>= 2;
403 bits += 2;
404 }
405
406 if (mask > 0)
407 {
408 bits++;
409 }
410
411 return bits;
412 }

414 public static int ConvertIndexToMask(int index)
415 {
416 if (index == 0)
