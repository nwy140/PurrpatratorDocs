31 public SyncRangedIntAttribute(int min, int max, IndicatorBits indicatorBits = IndicatorBits.None, KeyRate keyRate = KeyRate.UseDefault)
32 {
33 this.min = min;
34 this.max = max;
35 this.indicatorBits = indicatorBits;
36 this.keyRate = keyRate;
37
38 if (min < max)
39 {
40 smallest = min;
41 biggest = max;
42 }
43 else
44 {
45 smallest = max;
46 biggest = min;
47 }
48 uint range = (uint)(biggest - smallest);

52 public override int GetMaxBits(Type fieldType)
53 {
54 switch (indicatorBits)

71 private SerializationFlags Write(int value, int prevValue, byte[] buffer, ref int bitposition, int frameId, SerializationFlags writeFlags)
72 {
73
74 int clamped = (value > biggest) ? biggest : (value < smallest) ? smallest : value;
75
76 if (!IsForced(frameId, clamped, prevValue, writeFlags))
77 {
78 return SerializationFlags.None;
79 }
80
81 if (indicatorBits == IndicatorBits.IsZero)
82 {
83 if (clamped == 0)
84 {
85 buffer.Write(0, ref bitposition, 1);
86 return SerializationFlags.IsComplete;
87 }
88 buffer.Write(1, ref bitposition, 1);
89 }
90
91 else if (indicatorBits == IndicatorBits.IsZeroMidMinMax)
92 {
93 if (clamped == 0)
94 {
95 buffer.Write(0, ref bitposition, 1);
96 return SerializationFlags.IsComplete;
97 }
98 else if (clamped == min)
99 {
100 buffer.Write(1, ref bitposition, 1);
101 return SerializationFlags.IsComplete;
102 }
103 else if (clamped == max)
104 {
105 buffer.Write(3, ref bitposition, 1);
106 return SerializationFlags.IsComplete;
107 }
108 buffer.Write(3, ref bitposition, 1);
109 }
110
111 if (bitCount < 16)
112 buffer.Write((ulong)(clamped - smallest), ref bitposition, bitCount);
113 else
114 buffer.WritePackedBytes((ulong)(clamped - smallest), ref bitposition, bitCount);
115
116 return SerializationFlags.IsComplete;
117 }
118
119 private int Read(byte[] buffer, ref int bitposition, int frameId, SerializationFlags writeFlags)
120 {
121
122 if (indicatorBits == IndicatorBits.IsZero)
123 {
124 if (buffer.Read(ref bitposition, 1) == 0)
125 return 0;
126 }
127 else if (indicatorBits == IndicatorBits.IsZeroMidMinMax)
128 {
129 ulong indicator = buffer.Read(ref bitposition, 2);
130 switch (indicator)
131 {
132 case 0:
133 return 0;
134 case 1:
135 return min;
136 case 2:
137 return max;
138 }
139 }
140
141 if (bitCount < 16)
142 return (int)buffer.Read(ref bitposition, bitCount) + smallest;
143 else
144 return (int)((long)buffer.ReadPackedBytes(ref bitposition, bitCount) + smallest);
145
146 }
