149 public virtual void OnPostSimulate(int frameId, int subFrameId, bool isNetTick)
150 {
151 if (!IsMine)
152 return;
153
154 Frame frame = frames[frameId];
155
158 if (subFrameId == 0)
159 {
160 frame.Clear();
161 }
162
164 if (triggerQueued) //subFrameId < SimpleSyncSettings.sendEveryXTick)
165 {
166 frame.triggerMask |= (uint)1 << subFrameId;
167
168 if (Trigger(frame, subFrameId))
169 TriggerCosmetic(frame, subFrameId);
170
171 triggerQueued = false;
173 frame.content = FrameContents.Complete;
174 }
175 }

180 public virtual void OnIncrementFrame(int newFrameId, int newSubFrameId, int previousFrameId, int prevSubFrameId)
181 {

186 return;

189 {
190 int offset = (newSubFrameId == 0) ? (TickEngineSettings.sendEveryXTick - 1) : newSubFrameId - 1;
191 ApplySubframe(newFrameId, newSubFrameId, offset);

193 }
