262 public virtual SerializationFlags OnNetSerialize(int frameId, byte[] buffer, ref int bitposition, SerializationFlags writeFlags)
263 {
264
266
267 Frame frame = frames[frameId];
268 SerializationFlags flags = SerializationFlags.None;
269
270 bool iskeyframe = IsKeyframe(frameId);
271
272 if (!iskeyframe)
273 return flags;
274
275 if (respawnEnable)
276 {
278 int ticks = frame.ticksUntilRespawn;
279
280 if (ticks >= 0)
281 {
283 buffer.WriteBool(true, ref bitposition);
284 buffer.Write((ulong)ticks, ref bitposition, bitsForTicksUntilRespawn);
285 flags |= SerializationFlags.HasContent;
286 }
287 else
288 {
290 buffer.WriteBool(false, ref bitposition);
291 }
292 }
293
294 if (despawnEnable)
295 {
297 int ticks = frame.ticksUntilDespawn;
298 if (ticks >= 0)
299 {
301 buffer.WriteBool(true, ref bitposition);
302 buffer.Write((ulong)ticks, ref bitposition, bitsForTicksUntilDespawn);
303
304 flags |= SerializationFlags.HasContent;
305 }
306 else
307 {
309 buffer.WriteBool(false, ref bitposition);
310 }
311 }
312
313 return flags;
314 }
315
316 public SerializationFlags OnNetDeserialize(int originFrameId, byte[] buffer, ref int bitposition, FrameArrival frameArrival)
317 {
318
319 Frame frame = frames[originFrameId];
320 SerializationFlags flags = SerializationFlags.None;
321
322 bool iskeyframe = IsKeyframe(originFrameId);
323 if (!iskeyframe)
324 {
325 frame.content = FrameContents.Empty;
326 return flags;
327 }
328
329 if (respawnEnable)
330 {
332 if (buffer.ReadBool(ref bitposition))
333 {
334 frame.ticksUntilRespawn = (int)buffer.Read(ref bitposition, bitsForTicksUntilRespawn);
335 flags |= SerializationFlags.HasContent;
336 }
337 else
338 frame.ticksUntilRespawn = -1;
339 }
340
341 if (despawnEnable)
342 {
343 if (buffer.ReadBool(ref bitposition))
344 {
345 frame.ticksUntilDespawn = (int)buffer.Read(ref bitposition, bitsForTicksUntilDespawn);
346 flags |= SerializationFlags.HasContent;
347 }
348 else
349 frame.ticksUntilDespawn = -1;
350 }
351
352 frame.content = FrameContents.Complete;
353 return flags;
354 }
