408 public SerializationFlags OnNetSerialize(int frameId, byte[] buffer, ref int bitposition, SerializationFlags writeFlags)
409 {
410 Frame frame = frames[frameId];
411
412 bool hasTeleported = frame.hasTeleported;
413 bool mustSend = hasTeleported || (writeFlags & SerializationFlags.NewConnection) != 0;
414
418 if (!mustSend && !isActiveAndEnabled)
419 {
420 buffer.WriteBool(false, ref bitposition);
421 return SerializationFlags.None;
422 }
423
424 bool isKeyframe = IsKeyframe(frameId);
425
427 if (!mustSend && !isKeyframe)
428 {
429 bool hascontent = useDeltas && (prevSentFrame == null || !frame.cm.Equals(prevSentFrame.cm));
430
431 if (!hascontent)
432 {
433 buffer.WriteBool(false, ref bitposition);
434 prevSentFrame = frame;
435 //Debug.LogError("Skipping " + frameId);
436 return SerializationFlags.None;
437 }
438 }
439
440 SerializationFlags flags = SerializationFlags.HasContent;
441
442 //Debug.LogError("OUT " + frameId + " " + frame.m.position);
443
445 buffer.WriteBool(true, ref bitposition);
446
448 buffer.WriteBool(hasTeleported, ref bitposition);
449 if (hasTeleported)
450 {
451 //Debug.LogError(Time.time + " " + name + " " + frameId + " <b>SER TELE</b>");
452
453 transformCrusher.Write(frame.telecm, buffer, ref bitposition);
454 if (teleportReliable)
455 flags |= SerializationFlags.ForceReliable;
456 }
457
458
460 transformCrusher.Write(frame.cm, buffer, ref bitposition);
461 transformCrusher.Decompress(frame.m, frame.cm);
462 prevSentFrame = frame;
463
464 //if (frame.hasTeleported)

610 public override bool OnInterpolate(int snapFrameId, int targFrameId, float t)
611 {
612 if (skipInterpolation)
613 return false;
614
615 bool ready = base.OnInterpolate(snapFrameId, targFrameId, t);
616
617 if (!ready)
618 return false;
619
620 if (interpolation == Interpolation.None)
621 return false;
622
623 if (ReferenceEquals(targFrame, null))
624 return false;
625
626 if (snapFrame.content == FrameContents.Empty)
627 return false;
628
629 if (targFrame.content == FrameContents.Empty)
630 return false;
631
632 if (snapFrame.parentHash != targFrame.parentHash)
633 return false;
634
635 var snapM = snapFrame.hasTeleported ? snapFrame.telem : snapFrame.m;
636
637 if (interpolation == Interpolation.Linear/* || pre1Frame.content != FrameContents.Complete*/)
638 Matrix.Lerp(Matrix.reusable, snapM, targFrame.m, t);
640 else
641 Matrix.CatmullRomLerpUnclamped(Matrix.reusable, prevFrame.m, snapFrame.m, targFrame.m, t);
642
643 //if (transform.GetComponent<SyncPickup>())
644 //Debug.Log(name + " OnInterpolate Completed " + targFrame.m.position);
645
646 transformCrusher.Apply(transform, Matrix.reusable);
647
648 return true;
649 }
