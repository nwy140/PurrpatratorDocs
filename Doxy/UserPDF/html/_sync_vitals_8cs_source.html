169 public Consumption TryTrigger(IContactReactor icontactReactor, ContactEvent contactEvent, int compatibleMounts)
170 {
171 var reactor = (icontactReactor as IVitalsContactReactor);
172
173 if (ReferenceEquals(reactor, null))
174 return Consumption.None;
175
177 if (contactGroups != 0)
178 {
179 //var groups = (reactor as Component).GetComponent<ContactGroupAssign>();
180 IContactGroupsAssign groups = contactEvent.contactTrigger.ContactGroupsAssign;
181
182 int triggermask = ReferenceEquals(groups, null) ? 0 : groups.Mask;
183 if ((contactGroups.Mask & triggermask) == 0)
184 {
185#if UNITY_EDITOR
186 Debug.Log(name + " SyncVitals.TryTrigger() ContactGroup Mismatch. Cannot pick up '" + (contactEvent.contactTrigger as Component).transform.root.name + "' because its has a non-matching ContactGroupAssign.");
187#endif
188 return Consumption.None;
189 }
190 }
191
193 if ((compatibleMounts != defaultMountingMask) && (compatibleMounts & defaultMountingMask) == 0)
194 return Consumption.None;
195
196 Vital vital = vitals.GetVital(reactor.VitalNameType);
197 if (vital == null)
198 {
199 return Consumption.None;
200 }
201
203
204 //double charge = vpr.Charge;
205 Consumption consumed;
206
207 double amountConsumed;
208 {
210 if (IsMine)
211 {
212 double discharge = reactor.DischargeValue(contactEvent.contactType);
213 amountConsumed = vitals.ApplyCharges(discharge, reactor.AllowOverload, reactor.Propagate);
214 }
216 else
217 {
218 amountConsumed = vital.TestApplyChange(reactor, contactEvent);
219 }
220 }
221
222 var consumable = icontactReactor as IVitalsConsumable;
223 if (!ReferenceEquals(consumable, null))
224 consumed = TestConsumption(amountConsumed, consumable, contactEvent);
225 else
226 consumed = Consumption.None;
227
228 return consumed;
229 }
230
231 protected Consumption TestConsumption(double amountConsumed, IVitalsConsumable iva, ContactEvent contactEvent)
232 {
233
234 var consumption = iva.Consumption;
235 var discharge = iva.DischargeValue(contactEvent.contactType);
236
237 if (consumption == Consumption.None)
238 return Consumption.None;
239
240 if (consumption == Consumption.All)
241 {
242 if (amountConsumed != 0)
243 {
244 iva.Charges = 0;
245 return Consumption.All;
246 }
247 return Consumption.None;
248 }
249
250 var consumed = amountConsumed == 0 ? Consumption.None : discharge == amountConsumed ? Consumption.All : Consumption.Partial;
251 iva.Charges -= amountConsumed;
252 return consumed;
253 }

297 {
298
300 var frame = (IsMine) ? offtickFrame : frames[originFrameId];
301
303 if (!buffer.ReadBool(ref bitposition))
304 {
305 return SerializationFlags.None;

307
308 bool isKeyframe = IsKeyframe(originFrameId);
309 var flags = vitals.Deserialize(frame.vitalsData, buffer, ref bitposition, isKeyframe);
310
311 frame.content =
312 (flags & SerializationFlags.IsComplete) != 0 ? FrameContents.Complete :
313 (flags & SerializationFlags.HasContent) != 0 ? FrameContents.Partial :
314 FrameContents.Empty;
