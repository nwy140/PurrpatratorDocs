61 public static void PreSnapshot(int currentFrameId)
62 {
63
64 for (int i = 0; i < connections.Count; ++i)
65 if (!ReferenceEquals(perConnOffsets[connections[i]], null))
66 {
67 var offsets = perConnOffsets[connections[i]];
68
69 var originFrameId = offsets.ConvertFrameLocalToOrigin(currentFrameId);
70 float timeBeforeConsumption = offsets.validFrameMask[originFrameId] ? Time.time - offsets.frameArriveTime[originFrameId] : -1;
71
72 offsets.frameTimeBeforeConsumption[originFrameId] = timeBeforeConsumption;
73 //Debug.Log(timeBeforeConsumption);
74
75 offsets.SnapshotAdvance();
76 }

200 private static void LogNewConnection(int connId, int originFrameId, int frameCount, out ConnectionTickOffsets offsets)
201 {
202 int currentFrame = NetMaster.CurrentFrameId;
203
205 int startingFrameId = currentFrame + (TickEngineSettings.targetBufferSize /*+ 1*/);
206 while (startingFrameId >= frameCount)
207 startingFrameId -= frameCount;
208
209 int originToLocal = startingFrameId - originFrameId;
210 if (originToLocal < 0)
211 originToLocal += frameCount;
212
213 int localToOrigin = frameCount - originToLocal;
214 if (localToOrigin < 0)
215 localToOrigin += frameCount;
216
219 offsets = new ConnectionTickOffsets(connId, originToLocal, localToOrigin);
220
221 AddConnection(connId, offsets);
222
223 }

225 private static void AddConnection(int connId, ConnectionTickOffsets offsets = null)
226 {
227#if PUN_2_OR_NEWER
229 if (PhotonNetwork.LocalPlayer.ActorNumber == connId)
230 return;
231#endif
232 if (!connections.Contains(connId))
233 {
234 perConnOffsets.Add(connId, offsets);
235 connections.Add(connId);
237 NetMsgSends.newPlayers.Add(connId);
238 needToSendInitialForNewConn = true;
239 }
240 else
241 {
242 perConnOffsets[connId] = offsets;
243 }
244 }
