41 public static TransformCrusher CheckAgainstStatics(TransformCrusher tc, bool CheckElementCrusherAsWell = true)
42 {
43 if (ReferenceEquals(tc, null))
44 return null;
45
46 if (CheckElementCrusherAsWell)
47 {
48 tc.posCrusher = ElementCrusher.CheckAgainstStatics(tc.posCrusher);
49 tc.rotCrusher = ElementCrusher.CheckAgainstStatics(tc.rotCrusher);
50 tc.sclCrusher = ElementCrusher.CheckAgainstStatics(tc.sclCrusher);
51 }
52
53 int hash = tc.GetHashCode();
54 if (staticTransformCrushers.ContainsKey(hash))
55 {
56 return staticTransformCrushers[hash];
57 }
58
59 staticTransformCrushers.Add(hash, tc);
60 return tc;
61 }

223 public virtual void CacheValues()
224 {
225 for (int i = 0; i < 4; ++i)
226 {
227 cached_pBits[i] = (posCrusher == null) ? 0 : posCrusher.Cached_TotalBits[i];
228 cached_rBits[i] = (rotCrusher == null) ? 0 : rotCrusher.Cached_TotalBits[i]; //TallyBits((BitCullingLevel)i);
229 cached_sBits[i] = (sclCrusher == null) ? 0 : sclCrusher.Cached_TotalBits[i]; //TallyBits((BitCullingLevel)i);
230 _cached_total[i] = cached_pBits[i] + cached_rBits[i] + cached_sBits[i];
231 cached_total = Array.AsReadOnly(_cached_total);
232 }
233
