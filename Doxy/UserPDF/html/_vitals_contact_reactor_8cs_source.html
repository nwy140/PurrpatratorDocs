236 public virtual Consumption ConsumeCharges(double amount)
237 {
238 if (amount == 0)
239 return Consumption.None;
240
241 // Clamp incoming value to zero
242 double unclamped = _charges - amount;
243 var clamped = ((initialCharges >= 0) ? System.Math.Max(unclamped, 0) : System.Math.Min(unclamped, 0));
244
245 _charges = clamped;
246
247 if (/*useCharges && */clamped != initialCharges)
248 {
249 return (clamped == 0) ? Consumption.All : Consumption.Partial;
250 }
251 return Consumption.None;
252 }
253
254 protected virtual void Consume(Consumption consumed)
255 {
256 if (consumed == Consumption.None)
257 return;
258
259 // Despawn if we have consumed enough to trigger consumeDespawn
260 if (consumeDespawn != Consumption.None && syncState != null)
261 {
262 // If all was consumed, trigger despawn in both cases
263 if (consumed == Consumption.All)
264 syncState.Despawn(false);
265
266 // consumed == partial - despawn if consumeDespawn also is set to partial.
267 else if (consumeDespawn == Consumption.Partial)
268 {
269 syncState.Despawn(false);
270 }
271 }
272 }

371 public double DischargeValue(ContactType contactType = ContactType.Undefined)
372 {
373 double discharge;
374
375 switch (contactType)
376 {
377 case ContactType.Enter:
378 discharge = dischargeOnEnter;
379 break;
380
381 case ContactType.Stay:
382 discharge = dischargePerSec;
383 break;
384
385 case ContactType.Exit:
386 discharge = dischargeOnExit;
387 break;
388
389 case ContactType.Hitscan:
390 discharge = dischargeOnScan;
391 break;
392
393 default:
394 discharge = 0;
395 break;
396 }
397
398 if (useCharges)
399 {
400 if (discharge >= 0)
401 return System.Math.Min(discharge, _charges);
402 else
403 return System.Math.Max(discharge, _charges);
404 }
405
406 return discharge;
407 }
