51 public virtual IVitalsSystem ApplyVitalsSource(Object srcObj)
52 {
53 GameObject vitalsSrcGO;
54 Component vitalsSrcComp;
55
56 if (monitor == MonitorSource.Auto)
57 {
58 if (srcObj == null)
59 {
60 srcObj = (Object)GetComponentInParent<IVitalsSystem>();
61 monitor = MonitorSource.Self;
62 }
63
64 if (srcObj == null)
65 {
66 srcObj = (Object)OwnedIVitals.LastItem;
67 monitor = MonitorSource.Owned;
68 }
69 }
70
72 if (monitor == MonitorSource.Owned)
73 {
74 var ownedVitals = OwnedIVitals.LastItem;
75 vitalsSrcComp = ownedVitals as Component;
76 vitalsSrcGO = null;
77 }
78 else if (monitor == MonitorSource.Self)
79 {
80 vitalsSrcGO = gameObject;
81 vitalsSrcComp = null;
82 }
83 // Auto - try self first - if no vitals found fall back to owned
84 else
85 {
86 vitalsSource = srcObj;
87 vitalsSrcGO = srcObj as GameObject;
88 vitalsSrcComp = srcObj as Component;
89 }
90
91 IVitalsSystem vs;
92
94 if (vitalsSrcGO)
95 {
96 vs = FindIVitalComponentOnGameObj(vitalsSrcGO);
97
98 if (vs != null)
99 vitalsSource = (vs as Component).gameObject;
100
101 }
102 else if (vitalsSrcComp)
103 {
104 vs = vitalsSrcComp as IVitalsSystem;
105
106 if (monitor == MonitorSource.GameObject)
107 vitalsSource = vitalsSrcComp.gameObject;
108 }
109 else
110 {
111 vs = null;
112 vitalsSource = null;
113 }
114
115 return vs;
116
117 }
118
119 #endregion

121 private static IVitalsSystem FindIVitalComponentOnGameObj(GameObject go)
122 {
124 if (go)
125 {
126 IVitalsSystem ivitalcomp = go.GetComponentInParent<IVitalsSystem>();
127 if (ivitalcomp == null)
128 ivitalcomp = go.GetComponentInChildren<IVitalsSystem>();
129 return ivitalcomp;
130 }
131 return null;
132 }
