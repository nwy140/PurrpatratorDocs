76 public void CollectMyBounds()
77 {
78 var wbso = WorldBoundsSettings.Single;
79 if (!wbso)
80 return;
81
82 if (WorldBoundsSettings.Single.worldBoundsGroups.Count == 0)
83 WorldBoundsSettings.Single.worldBoundsGroups.Add(new WorldBoundsGroup());
84
85 // If this is no longer a existing worldbounds layer, reset to default
86 if (worldBoundsGrp >= WorldBoundsSettings.Single.worldBoundsGroups.Count)
87 worldBoundsGrp = 0;
88
89 var grp = wbso.worldBoundsGroups[worldBoundsGrp];
90
91 if (factorIn == BoundsTools.BoundsType.Manual)
92 {
93 myBounds = manualBounds;
94 myBoundsCount = 1;
95 }
96 else
97 {
98 myBounds = BoundsTools.CollectMyBounds(gameObject, factorIn, out myBoundsCount, includeChildren, false);
99 }
100
101 // Remove this from all Groups then readd to the one it currently belongs to.
102 WorldBoundsSettings.RemoveWorldBoundsFromAll(this);
103
104 if (myBoundsCount > 0 && enabled)
105 {
106 if (!grp.activeWorldBounds.Contains(this))
107 {
108 grp.activeWorldBounds.Add(this);
109 grp.RecalculateWorldCombinedBounds();
110 }
111 }
112
113 if (OnWorldBoundsChange != null)
114 OnWorldBoundsChange();
115 }

140 public void FactorInBounds(bool b)
141 {
142 if (this == null)
143 return;
144
145 // if worldboundsgrp is no longer a layer, reset to default
146 if (worldBoundsGrp >= WorldBoundsSettings.Single.worldBoundsGroups.Count)
147 worldBoundsGrp = 0;
148
149 var grp = WorldBoundsSettings.Single.worldBoundsGroups[worldBoundsGrp];
150
151 if (b)
152 {
153 if (!grp.activeWorldBounds.Contains(this))
154 grp.activeWorldBounds.Add(this);
155 }
156 else
157 {
158 grp.activeWorldBounds.Remove(this);
159 }
160
161 grp.RecalculateWorldCombinedBounds();
