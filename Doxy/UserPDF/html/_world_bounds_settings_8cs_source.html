228 private static HashSet<string> namecheck = new HashSet<string>();
229 public static void EnsureNamesAreUnique(int newestIndex = -1)
230 {
231 namecheck.Clear();
232 var wbs = Single.worldBoundsGroups;
233 bool haschanged = false;
234
235 for (int i = 0; i < wbs.Count; ++i)
236 {
237 var wbsi = wbs[i];
238 // Try adding newest changed last, so it will get its named changed, rather than an existing duplicate
239 if (i == newestIndex && i != 0)
240 continue;
241
242 if (i == 0)
243 {
244 if (wbsi.name != WorldBoundsGroup.defaultName)
245 {
246 haschanged = true;
247 wbsi.name = WorldBoundsGroup.defaultName;
248 }
249 }
250 else
251 while (namecheck.Contains(wbsi.name))
252 {
253 haschanged = true;
254 wbsi.name += "X";
255 }
256
257 namecheck.Add(wbsi.name);
258 }
259
260 // Change the newest changed name as needed (if supplied and valid) last
261 if (newestIndex > 0 && newestIndex < wbs.Count)
262 while (namecheck.Contains(wbs[newestIndex].name))
263 {
264 haschanged = true;
265 wbs[newestIndex].name += "X";
266 }
267
268 if (haschanged)
269 {
270 EditorUtility.SetDirty(Single);
271 }
272 }
