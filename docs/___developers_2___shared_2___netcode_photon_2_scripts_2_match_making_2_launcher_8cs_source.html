238 public IEnumerator CheckAllPlayersReady()
239 {
240
241 var players = PhotonNetwork.PlayerList;
242
243
244 // This is just using a shorthand via Linq instead of having a loop with a counter
245 // for checking whether all players in the list have the key "Ready" in their custom properties
246 if (players.All(p => p.CustomProperties.ContainsKey("Ready") && (bool)p.CustomProperties["Ready"]) && PhotonNetwork.CurrentRoom.PlayerCount == 2)
247 {
248
249 Debug.Log("All players are ready!");
