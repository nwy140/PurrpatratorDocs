16 public class AutoZoom : MonoBehaviour
17 {
18 public static List<Transform> watched = new List<Transform>();
19 public const float MAX_FOV = 75;
20 public const float MIN_FOV = 15;
21
22 [Range(.1f, .5f)]
23 [HideInInspector] public float window = .25f;
24 [ValueType("/sec")]
25 [HideInInspector] public float panRate = 20f;
26 [ValueType("/sec")]
27 [HideInInspector] public float zoomRate = 200f;
28
29 // Cache
30 private Camera cam;
31
32 private void Awake()
33 {
34 cam = GetComponent<Camera>();
35
37 if (!cam)
38 watched.Add(transform);
39 }
40
41 private void OnDestroy()
42 {
43 if (watched.Contains(transform))
44 watched.Remove(transform);
45
46 }
47
48 private void LateUpdate()
49 {
50 if (!cam || !cam.isActiveAndEnabled)
51 return;
52
53 Bounds bounds = new Bounds();
54
55 for (int i = 0; i < watched.Count; ++i)
56 {
57 Vector2 screenpoint = cam.WorldToViewportPoint(watched[i].position);
58 bounds.Encapsulate(screenpoint + new Vector2(-0.5f, -0.5f));
59 }
60
61 if (watched.Count > 0)
62 {
63 cam.transform.Rotate(new Vector3(0, 1, 0), (bounds.center.x /*- .25f*/) * Time.deltaTime * panRate);
64 float xAngle = bounds.extents.x - window;
65 float yAngle = bounds.extents.y - window;
66 float angle = xAngle > yAngle ? xAngle : yAngle;
67
68 float fov = cam.fieldOfView + angle * Time.deltaTime * zoomRate;
69
70 cam.fieldOfView = Mathf.Clamp(fov, MIN_FOV, MAX_FOV);
71 }
72 }
