187 void Apply()
188 {
189
190 if (!IsMine)
191 return;
192
193 Vector3 move, turn;
194 SumKeys(out move, out turn);
195
197
198 if (rb && !rb.isKinematic)
199 {
200
203 if (rb)
204 if (clampToCrusher && tc != null)
205 rb.MovePosition(tc.PosCrusher.Clamp(rb.position));
206
207 move *= moveForce * Time.deltaTime;
208
209 if (moveRelative)
210 rb.AddRelativeForce(move, ForceMode.VelocityChange);
211 else
212 rb.AddForce(move, ForceMode.VelocityChange);
213
214 }
215#if !DISABLE_PHYSICS_2D
216 else if (rb2D && !rb2D.isKinematic)
217 {
220 if (rb2D)
221 if (clampToCrusher && tc != null)
222 rb2D.MovePosition(tc.PosCrusher.Clamp(rb2D.position));
223
224 move *= moveForce * Time.deltaTime;
225
226 if (moveRelative)
227 rb2D.AddRelativeForce(move, ForceMode2D.Impulse);
228 else
229 rb2D.AddForce(move, ForceMode2D.Impulse);
230 }
231#endif
232 else
233 {
234
235 Vector3 pos = (rb) ? rb.position : transform.position;
236
237 if (moveRelative)
238 pos += (transform.localRotation * move) * moveSpeed * Time.deltaTime;
239 else
240 pos += move * moveSpeed * Time.deltaTime;
241
242 // If we have a reference to the transform crusher being used for compression, lets contrain our movements to our network limits.
243 if (clampToCrusher && tc != null && tc.PosCrusher != null)
244 pos = tc.PosCrusher.Clamp(pos);
245
246 if (rb)
247 //rb.position = pos;
248 rb.MovePosition(pos);
249 else
250 {
251 transform.position = pos;
252 }
253 }
254
256
257 if (rb && !rb.isKinematic)
258 {
259 turn *= turnForce * Time.deltaTime;
260 rb.AddRelativeTorque(turn, ForceMode.VelocityChange);
261 }
262 else
263 {
264
265 if (clampToCrusher && tc != null && tc.RotCrusher.TRSType != TRSType.Quaternion)
266 {
267 Vector3 v = tc.RotCrusher.Clamp(transform.eulerAngles += turn * turnSpeed * Time.deltaTime);
268 transform.localEulerAngles = v;
269 }
270 else
271 {
272 transform.rotation *= Quaternion.Euler(turn * turnSpeed * Time.deltaTime);
273 }
274 }
275 }
