36 public override void OnInspectorGUI()
37 {
38 spline = target as BezierSpline;
39 EditorGUI.BeginChangeCheck();
40 bool loop = EditorGUILayout.Toggle("Loop", spline.Loop);
41 if (EditorGUI.EndChangeCheck())
42 {
43 Undo.RecordObject(spline, "Toggle Loop");
44 EditorUtility.SetDirty(spline);
45 spline.Loop = loop;
46 }
47
48 if (selectedIndex >= 0 && selectedIndex < spline.ControlPointCount)
49 {
50 DrawSelectedPointInspector();
51 }
52
53 if (GUILayout.Button("Add Curve"))
54 {
55 Undo.RecordObject(spline, "Add Curve");
56 spline.AddCurve();
57 EditorUtility.SetDirty(spline);
58 }
59 }
60
61 private void DrawSelectedPointInspector()
62 {
63 GUILayout.Label("Selected Point");
64 EditorGUI.BeginChangeCheck();
65 Vector3 point = EditorGUILayout.Vector3Field("Position", spline.GetControlPoint(selectedIndex));
66 if (EditorGUI.EndChangeCheck())
67 {
68 Undo.RecordObject(spline, "Move Point");
69 EditorUtility.SetDirty(spline);
70 spline.SetControlPoint(selectedIndex, point);
71 }
72 EditorGUI.BeginChangeCheck();
73 BezierControlPointMode mode = (BezierControlPointMode)EditorGUILayout.EnumPopup("Mode", spline.GetControlPointMode(selectedIndex));
74 if (EditorGUI.EndChangeCheck())
75 {
76 Undo.RecordObject(spline, "Change Point Mode");
77 spline.SetControlPointMode(selectedIndex, mode);
78 EditorUtility.SetDirty(spline);
79 }
80 }
81
82 private void OnSceneGUI()
83 {
84 spline = target as BezierSpline;
85 handleTransform = spline.transform;
86 handleRotation = Tools.pivotRotation == PivotRotation.Local ?
87 handleTransform.rotation : Quaternion.identity;
88
89 Vector3 p0 = ShowPoint(0);
90 for (int i = 1; i < spline.ControlPointCount; i += 3)
91 {
92 Vector3 p1 = ShowPoint(i);
93 Vector3 p2 = ShowPoint(i + 1);
94 Vector3 p3 = ShowPoint(i + 2);
95
96 Handles.color = Color.gray;
97 Handles.DrawLine(p0, p1);
98 Handles.DrawLine(p2, p3);
99
100 Handles.DrawBezier(p0, p3, p1, p2, Color.white, null, 2f);
101 p0 = p3;
102 }
103 ShowDirections();
104 }
105
106 private void ShowDirections()
107 {
108 Handles.color = Color.green;
109 Vector3 point = spline.GetPoint(0f);
110 Handles.DrawLine(point, point + spline.GetDirection(0f) * directionScale);
111 int steps = stepsPerCurve * spline.CurveCount;
112 for (int i = 1; i <= steps; i++)
113 {
114 point = spline.GetPoint(i / (float)steps);
115 Handles.DrawLine(point, point + spline.GetDirection(i / (float)steps) * directionScale);
116 }
117 }
118
119 private Vector3 ShowPoint(int index)
120 {
121 Vector3 point = handleTransform.TransformPoint(spline.GetControlPoint(index));
122 float size = HandleUtility.GetHandleSize(point);
123 if (index == 0)
124 {
125 size *= 2f;
126 }
127 Handles.color = modeColors[(int)spline.GetControlPointMode(index)];
128
129
130 #if UNITY_5_6_OR_NEWER
131 if (Handles.Button(point, handleRotation, size * handleSize, size * pickSize, Handles.DotHandleCap))
132 #else
133 if (Handles.Button(point, handleRotation, size * handleSize, size * pickSize, Handles.DotCap))
134 #endif
135 {
136 selectedIndex = index;
137 Repaint();
138 }
139
140 if (selectedIndex == index)
141 {
142 EditorGUI.BeginChangeCheck();
143 point = Handles.DoPositionHandle(point, handleRotation);
144 if (EditorGUI.EndChangeCheck())
145 {
146 Undo.RecordObject(spline, "Move Point");
147 EditorUtility.SetDirty(spline);
148 spline.SetControlPoint(index, handleTransform.InverseTransformPoint(point));
149 }
150 }
151 return point;
152 }
153 }
