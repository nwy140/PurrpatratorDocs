92 public void OnStartFollowing()

107 void Follow()
108 {
109 cameraOffset.z = -distance;
110 cameraOffset.y = height;
111
112 cameraTransform.position = Vector3.Lerp(cameraTransform.position, this.transform.position +this.transform.TransformVector(cameraOffset), smoothSpeed*Time.deltaTime);

114 cameraTransform.LookAt(this.transform.position + centerOffset);

116 }
