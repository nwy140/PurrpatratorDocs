223 private void SendChatMessage(string inputLine)
224 {
225 if (string.IsNullOrEmpty(inputLine))
226 {
227 return;
228 }
229 if ("test".Equals(inputLine))
230 {
231 if (this.TestLength != this.testBytes.Length)
232 {
233 this.testBytes = new byte[this.TestLength];
234 }
235
236 this.chatClient.SendPrivateMessage(this.chatClient.AuthValues.UserId, this.testBytes, true);
237 }
238
239
240 bool doingPrivateChat = this.chatClient.PrivateChannels.ContainsKey(this.selectedChannelName);
241 string privateChatTarget = string.Empty;
242 if (doingPrivateChat)
243 {
244 // the channel name for a private conversation is (on the client!!) always composed of both user's IDs: "this:remote"
245 // so the remote ID is simple to figure out
246
247 string[] splitNames = this.selectedChannelName.Split(new char[] { ':' });
248 privateChatTarget = splitNames[1];
249 }
250 //UnityEngine.Debug.Log("selectedChannelName: " + selectedChannelName + " doingPrivateChat: " + doingPrivateChat + " privateChatTarget: " + privateChatTarget);
251
252
253 if (inputLine[0].Equals('\\'))
254 {
255 string[] tokens = inputLine.Split(new char[] {' '}, 2);
256 if (tokens[0].Equals("\\help"))
257 {
258 this.PostHelpToCurrentChannel();
259 }
260 if (tokens[0].Equals("\\state"))
261 {
262 int newState = 0;
263
264
265 List<string> messages = new List<string>();
266 messages.Add ("i am state " + newState);
267 string[] subtokens = tokens[1].Split(new char[] {' ', ','});
268
269 if (subtokens.Length > 0)
270 {
271 newState = int.Parse(subtokens[0]);
272 }
273
274 if (subtokens.Length > 1)
275 {
276 messages.Add(subtokens[1]);
277 }
278
279 this.chatClient.SetOnlineStatus(newState,messages.ToArray()); // this is how you set your own state and (any) message
280 }
281 else if ((tokens[0].Equals("\\subscribe") || tokens[0].Equals("\\s")) && !string.IsNullOrEmpty(tokens[1]))
282 {
283 this.chatClient.Subscribe(tokens[1].Split(new char[] {' ', ','}));
284 }
285 else if ((tokens[0].Equals("\\unsubscribe") || tokens[0].Equals("\\u")) && !string.IsNullOrEmpty(tokens[1]))
286 {
287 this.chatClient.Unsubscribe(tokens[1].Split(new char[] {' ', ','}));
288 }
289 else if (tokens[0].Equals("\\clear"))
290 {
291 if (doingPrivateChat)
292 {
293 this.chatClient.PrivateChannels.Remove(this.selectedChannelName);
294 }
295 else
296 {
297 ChatChannel channel;
298 if (this.chatClient.TryGetChannel(this.selectedChannelName, doingPrivateChat, out channel))
299 {
300 channel.ClearMessages();
301 }
302 }
303 }
304 else if (tokens[0].Equals("\\msg") && !string.IsNullOrEmpty(tokens[1]))
305 {
306 string[] subtokens = tokens[1].Split(new char[] {' ', ','}, 2);
307 if (subtokens.Length < 2) return;
308
309 string targetUser = subtokens[0];
310 string message = subtokens[1];
311 this.chatClient.SendPrivateMessage(targetUser, message);
312 }
313 else if ((tokens[0].Equals("\\join") || tokens[0].Equals("\\j")) && !string.IsNullOrEmpty(tokens[1]))
314 {
315 string[] subtokens = tokens[1].Split(new char[] { ' ', ',' }, 2);
316
317 // If we are already subscribed to the channel we directly switch to it, otherwise we subscribe to it first and then switch to it implicitly
318 if (this.channelToggles.ContainsKey(subtokens[0]))
319 {
320 this.ShowChannel(subtokens[0]);
321 }
322 else
323 {
324 this.chatClient.Subscribe(new string[] { subtokens[0] });
325 }
326 }
327 else
328 {
329 Debug.Log("The command '" + tokens[0] + "' is invalid.");
330 }
331 }
332 else
333 {
334 if (doingPrivateChat)
335 {
336 this.chatClient.SendPrivateMessage(privateChatTarget, inputLine);
337 }
338 else
339 {
340 this.chatClient.PublishMessage(this.selectedChannelName, inputLine);
341 }
342 }

496 {
497 if (this.channelToggles.ContainsKey(channelName))
498 {
499 Toggle t = this.channelToggles[channelName];

501

503
504 Debug.Log("Unsubscribed from channel '" + channelName + "'.");
505
506 // Showing another channel if the active channel is the one we unsubscribed from before
507 if (channelName == this.selectedChannelName && this.channelToggles.Count > 0)
508 {
509 IEnumerator<KeyValuePair<string, Toggle>> firstEntry = this.channelToggles.GetEnumerator();
510 firstEntry.MoveNext();
511
512 this.ShowChannel(firstEntry.Current.Key);
513
514 firstEntry.Current.Value.isOn = true;
515 }
516 }
517 else

519 Debug.Log("Can't unsubscribe from channel '" + channelName + "' because you are currently not subscribed to it.");
