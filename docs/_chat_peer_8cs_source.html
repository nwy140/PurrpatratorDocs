103 private string GetNameServerAddress()
104 {
105 var protocolPort = 0;
106 ProtocolToNameServerPort.TryGetValue(this.TransportProtocol, out protocolPort);
107
108 if (this.NameServerPortOverride != 0)
109 {
110 this.Listener.DebugReturn(DebugLevel.INFO, string.Format("Using NameServerPortInAppSettings as port for Name Server: {0}", this.NameServerPortOverride));
111 protocolPort = this.NameServerPortOverride;
112 }
113
114 switch (this.TransportProtocol)
115 {
116 case ConnectionProtocol.Udp:
117 case ConnectionProtocol.Tcp:
118 return string.Format("{0}:{1}", NameServerHost, protocolPort);
119 case ConnectionProtocol.WebSocket:
120 return string.Format("ws://{0}:{1}", NameServerHost, protocolPort);
121 case ConnectionProtocol.WebSocketSecure:
122 return string.Format("wss://{0}:{1}", NameServerHost, protocolPort);
123 default:
124 throw new ArgumentOutOfRangeException();
125 }
126 }

142 public bool AuthenticateOnNameServer(string appId, string appVersion, string region, AuthenticationValues authValues)
143 {
144 if (this.DebugOut >= DebugLevel.INFO)
145 {
146 this.Listener.DebugReturn(DebugLevel.INFO, "OpAuthenticate()");
147 }
148
149 var opParameters = new Dictionary<byte, object>();
150
151 opParameters[ParameterCode.AppVersion] = appVersion;
152 opParameters[ParameterCode.ApplicationId] = appId;
153 opParameters[ParameterCode.Region] = region;
154
155 if (authValues != null)
156 {
157 if (!string.IsNullOrEmpty(authValues.UserId))
158 {
159 opParameters[ParameterCode.UserId] = authValues.UserId;
160 }
161
162 if (authValues.AuthType != CustomAuthenticationType.None)
163 {
164 opParameters[ParameterCode.ClientAuthenticationType] = (byte) authValues.AuthType;
165 if (authValues.Token != null)
166 {
167 opParameters[ParameterCode.Secret] = authValues.Token;
168 }
169 else
170 {
171 if (!string.IsNullOrEmpty(authValues.AuthGetParameters))
172 {
173 opParameters[ParameterCode.ClientAuthenticationParams] = authValues.AuthGetParameters;
174 }
175 if (authValues.AuthPostData != null)
176 {
177 opParameters[ParameterCode.ClientAuthenticationData] = authValues.AuthPostData;
178 }
179 }
180 }
181 }
182
183 return this.SendOperation(ChatOperationCode.Authenticate, opParameters, new SendOptions() { Reliability = true, Encrypt = this.IsEncryptionAvailable });
184 }
