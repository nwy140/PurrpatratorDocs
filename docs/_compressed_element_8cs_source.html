230 public static explicit operator Quaternion(CompressedElement ce)
231 {
232 Element e = ce.Decompress();
233 var crusher = ce.crusher;
234 var trsType = crusher.TRSType;
235
236 if (trsType == TRSType.Quaternion)
237 {
238 return e.quat;
239 }
240 else if (trsType == TRSType.Euler)
241 {
242#if !DEVELOPMENT_BUILD
243 Debug.LogWarning("Casting a CompressedElement of TRSType.Euler to a Quaternion using Quaternion.Euler(). Is this intentional? Cast to Vector3 and convert to Quaternion yourself to silence this warning.");
244#endif
245 return Quaternion.Euler(e.v);
246 }
247 else
248 {
249#if !DEVELOPMENT_BUILD
250 Debug.LogError("Trying to cast a CompresedElement of " + trsType + " to a quaternion, even though it is not a rotation type. Are you using the correct ElementCrusher to compressed this value?");
251#endif
252 return e.quat;

549 public static BitCullingLevel GetGuessableBitCullLevel(CompressedElement a, CompressedElement b, BitCullingLevel maxCullLvl)
550 {
551 for (BitCullingLevel lvl = maxCullLvl; lvl > 0; lvl--)
552 {
553 a.ZeroLowerBits(uppers, lvl);
554 b.ZeroUpperBits(lowers, lvl);
555
556 if ((uppers.cx | lowers.cx) == b.cx &&
557 (uppers.cy | lowers.cy) == b.cy &&
558 (uppers.cz | lowers.cz) == b.cz)
559 return lvl;
560 }
561 return BitCullingLevel.NoCulling;
562 }

582 public static BitCullingLevel FindBestBitCullLevel(CompressedElement a, CompressedElement b, BitCullingLevel maxCulling)
583 {
584 ElementCrusher ec = a.crusher;
585 if (ec == null)
586 {
587 UnityEngine.Debug.Log("NUL CE CRUSHER FindBestBitCullLevel");
588 return BitCullingLevel.NoCulling;
589 }
590
592 if (ec.TRSType == TRSType.Quaternion)
593 {
594 if ((ulong)a.cQuat == (ulong)b.cQuat)
595 return BitCullingLevel.DropAll;
596 else
597 return BitCullingLevel.NoCulling;
598 }
599
600 if (maxCulling == BitCullingLevel.NoCulling || !TestMatchingUpper(a, b, BitCullingLevel.DropThird))
601 return BitCullingLevel.NoCulling;
602
603 if (maxCulling == BitCullingLevel.DropThird || !TestMatchingUpper(a, b, BitCullingLevel.DropHalf))
604 return BitCullingLevel.DropThird;
605
606 if (maxCulling == BitCullingLevel.DropHalf || !TestMatchingUpper(a, b, BitCullingLevel.DropAll))
607 return BitCullingLevel.DropHalf;
608
609 // both values are the same
610 return BitCullingLevel.DropAll;
611 }
