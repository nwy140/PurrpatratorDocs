174 public void PostSnapshot()
175 {
176 int frameCount = TickEngineSettings.frameCount;
177 int currFrameId = NetMaster.CurrentFrameId;
178
179 int origFrameId = currFrameId + localToOriginFrame;
180 if (origFrameId >= frameCount)
181 origFrameId -= frameCount;
182
183 int invalidate = origFrameId - (TickEngineSettings.quaterFrameCount);
184 if (invalidate < 0)
185 invalidate += frameCount;
186
189 validFrameMask.ClearBitsBefore(invalidate, TickEngineSettings.quaterFrameCount);
190
191 if (advanceCount > 0)
192 hadInitialSnapshot = true;
193
194 if (advanceCount != 1)
195 {
196 localToOriginFrame += (advanceCount - 1);
197 if (localToOriginFrame < 0)
198 localToOriginFrame += frameCount;
199 else if (localToOriginFrame >= frameCount)
200 localToOriginFrame -= frameCount;
201
202 originToLocalFrame = frameCount - localToOriginFrame;
203 if (originToLocalFrame < 0)
204 originToLocalFrame += frameCount;
205 }
206 }
