58 var contactType = contactEvent.contactType;
59
60 if (triggerOn != 0 && (contactType & triggerOn) == 0)
61 return Consumption.None;

64 if (IsPickup)

68 int mountableTo = syncState.mountableTo;

75 return Consumption.None;
