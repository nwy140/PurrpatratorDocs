15 public static Component ComponentCopy(this Component original, GameObject destination)
16 {
17 System.Type type = original.GetType();
18 Component copy = destination.AddComponent(type);
19 // Copied fields can be restricted with BindingFlags
20 FieldInfo[] fields = type.GetFields();
21 foreach (System.Reflection.FieldInfo field in fields)
22 {
23 field.SetValue(copy, field.GetValue(original));
24 }
25 return copy;
26 }

28 public static T GetCopyOf<T>(this T comp, T other) where T : Component
29 {
30 if (comp == null || other == null)
31 return comp;
32
33 Type type = comp.GetType();
34 if (type != other.GetType()) return null; // type mis-match
35 BindingFlags flags = BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.Default | BindingFlags.DeclaredOnly;
36 PropertyInfo[] pinfos = type.GetProperties(flags);
37 foreach (var pinfo in pinfos)
38 {
39 if (pinfo.CanWrite)
40 {
41 try
42 {
44 bool isobsolete = Attribute.IsDefined(pinfo, typeof(System.ObsoleteAttribute));
45 if (!isobsolete)
46 pinfo.SetValue(comp, pinfo.GetValue(other, null), null);
47 }
48 catch { } // In case of NotImplementedException being thrown. For some reason specifying that exception didn't seem to catch it, so I didn't catch anything specific.
49 }
50 }
51 FieldInfo[] finfos = type.GetFields(flags);
52 foreach (var finfo in finfos)
53 {
54 finfo.SetValue(comp, finfo.GetValue(other));
55 }
56 return comp as T;
57 }

59 public static T AddColliderCopy<T>(this GameObject go, T toAdd) where T : Collider
60 {
61 T copy = go.AddComponent(toAdd.GetType()).GetCopyOf(toAdd) as T;
62 copy.sharedMaterial = toAdd.sharedMaterial;
63 // Not sure why this isn't copying.
