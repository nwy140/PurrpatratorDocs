108 private void Update()
109 {
110 if (!this.pView.IsMine)
111 {
112 return;
113 }
114
115 // we'll limit how often this update may run at all (to avoid too frequent changes and flooding the server with SetInterestGroups calls)
116 this.timeSinceUpdate += Time.deltaTime;
117 if (this.timeSinceUpdate < this.timeBetweenUpdatesMin)
118 {
119 return;
120 }
121
122 this.lastPosition = this.currentPosition;
123 this.currentPosition = transform.position;
124
125 // This is a simple position comparison of the current and the previous position.
126 // When using Network Culling in a bigger project keep in mind that there might
127 // be more transform-related options, e.g. the rotation, or other options to check.
128 if (this.currentPosition != this.lastPosition)
129 {
130 if (this.HaveActiveCellsChanged())
131 {
132 this.UpdateInterestGroups();
133 this.timeSinceUpdate = 0;
134 }
135 }

171 private bool HaveActiveCellsChanged()
172 {
173 if (this.cullArea.NumberOfSubdivisions == 0)
174 {
175 return false;
176 }
177
178 this.previousActiveCells = new List<byte>(this.activeCells);
179 this.activeCells = this.cullArea.GetActiveCells(transform.position);
180
181 // If the player leaves the area we insert the whole area itself as an active cell.
182 // This can be removed if it is sure that the player is not able to leave the area.
183 while (this.activeCells.Count <= this.cullArea.NumberOfSubdivisions)
184 {
185 this.activeCells.Add(this.cullArea.FIRST_GROUP_ID);
186 }
187
188 if (this.activeCells.Count != this.previousActiveCells.Count)
189 {
190 return true;
191 }
192
193 if (this.activeCells[this.cullArea.NumberOfSubdivisions] != this.previousActiveCells[this.cullArea.NumberOfSubdivisions])
194 {
195 return true;
196 }
197
198 return false;
199 }

204 private void UpdateInterestGroups()
205 {
206 List<byte> disable = new List<byte>(0);
207
208 foreach (byte groupId in this.previousActiveCells)
209 {
210 if (!this.activeCells.Contains(groupId))
211 {
212 disable.Add(groupId);
213 }
214 }
215
216 PhotonNetwork.SetInterestGroups(disable.ToArray(), this.activeCells.ToArray());
217 }

226 public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
227 {
228 // If the player leaves the area we insert the whole area itself as an active cell.
229 // This can be removed if it is sure that the player is not able to leave the area.
230 while (this.activeCells.Count <= this.cullArea.NumberOfSubdivisions)
231 {
232 this.activeCells.Add(this.cullArea.FIRST_GROUP_ID);
233 }
234
235 if (this.cullArea.NumberOfSubdivisions == 1)
236 {
237 this.orderIndex = (++this.orderIndex % this.cullArea.SUBDIVISION_FIRST_LEVEL_ORDER.Length);
238 this.pView.Group = this.activeCells[this.cullArea.SUBDIVISION_FIRST_LEVEL_ORDER[this.orderIndex]];
239 }
240 else if (this.cullArea.NumberOfSubdivisions == 2)
241 {
242 this.orderIndex = (++this.orderIndex % this.cullArea.SUBDIVISION_SECOND_LEVEL_ORDER.Length);
243 this.pView.Group = this.activeCells[this.cullArea.SUBDIVISION_SECOND_LEVEL_ORDER[this.orderIndex]];
244 }
245 else if (this.cullArea.NumberOfSubdivisions == 3)
246 {
247 this.orderIndex = (++this.orderIndex % this.cullArea.SUBDIVISION_THIRD_LEVEL_ORDER.Length);
248 this.pView.Group = this.activeCells[this.cullArea.SUBDIVISION_THIRD_LEVEL_ORDER[this.orderIndex]];
249 }
