236 public void ClearBitsBefore(int start, int count)
237 {
238#if UNITY_EDITOR || DEVELOPMENT_BUILD
239 if (start > 63 || start < 0)
240 RangeError(start);
241#endif
242 ulong mask = count == 64 ? ulong.MaxValue : ((ulong)1 << count) - 1;
243 ulong mask1, mask2;
244 int offset = start - count;
245
246
247 if (offset >= 0)
248 {
249 mask1 = mask << offset;
250 mask2 = mask >> (bitcount - offset);
251 }
252 else
253 {
254 mask1 = mask >> -offset;
255 mask2 = mask << (bitcount + offset);
256 }
257
258 bitmask &= ~mask1 & ~mask2 & alltrue;
259
260 }
264 public int CountValidRange(int start, int lookahead)
265 {
266#if UNITY_EDITOR || DEVELOPMENT_BUILD
267 if (start > 63 || start < 0)
268 RangeError(start);
269#endif
271 for (int i = lookahead; i >= 0; --i)
272 {
273 int b = start + i;
274 if (b >= bitcount)
275 b -= bitcount;
276
277 if ((bitmask & ((ulong)1 << b)) != 0)
278 return i + 1;
279 }

281 return 0;
