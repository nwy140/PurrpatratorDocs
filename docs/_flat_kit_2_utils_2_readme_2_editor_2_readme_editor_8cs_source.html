331 private void ConfigureUrp() {
332 string path = AssetDatabase.GUIDToAssetPath(UrpPipelineAssetGuid.ToString());
333 if (path == null) {
334 Debug.LogError($"[{AssetName}] Couldn't find the URP pipeline asset. " +
335 "Have you unpacked the URP package?");
336 return;
337 }
338
339 var pipelineAsset = AssetDatabase.LoadAssetAtPath<RenderPipelineAsset>(path);
340 if (pipelineAsset == null) {
341 Debug.LogError($"[{AssetName}] Couldn't load the URP pipeline asset.");
342 return;
343 }
344
345 Debug.Log(
346 $"<b>[{AssetName}]</b> Set the render pipeline asset in the Graphics settings to the {AssetName} example.");
347 GraphicsSettings.renderPipelineAsset = pipelineAsset;
348 GraphicsSettings.defaultRenderPipeline = pipelineAsset;
349
350 ChangePipelineAssetAllQualityLevels(pipelineAsset);
351 }
352
353 private void ConfigureBuiltIn() {
354 GraphicsSettings.renderPipelineAsset = null;
355 Debug.Log($"<b>[{AssetName}]</b> Cleared the render pipeline asset in the Graphics settings.");
356
357 ChangePipelineAssetAllQualityLevels(null);
358 }

360 private void ChangePipelineAssetAllQualityLevels(RenderPipelineAsset pipelineAsset) {
361 var originalQualityLevel = QualitySettings.GetQualityLevel();
362
363 var logString = $"<b>[{AssetName}]</b> Set the render pipeline asset for the quality levels:";
364
365 for (int i = 0; i < QualitySettings.names.Length; i++) {
366 logString += $"\n\t{QualitySettings.names[i]}";
367 QualitySettings.SetQualityLevel(i, false);
368 QualitySettings.renderPipeline = pipelineAsset;
369 }
370
371 Debug.Log(logString);
372
373 QualitySettings.SetQualityLevel(originalQualityLevel, false);
374 }
