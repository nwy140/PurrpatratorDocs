819 if (this.NameServerPortInAppSettings != 0)

1137 private bool Connect(string serverAddress, string proxyServerAddress, ServerConnection serverType)
1138 {

1140
1141 if (this.State == ClientState.Disconnecting)
1142 {
1143 this.DebugReturn(DebugLevel.ERROR, "Connect() failed. Can't connect while disconnecting (still). Current state: " + this.State);
1144 return false;
1145 }
1146
1147 // when using authMode AuthOnce or AuthOnceWSS, the token must be available for the init request. if it's null in that case, don't connect
1148 if (this.AuthMode != AuthModeOption.Auth && serverType != ServerConnection.NameServer && this.TokenForInit == null)
1149 {
1150 this.DebugReturn(DebugLevel.ERROR, "Connect() failed. Can't connect to " + serverType + " with Token == null in AuthMode: " + this.AuthMode);
1151 return false;

1171 State = ClientState.ConnectingToGameServer;
