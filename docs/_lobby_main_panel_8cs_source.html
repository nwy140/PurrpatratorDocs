331 private void UpdateCachedRoomList(List<RoomInfo> roomList)
332 {
333 foreach (RoomInfo info in roomList)
334 {
335 // Remove room from cached room list if it got closed, became invisible or was marked as removed
336 if (!info.IsOpen || !info.IsVisible || info.RemovedFromList)
337 {
338 if (cachedRoomList.ContainsKey(info.Name))
339 {
340 cachedRoomList.Remove(info.Name);
341 }
342
343 continue;
344 }
345
346 // Update cached room info
347 if (cachedRoomList.ContainsKey(info.Name))
348 {
349 cachedRoomList[info.Name] = info;
350 }
351 // Add new room info to cache
352 else
353 {
354 cachedRoomList.Add(info.Name, info);
355 }
356 }
357 }
