45 public Matrix(TransformCrusher crusher, Transform transform)
46 {
47 this.crusher = crusher;
48
49 bool lclpos = (crusher == null || crusher.PosCrusher == null || crusher.PosCrusher.local);
50 this.position = (lclpos) ? transform.localPosition : transform.position;
51
52 // Not sure the idea option for scale... lossy or local.
53 bool lclscl = (crusher == null || crusher.SclCrusher == null || crusher.SclCrusher.local);
54 this.scale = (lclscl) ? transform.localScale : transform.lossyScale;
55
56 bool lclrot = (crusher == null || crusher.RotCrusher == null || crusher.RotCrusher.local);
57 if (crusher != null && crusher.RotCrusher != null && crusher.RotCrusher.TRSType == TRSType.Quaternion)
58 this.rotation = (lclrot) ? transform.localRotation : transform.rotation;
59 else
60 this.rotation = (lclrot) ? transform.localEulerAngles : transform.eulerAngles;
61 }

206 public static Matrix Lerp(Matrix target, Matrix start, Matrix end, float t)
207 {
208 var crusher = end.crusher;
209 target.crusher = crusher;
210
211 target.position = Vector3.Lerp(start.position, end.position, t);
212
213 if (crusher != null && crusher.RotCrusher != null)
214 {
215 if (crusher.RotCrusher.TRSType == TRSType.Quaternion)
216 target.rotation = Quaternion.Slerp((Quaternion)start.rotation, (Quaternion)end.rotation, t);
217 else
218 {
219 var srot = (Vector3)start.rotation;
220 var erot = (Vector3)end.rotation;
221 var ydelta = srot.y - erot.y;
222 var zdelta = srot.z - erot.z;
223
224
225 Vector3 unfucked = new Vector3(
226 erot.x,
227 ydelta > 180 ? erot.y + 360 : ydelta < -180 ? erot.y - 360 : erot.y,
228 zdelta > 180 ? erot.z + 360 : zdelta < -180 ? erot.z - 360 : erot.z
229 );
230
231 target.rotation = Vector3.Lerp(srot, (Vector3)unfucked, t);
232 }
233 }
234 else
235 target.rotation = end.rotation;
236
237 target.scale = Vector3.Lerp(start.scale, end.scale, t);
238
239 return target;
240 }
