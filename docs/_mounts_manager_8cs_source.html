43 public static MountsManager EstablishMounts(GameObject go)
44 {
46 NetObject netObj = go.GetComponent<NetObject>();
47
48 if (!netObj)
49 netObj = go.GetComponentInParent<NetObject>();
50
51 if (!netObj)
52 netObj = go.transform.root.gameObject.AddComponent<NetObject>();
53
54#if PUN_2_OR_NEWER
55 MountsManager mountsLookup = netObj.transform.GetNestedComponentInChildren<MountsManager, NetObject>(true);
56#else
57 MountsManager mountsLookup = null;
58#endif
60 if (mountsLookup && mountsLookup.gameObject != go)
61 {
62 Object.DestroyImmediate(mountsLookup);
63 }
64
66 if (!mountsLookup)
67 mountsLookup = netObj.gameObject.AddComponent<MountsManager>();
68
70 var rootMount = netObj.gameObject.GetComponent<Mount>();
71 if (rootMount == null)
72 {
73 rootMount = netObj.gameObject.gameObject.AddComponent<Mount>();
74 rootMount.mountType = new MountSelector(0);
75 }
76 else if (rootMount.mountType.id != 0)
77 {
78 rootMount.mountType.id = 0;
79 }
80
81 mountsLookup.CollectMounts();
82
83 return mountsLookup;
84 }

103 public List<Mount> CollectMounts(bool force = false)
104 {
105
106 indexedMounts.Clear();
107 mountIdLookup.Clear();
108
109#if PUN_2_OR_NEWER
111 transform.GetNestedComponentsInChildren<Mount, NetObject>(indexedMounts);
112#endif
113
114 for (int i = 0, cnt = indexedMounts.Count; i < cnt; ++i)
115 {
116 var mount = indexedMounts[i];
117 mount.componentIndex = i;
118
119 int mountTypeId = mount.mountType.id;
120 if (!mountIdLookup.ContainsKey(mountTypeId))
121 mountIdLookup.Add(mountTypeId, mount);
122 }
123
124 return indexedMounts;
125 }

188 public static void DrawAllMountMappings(Mount thismount, MountsManager mountsLookup)
189 {
190 EnsureStylesExists();
191
192 int usedmask = 0;
193
194 int cnt = mountsLookup.indexedMounts.Count;
195
196 EditorGUILayout.LabelField("All Mounts On NetObject", (GUIStyle)"BoldLabel");
197
199 if (mountsLookup.indexedMounts.Count > MountSettings.mountTypeCount)
200 EditorGUILayout.HelpBox("NetObject has too many mounts. Mount count is limited to the count value in MountSettings.", MessageType.Error);
201
202 if (cnt > 0)
203 {
204 EditorGUILayout.BeginVertical((GUIStyle)"HelpBox");
205
206
207 for (int i = 0; i < mountsLookup.indexedMounts.Count; ++i)
208 {
209
210 SerializedObject mount = new SerializedObject(mountsLookup.indexedMounts[i]);
211
212
213 var mountType = mount.FindProperty("mountType");
214 var index = mountType.FindPropertyRelative("id");
215
217 if (i == 0)
218 {
219 if (index.intValue != 0)
220 {
221 index.intValue = 0;
222 mount.ApplyModifiedProperties();
223 }
224 }
226 else if (index.intValue == 0)
227 {
228 index.intValue = i;
229 mount.ApplyModifiedProperties();
230 Debug.LogWarning("Only the first root Mount on a NetOjbect can be set to mountType 'Root'");
231 }
232
233 int idx = index.intValue;
234
236 int idxmask = 1 << idx;
237 bool isAlreadyUsed = (idxmask & usedmask) != 0;
238 usedmask |= idxmask;
239
240 bool isthis = (ReferenceEquals(mount.targetObject, thismount));
241 const int CNT_WIDTH = 30;
242 const int LABL_PAD = 2;
243
245 EditorGUILayout.BeginHorizontal(isthis ? mountHiliteStyle : mountNoliteStyle);
246
248 Rect indexrect = EditorGUILayout.GetControlRect(GUILayout.MaxWidth(CNT_WIDTH), GUILayout.MinWidth(CNT_WIDTH));
249 indexrect.xMin += 8;
250
252 if (isAlreadyUsed)
253 EditorGUI.LabelField(new Rect(indexrect) { xMin = indexrect.xMin - 3 }, new GUIContent(EditorGUIUtility.FindTexture("CollabError"), "More than one Mount has the same compatible mount selection. Only the first will be used."));
254 else
255 {
257 EditorGUI.BeginDisabledGroup(!isthis);
258 EditorGUI.LabelField(indexrect, i.ToString(), mountlabelstyle);
259 EditorGUI.EndDisabledGroup();
260 }
261
263 float selectorWidth = EditorGUIUtility.labelWidth - (CNT_WIDTH + LABL_PAD + 8);
264 EditorGUI.BeginDisabledGroup(i == 0);
265 EditorGUILayout.PropertyField(mountType, GUIContent.none, GUILayout.MaxWidth(selectorWidth), GUILayout.MinWidth(selectorWidth));

272 Rect thisrect = EditorGUILayout.GetControlRect(GUILayout.MinWidth(40));
273 thisrect.xMin += 16;
274 EditorGUI.LabelField(thisrect, "This", mountlabelstyle);
275 }

280 EditorGUI.EndDisabledGroup();
