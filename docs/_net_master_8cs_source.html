172 void Update()
173 {
175 if (NetObject.activeControlledNetObjs.Count == 0 && NetObject.activeUncontrolledNetObjs.Count == 0)
176 return;
177
178 if (!TickEngineSettings.single.enableTickEngine)
179 return;

181 if (simulationHasRun)
182 PostSimulate();

184 DoubleTime.SnapUpdate();
185
186 NormTimeSinceFixed = (Time.time - Time.fixedTime) / Time.fixedDeltaTime;
187
188 NetMasterCallbacks.OnPreUpdateCallbacks();
189
190 float t = (Time.time - lastSentTickTime) / (TickEngineSettings.netTickInterval);
191
192 // Interpolate NetObjects
193 NetObject.NetObjDictsLocked = true;
194 foreach (var no in NetObject.activeUncontrolledNetObjs.Values)
195 no.OnInterpolate(_prevFrameId, _currFrameId, t);
196 NetObject.NetObjDictsLocked = false;
197
198 // Interpolate Others
199 NetMasterCallbacks.OnInterpolateCallbacks(_prevFrameId, _currFrameId, t);

230 private static void IncrementFrameId()
231 {
232 _prevSubFrameId = _currSubFrameId;
233 _currSubFrameId++;
234
235 if (_currSubFrameId >= TickEngineSettings.sendEveryXTick)
236 {
237 _currSubFrameId = 0;
238 _prevFrameId = _currFrameId;
239
240 _currFrameId++;
241 if (_currFrameId >= TickEngineSettings.frameCount)
242 _currFrameId = 0;
243 }
244
245 NetMasterCallbacks.OnIncrementFrameCallbacks(_currFrameId, _currSubFrameId, _prevFrameId, _prevSubFrameId);
246
247 if (_currSubFrameId == 0)
248 {
250 TickManager.PreSnapshot(_currFrameId);
251
252 // Snapshot NetObjects
253 NetObject.NetObjDictsLocked = true;
254 foreach (var no in NetObject.activeUncontrolledNetObjs.Values)
255 no.OnSnapshot(_currFrameId);
256 NetObject.NetObjDictsLocked = false;
257
258 // Snapshot Others
259 NetMasterCallbacks.OnSnapshotCallbacks(_currFrameId);
260
261 TickManager.PostSnapshot(_currFrameId);
262
263 lastSentTickTime = Time.fixedTime;
264 }

370 private static void SerializeNetObjDict(NonAllocDictionary<int, NetObject> dict, byte[] buffer, ref int bitposition, ref SerializationFlags flags, SerializationFlags writeFlags)
371 {
372 foreach (var no in dict.Values)
373 {
374
376 int holdStartPos = bitposition;
377
379 buffer.WritePackedBytes((uint)no.ViewID, ref bitposition, 32);
380
382 int holdHasDataPos = bitposition;
383 buffer.WriteBool(true, ref bitposition);
384
386 int holdDataSizePos = bitposition;
387 bitposition += BITS_FOR_NETOBJ_SIZE;
388
389 var objflags = no.OnNetSerialize(_currFrameId, buffer, ref bitposition, writeFlags);
390
392 if (objflags == SerializationFlags.None)
393 {
395 if (no.SkipWhenEmpty)
396 {
397 bitposition = holdStartPos;
398 }
399 else
400 {
401 bitposition = holdHasDataPos;
402 buffer.WriteBool(false, ref bitposition);
403 }
404 }
405 else
406 {
408 flags |= objflags;

421 public static void ReceiveMessage(object conn, int connId, byte[] buffer)
422 {
423 int frameCount = TickEngineSettings.frameCount;
424 int bitposition = 0;
425
426
428 int originFrameId = (int)buffer.Read(ref bitposition, TickEngineSettings.frameCountBits);
429
430 //Debug.Log("in " + originFrameId);
431
432#if DEVELOPMENT_BUILD || UNITY_EDITOR
433 if (originFrameId < 0 || originFrameId >= frameCount)
434 Debug.LogError("INVALID FRAME ID " + originFrameId + " CORRUPT STREAM LIKELY DUE TO READ/WRITE DISAGREEMENT.");
435#endif
436 FrameArrival arrival;
437 /*var offsets = */
438 TickManager.LogIncomingFrame(connId, originFrameId, out arrival);
439
441 //NetMasterCallbacks.OnPreDeserializeTickCallbacks(localFrameId, buffer, ref bitposition);
442
444 while (true)
445 {
447 int viewid = (int)buffer.ReadPackedBytes(ref bitposition, 32);
448
449 if (viewid == 0)
450 break;
451
453 bool hasData = buffer.ReadBool(ref bitposition);
454
456 if (!hasData)
457 continue;
458
460 int holdDataSizePos = bitposition;
461 int bitcount = (int)buffer.Read(ref bitposition, BITS_FOR_NETOBJ_SIZE);
462 int expectedBitPosition = holdDataSizePos + bitcount;
463
464 var pv = PhotonNetwork.GetPhotonView(viewid);
465 var netobj = pv ? pv.GetComponent<NetObject>() : null;
466
468 if (ReferenceEquals(netobj, null))
469 {
470 bitposition = expectedBitPosition;
471 continue;
472 }
473
475 if (netobj.IgnoreNonControllerUpdates)
476 {
477 int controllerActorNr = pv.ControllerActorNr;
478 int ownerActorNr = pv.OwnerActorNr;
479
480 if (controllerActorNr == -1)
481 {
482#if UNITY_EDITOR || DEVELOPMENT_BUILD
483 Debug.Log(pv.name + ":" + pv.ViewID + " OwnershipUpdate has not yet arrived from master, so owner is still unknown. Accepting update from Actor " + connId + " as temporary controller until OwnershipUpdate arrives and sets the Owner/Controller.");
484#endif
485 pv.ControllerActorNr = connId;
486 }
487 else if (controllerActorNr != connId && ownerActorNr != connId)
488 {
489 {
490#if UNITY_EDITOR || DEVELOPMENT_BUILD
491 Debug.LogError("fid: " + originFrameId + "Update from ActorId: " + connId + " for object " + netobj.name + ":" + pv.ViewID + " , but that conn does not have controller authority. Current cowner:controller is "
492 + pv.OwnerActorNr + ":" + controllerActorNr);
493#endif
494 bitposition = expectedBitPosition;
495 continue;
496 }
497 }
498 }
499
500
501
502 netobj.OnDeserialize(connId, originFrameId, buffer, ref bitposition, hasData, arrival);
503
504
505#if UNITY_EDITOR
506 if (bitposition != expectedBitPosition)
507 Debug.LogWarning("Deserialization mismatch for object " + netobj.name + " viewId: " + netobj.ViewID + " owned by: " + netobj.photonView.OwnerActorNr +
508 ". Bitposition off by: " + (expectedBitPosition - bitposition) + " bits.");
509#endif
511 bitposition = expectedBitPosition;
512
513 }

516 public static FrameArrival CheckFrameArrival(int incomingFrame)
517 {
518 int delta = incomingFrame - _prevFrameId;
519
520 if (delta == 0)
521 return FrameArrival.IsSnap;
522
523 // change negative values into positive to check for wrap around.
524 if (delta < 0)
525 delta += TickEngineSettings.frameCount;
526
527 if (delta == 1)
528 return FrameArrival.IsTarget;
529
530 if (delta >= TickEngineSettings.halfFrameCount)
531 return FrameArrival.IsLate;
532
533 return FrameArrival.IsFuture;
534
535 }
