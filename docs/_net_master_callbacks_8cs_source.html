87 public static bool CallbacksLocked
88 {
89 set
90 {
91 callbacksLocked = value;
92
93 if (!value)
94 {
96 while (pendingRegistrations.Count > 0)
97 {
98 var que = pendingRegistrations.Dequeue();
99 RegisterCallbackInterfaces(que.comp, que.register, false);
100 }
101
102 while (postCallbackActions.Count > 0)
103 {
104 postCallbackActions.Dequeue().Invoke();
105 }
106 }
107 }
