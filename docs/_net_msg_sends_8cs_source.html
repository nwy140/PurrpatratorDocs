37 public static void Send(this byte[] buffer, int bitposition, UnityEngine.Object refObj, SerializationFlags flags, bool flush = false)
38 {
39
40 var currentRoom = PhotonNetwork.CurrentRoom;
41
42 if (PhotonNetwork.OfflineMode || currentRoom == null || currentRoom.Players == null)
43 {
44 return;
45 }
46
47 bool sendToSelf = (flags & SerializationFlags.SendToSelf) != 0;
48
49 // no need to send OnSerialize messages while being alone (these are not buffered anyway)
50 if (!sendToSelf && !TickEngineSettings.single.sendWhenSolo && currentRoom.Players.Count <= 1)
51 {
52 return;
53 }
54
55 ReceiveGroup sendTo = sendToSelf ? ReceiveGroup.All : ReceiveGroup.Others;
56
57
58 int bytecount = (bitposition + 7) >> 3;
59
60 var nc = PhotonNetwork.NetworkingClient;
61
62 DeliveryMode deliveryMode;
63
64 if (newPlayers.Count > 0)
65 {
66 deliveryMode = DeliveryMode.Reliable;
67 newPlayers.Clear();
68 }
69 else
70 {
71 bool forceReliable = (flags & SerializationFlags.ForceReliable) != 0;
72 deliveryMode = unreliableCapable ? (forceReliable ? DeliveryMode.ReliableUnsequenced : DeliveryMode.Unreliable) : DeliveryMode.Reliable;
73 }
74
75 //if (deliveryMode != DeliveryMode.Unreliable)
76 // Debug.LogError("Forced Reliable Send");
77
78 SendOptions sendOptions = new SendOptions() { DeliveryMode = deliveryMode };
79 var peer = PhotonNetwork.NetworkingClient.LoadBalancingPeer;
80
81#if PUN_2_19_OR_NEWER
82 if (peer.SerializationProtocolType == SerializationProtocol.GpBinaryV16)
83 {
84 System.ArraySegment<byte> slice = new System.ArraySegment<byte>(buffer, 0, bytecount);
85 nc.OpRaiseEvent(NetMsgCallbacks.DEF_MSG_ID, slice, opts[(int)sendTo], sendOptions);
86 }
87 else
88 {
89 ByteArraySlice slice = peer.ByteArraySlicePool.Acquire(buffer, 0, bytecount);
90 nc.OpRaiseEvent(NetMsgCallbacks.DEF_MSG_ID, slice, opts[(int)sendTo], sendOptions);
91 }
92#else
93 System.ArraySegment<byte> slice = new System.ArraySegment<byte>(buffer, 0, bytecount);
94 nc.OpRaiseEvent(NetMsgCallbacks.DEF_MSG_ID, slice, opts[(int)sendTo], sendOptions);
95#endif
96 if (flush)
97 {
98 while (peer.SendOutgoingCommands());
99 }
100 }
