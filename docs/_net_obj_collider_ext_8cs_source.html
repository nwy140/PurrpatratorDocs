25 public static void IndexColliders(this NetObject netObj)
26 {
27 var indexed = netObj.indexedColliders;
28 var lookup = netObj.colliderLookup;
29
30 lookup.Clear();
31 indexed.Clear();
32
33#if PUN_2_OR_NEWER
35 netObj.transform.GetNestedComponentsInChildren<Component, NetObject>(reusableComponents);
36#endif
38 int compCnt = reusableComponents.Count;
39 for (int c = 0; c < compCnt; ++c)
40 {
41 Component comp = reusableComponents[c];
42 Collider collider = comp as Collider;
43 if (collider)
44 {
45 indexed.Add(comp);
46 }
47 else
48 {
49 Collider2D collider2D = comp as Collider2D;
50 if (collider2D)
51 indexed.Add(comp);
52 }
53 }
54
56 for (int i = 0, cnt = indexed.Count; i < cnt; ++i)
57 lookup.Add(indexed[i], i);
58
59 netObj.bitsForColliderIndex = (indexed.Count - 1).GetBitsForMaxValue();
60 }

65 public static int GetFirstChildCollider(this Transform transform, ref Component firstFoundCollider, bool countTriggers, bool countNonTriggers)
