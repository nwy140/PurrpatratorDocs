404 private void DetermineActiveAndControlled(bool amController)
405 {
406 int key = photonView.ViewID;
407
408 if (netObjDictsLocked)
409 {
410 pendingActiveNetObjDictChanges.Enqueue(this);
411 return;
412 }
413
414 bool InControllerOfList = activeControlledNetObjs.ContainsKey(key);
415 bool InOthersList = activeUncontrolledNetObjs.ContainsKey(key);
416
417 if (isActiveAndEnabled)
418 {
419 if (amController)
420 {
421 if (!InControllerOfList)
422 activeControlledNetObjs.Add(key, this);
423
424 if (InOthersList)
425 activeUncontrolledNetObjs.Remove(key);
426
427 }
428 else
429 {
430
431 if (InControllerOfList)
432 activeControlledNetObjs.Remove(key);
433
434 if (!InOthersList)
435 activeUncontrolledNetObjs.Add(key, this);
436 }
437 }
438 else
439 {
440
441 if (InControllerOfList)
442 activeControlledNetObjs.Remove(key);
443
444 if (InOthersList)
445 activeUncontrolledNetObjs.Remove(key);
446 }
447
448 //if (GetComponent<SyncPickup>())
449 // Debug.Log(Time.time + " " + name + " <b>Change Controller</b>" + amController);
450 }

526 {
527 var comp = reusableComponents[index];
529 if (comp == this)
530 continue;
