117 internal static float Generate(float x)
118 {
119 int i0 = FastFloor(x);
120 int i1 = i0 + 1;
121 float x0 = x - i0;
122 float x1 = x0 - 1.0f;
123
124 float n0, n1;
125
126 float t0 = 1.0f - x0 * x0;
127 t0 *= t0;
128 n0 = t0 * t0 * grad(perm[i0 & 0xff], x0);
129
130 float t1 = 1.0f - x1 * x1;
131 t1 *= t1;
132 n1 = t1 * t1 * grad(perm[i1 & 0xff], x1);
133 // The maximum value of this noise is 8*(3/4)^4 = 2.53125
134 // A factor of 0.395 scales to fit exactly within [-1,1]
135 return 0.395f * (n0 + n1);
136 }

144 internal static float Generate(float x, float y)
145 {
146 const float F2 = 0.366025403f; // F2 = 0.5*(sqrt(3.0)-1.0)
147 const float G2 = 0.211324865f; // G2 = (3.0-Math.sqrt(3.0))/6.0
148
149 float n0, n1, n2; // Noise contributions from the three corners
150
151 // Skew the input space to determine which simplex cell we're in
152 float s = (x + y) * F2; // Hairy factor for 2D
153 float xs = x + s;
154 float ys = y + s;
155 int i = FastFloor(xs);
156 int j = FastFloor(ys);
157
158 float t = (float)(i + j) * G2;
159 float X0 = i - t; // Unskew the cell origin back to (x,y) space
160 float Y0 = j - t;
161 float x0 = x - X0; // The x,y distances from the cell origin
162 float y0 = y - Y0;
163
164 // For the 2D case, the simplex shape is an equilateral triangle.
165 // Determine which simplex we are in.
166 int i1, j1; // Offsets for second (middle) corner of simplex in (i,j) coords
167 if (x0 > y0) { i1 = 1; j1 = 0; } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
168 else { i1 = 0; j1 = 1; } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
169
170 // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
171 // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
172 // c = (3-sqrt(3))/6
173
174 float x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y) unskewed coords
175 float y1 = y0 - j1 + G2;
176 float x2 = x0 - 1.0f + 2.0f * G2; // Offsets for last corner in (x,y) unskewed coords
177 float y2 = y0 - 1.0f + 2.0f * G2;
178
179 // Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
180 int ii = Mod(i, 256);
181 int jj = Mod(j, 256);
182
183 // Calculate the contribution from the three corners
184 float t0 = 0.5f - x0 * x0 - y0 * y0;
185 if (t0 < 0.0f) n0 = 0.0f;
186 else
187 {
188 t0 *= t0;
189 n0 = t0 * t0 * grad(perm[ii + perm[jj]], x0, y0);
190 }
191
192 float t1 = 0.5f - x1 * x1 - y1 * y1;
193 if (t1 < 0.0f) n1 = 0.0f;
194 else
195 {
196 t1 *= t1;
197 n1 = t1 * t1 * grad(perm[ii + i1 + perm[jj + j1]], x1, y1);
198 }
199
200 float t2 = 0.5f - x2 * x2 - y2 * y2;
201 if (t2 < 0.0f) n2 = 0.0f;
202 else
203 {
204 t2 *= t2;
205 n2 = t2 * t2 * grad(perm[ii + 1 + perm[jj + 1]], x2, y2);
206 }
207
208 // Add contributions from each corner to get the final noise value.
209 // The result is scaled to return values in the interval [-1,1].
210 return 40.0f * (n0 + n1 + n2); // TODO: The scale factor is preliminary!
211 }

214 internal static float Generate(float x, float y, float z)
215 {
216 // Simple skewing factors for the 3D case
217 const float F3 = 0.333333333f;
218 const float G3 = 0.166666667f;
219
220 float n0, n1, n2, n3; // Noise contributions from the four corners
221
222 // Skew the input space to determine which simplex cell we're in
223 float s = (x + y + z) * F3; // Very nice and simple skew factor for 3D
224 float xs = x + s;
225 float ys = y + s;
226 float zs = z + s;
227 int i = FastFloor(xs);
228 int j = FastFloor(ys);
229 int k = FastFloor(zs);
230
231 float t = (float)(i + j + k) * G3;
232 float X0 = i - t; // Unskew the cell origin back to (x,y,z) space
233 float Y0 = j - t;
234 float Z0 = k - t;
235 float x0 = x - X0; // The x,y,z distances from the cell origin
236 float y0 = y - Y0;
237 float z0 = z - Z0;
238
239 // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
240 // Determine which simplex we are in.
241 int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k) coords
242 int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k) coords
243
244 /* This code would benefit from a backport from the GLSL version! */
245 if (x0 >= y0)
246 {
247 if (y0 >= z0)
248 { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 1; k2 = 0; } // X Y Z order
249 else if (x0 >= z0) { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 0; k2 = 1; } // X Z Y order
250 else { i1 = 0; j1 = 0; k1 = 1; i2 = 1; j2 = 0; k2 = 1; } // Z X Y order
251 }
252 else
253 { // x0<y0
254 if (y0 < z0) { i1 = 0; j1 = 0; k1 = 1; i2 = 0; j2 = 1; k2 = 1; } // Z Y X order
255 else if (x0 < z0) { i1 = 0; j1 = 1; k1 = 0; i2 = 0; j2 = 1; k2 = 1; } // Y Z X order
256 else { i1 = 0; j1 = 1; k1 = 0; i2 = 1; j2 = 1; k2 = 0; } // Y X Z order
257 }
258
259 // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
260 // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
261 // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
262 // c = 1/6.
263
264 float x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z) coords
265 float y1 = y0 - j1 + G3;
266 float z1 = z0 - k1 + G3;
267 float x2 = x0 - i2 + 2.0f * G3; // Offsets for third corner in (x,y,z) coords
268 float y2 = y0 - j2 + 2.0f * G3;
269 float z2 = z0 - k2 + 2.0f * G3;
270 float x3 = x0 - 1.0f + 3.0f * G3; // Offsets for last corner in (x,y,z) coords
271 float y3 = y0 - 1.0f + 3.0f * G3;
272 float z3 = z0 - 1.0f + 3.0f * G3;
273
274 // Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
275 int ii = Mod(i, 256);
276 int jj = Mod(j, 256);
277 int kk = Mod(k, 256);
278
279 // Calculate the contribution from the four corners
280 float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
281 if (t0 < 0.0f) n0 = 0.0f;
282 else
283 {
284 t0 *= t0;
285 n0 = t0 * t0 * grad(perm[ii + perm[jj + perm[kk]]], x0, y0, z0);
286 }
287
288 float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
289 if (t1 < 0.0f) n1 = 0.0f;
290 else
291 {
292 t1 *= t1;
293 n1 = t1 * t1 * grad(perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]], x1, y1, z1);
294 }
295
296 float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
297 if (t2 < 0.0f) n2 = 0.0f;
298 else
299 {
300 t2 *= t2;
301 n2 = t2 * t2 * grad(perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]], x2, y2, z2);
302 }
303
304 float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
305 if (t3 < 0.0f) n3 = 0.0f;
306 else
307 {
308 t3 *= t3;
309 n3 = t3 * t3 * grad(perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]], x3, y3, z3);
310 }
311
312 // Add contributions from each corner to get the final noise value.
313 // The result is scaled to stay just inside [-1,1]
314 return 32.0f * (n0 + n1 + n2 + n3); // TODO: The scale factor is preliminary!
315 }

360 private static float grad(int hash, float x)
361 {
362 int h = hash & 15;
363 float grad = 1.0f + (h & 7); // Gradient value 1.0, 2.0, ..., 8.0
364 if ((h & 8) != 0) grad = -grad; // Set a random sign for the gradient
365 return (grad * x); // Multiply the gradient with the distance
366 }
367
368 private static float grad(int hash, float x, float y)
369 {
370 int h = hash & 7; // Convert low 3 bits of hash code
371 float u = h < 4 ? x : y; // into 8 simple gradient directions,
372 float v = h < 4 ? y : x; // and compute the dot product with (x,y).
373 return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -2.0f * v : 2.0f * v);
374 }
375
376 private static float grad(int hash, float x, float y, float z)
377 {
378 int h = hash & 15; // Convert low 4 bits of hash code into 12 simple
379 float u = h < 8 ? x : y; // gradient directions, and compute dot product.
380 float v = h < 4 ? y : h == 12 || h == 14 ? x : z; // Fix repeats at h = 12 to 15
381 return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -v : v);
382 }
383
384 private static float grad(int hash, float x, float y, float z, float t)
385 {
386 int h = hash & 31; // Convert low 5 bits of hash code into 32 simple
387 float u = h < 24 ? x : y; // gradient directions, and compute dot product.
388 float v = h < 16 ? y : z;
389 float w = h < 8 ? z : t;
390 return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -v : v) + ((h & 4) != 0 ? -w : w);
391 }
