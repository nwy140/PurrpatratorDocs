37 void IPointerClickHandler.OnPointerClick(PointerEventData eventData)
38 {
39 if (!PhotonNetwork.InRoom || (this.ModifierKey != KeyCode.None && !Input.GetKey(this.ModifierKey)) || eventData.button != this.Button)
40 {
41 return;
42 }
43
44
45 switch (this.InstantiateType)
46 {
47 case InstantiateOption.Mine:
48 PhotonNetwork.Instantiate(this.Prefab.name, eventData.pointerCurrentRaycast.worldPosition + new Vector3(0, 0.5f, 0), Quaternion.identity, 0);
49 break;
50 case InstantiateOption.Scene:
51 PhotonNetwork.InstantiateRoomObject(this.Prefab.name, eventData.pointerCurrentRaycast.worldPosition + new Vector3(0, 0.5f, 0), Quaternion.identity, 0, null);
52 break;
53 }
