54 public IEnumerator ClickFlash()
55 {
56 if (isFlashing)
57 {
58 yield break;
59 }
60 isFlashing = true;
61
62 this.originalMaterial = GetComponent<Renderer>().material;
63 if (!this.originalMaterial.HasProperty("_EmissionColor"))
64 {
65 Debug.LogWarning("Doesn't have emission, can't flash " + gameObject);
66 yield break;
67 }
68
69 bool wasEmissive = this.originalMaterial.IsKeywordEnabled("_EMISSION");
70 this.originalMaterial.EnableKeyword("_EMISSION");
71
72 this.originalColor = this.originalMaterial.GetColor("_EmissionColor");
73 this.originalMaterial.SetColor("_EmissionColor", Color.white);
74
75 for (float f = 0.0f; f <= 1.0f; f += 0.08f)
76 {
77 Color lerped = Color.Lerp(Color.white, this.originalColor, f);
78 this.originalMaterial.SetColor("_EmissionColor", lerped);
79 yield return null;
80 }
81
82 this.originalMaterial.SetColor("_EmissionColor", this.originalColor);
83 if (!wasEmissive) this.originalMaterial.DisableKeyword("_EMISSION");
84 isFlashing = false;
85 }
