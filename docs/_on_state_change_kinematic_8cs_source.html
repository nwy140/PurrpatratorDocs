109 protected virtual void SetOwnedKinematics(ObjState state)
110 {
111 if (RigidbodyType == RigidbodyType.None)
112 return;
113
114 KinematicSetting kinematicSetting;
115
116
117 if (state == ObjState.Despawned && onDespawned != KinematicSetting.Ignore)
118 kinematicSetting = onDespawned;
119 else if ((state & ObjState.Anchored) != 0 && onAnchored != KinematicSetting.Ignore)
120 kinematicSetting = onAnchored;
121 else if ((state & ObjState.Mounted) != 0 && onMounted != KinematicSetting.Ignore)
122 kinematicSetting = onMounted;
123 else if ((state & ObjState.Transit) != 0 && onTransit != KinematicSetting.Ignore)
124 kinematicSetting = onTransit;
125 else if ((state & ObjState.Dropped) != 0 && onDropped != KinematicSetting.Ignore)
126 kinematicSetting = onDropped;
127
128 else if ((state & ObjState.Visible) != 0 && onVisible != KinematicSetting.Ignore)
129 kinematicSetting = onVisible;
130 else
131 kinematicSetting = KinematicSetting.Default;
132
133 bool isKinematic =
134 (kinematicSetting == KinematicSetting.NonKinematic) ? false :
135 (kinematicSetting == KinematicSetting.Kinematic) ? true :
136 kinematicDefault;
137
138 if (RigidbodyType == RigidbodyType.RB)
139 {
140 // setting the rb to the transform is needed here to harden any recent transform.pos changes. Otherwise this switch will negate them
141 rb.position = transform.position;
142 if (isKinematic)
143 {
144#if UNITY_2018_3_OR_NEWER
145 rb.collisionDetectionMode = CollisionDetectionMode.ContinuousSpeculative;
146#endif
147 rb.isKinematic = true;
148 }
149 else
150 {
151 rb.isKinematic = false;
152 }
153 // Be sure to not interpolate attache objects. Even as kinematic, a moving parent will create position desync
154 rb.interpolation = (state != ObjState.Despawned && (state & ObjState.Mounted) == 0) ? (RigidbodyInterpolation)interpolateDefault : RigidbodyInterpolation.None;
155
157 //rb.interpolation = RigidbodyInterpolation.None;
158 }
159 else
160 {
161 rb2d.position = transform.position;
162 rb2d.isKinematic = isKinematic;
163 rb2d.simulated = !isKinematic;
164 rb2d.interpolation = (state != ObjState.Despawned && (state & ObjState.Mounted) == 0) ? (RigidbodyInterpolation2D)interpolateDefault : RigidbodyInterpolation2D.None;
165 }
166
167 currentState = state;
168 }
