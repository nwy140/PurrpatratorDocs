45
46 public static PackObjectInfo GetPackObjectInfo(Type type)
47 {
48 PackObjectInfo info;
49 if (packObjInfoLookup.TryGetValue(type, out info))
50 return info;
51
53 var packType = Type.GetType("Pack_" + type.Name);
54 if (packType != null)
55 {
56 Debug.LogError("BRUTE FORCE Pack_" + type.Name + ". This shouldn't happen.");
57 packType.GetMethod("Initialize").Invoke(null, null);
58 }
59
60 if (packObjInfoLookup.TryGetValue(type, out info))
61 return info;
62
63#if UNITY_EDITOR
64 Debug.LogWarning("PackObject code has not been generated for type '" + type.Name + "'. Is Code Generation disabled in " + typeof(PackObjectSettings).Name + "?");
65#endif
66 return null;
67 }
