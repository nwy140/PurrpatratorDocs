34 public ParameterSettings(int hash, ParameterDefaults defs, ref int paramCount, AnimatorControllerParameterType paramType)
35 {
36 this.hash = hash;
37 this.paramType = paramType;
38
39 switch (paramType)
40 {
41 case AnimatorControllerParameterType.Float:
42 include = defs.includeFloats;
43 interpolate = defs.interpolateFloats;
44 extrapolate = defs.extrapolateFloats;
45 defaultValue = (float)defs.defaultFloat;
46 fcrusher = new LiteFloatCrusher(LiteFloatCompressType.Half16, 0, 1, true);
47 break;
48
49 case AnimatorControllerParameterType.Int:
50 include = defs.includeInts;
51 interpolate = defs.interpolateInts;
52 extrapolate = defs.extrapolateInts;
53 defaultValue = (int)defs.defaultInt;
54 icrusher = new LiteIntCrusher();
55 break;
56
57 case AnimatorControllerParameterType.Bool:
58 include = defs.includeBools;
59 interpolate = ParameterInterpolation.Hold;
60 extrapolate = ParameterExtrapolation.Hold;
61 defaultValue = (bool)defs.defaultBool;
62 break;
63
64 case AnimatorControllerParameterType.Trigger:
65 include = defs.includeTriggers;
66 interpolate = ParameterInterpolation.Default;
67 extrapolate = ParameterExtrapolation.Default;
68 defaultValue = (bool)defs.defaultTrigger;
69 break;
70
71 default:
72 break;
73 }
74 }

81 public static List<string> RebuildParamSettings(Animator a, ref ParameterSettings[] paraSettings, ref int paramCount, ParameterDefaults defs)
82 {
83
84#if UNITY_EDITOR
85 var ac = a.GetController();
86 var parms = ac.parameters;
87#else
88 var parms = a.parameters;
89#endif
90 rebuiltHashes.Clear();
91 rebuiltSettings.Clear();
92#if UNITY_EDITOR
93 reusableNameList.Clear();
94#endif
95 bool hasChanged = false;
96
97 paramCount = parms.Length;
98 for (int i = 0; i < paramCount; ++i)
99 {
100 var p = parms[i];
101#if UNITY_EDITOR
102 reusableNameList.Add(p.name);
103#endif
104 var hash = p.nameHash;
105 int pos = GetHashIndex(paraSettings, hash);
106 if (pos != i)
107 hasChanged = true;
109 rebuiltHashes.Add(hash);
110 rebuiltSettings.Add((pos == -1) ? new ParameterSettings(hash, defs, ref paramCount, p.type) : paraSettings[pos]);
111 }
112
113 if (hasChanged)
114 paraSettings = rebuiltSettings.ToArray();
115
116#if UNITY_EDITOR
117 return reusableNameList;
118#else
119 return null;
120#endif
121 }
122
123 private static int GetHashIndex(ParameterSettings[] ps, int lookfor)
124 {
125 for (int i = 0, cnt = ps.Length; i < cnt; ++i)
126 if (ps[i].hash == lookfor)
127 return i;
128 return -1;
129 }
