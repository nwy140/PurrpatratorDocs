241 void ProcessInputs()
242 {
243 if (Input.GetButtonDown("Fire1"))
244 {
245 // we don't want to fire when we interact with UI buttons for example. IsPointerOverGameObject really means IsPointerOver*UI*GameObject
246 // notice we don't use on on GetbuttonUp() few lines down, because one can mouse down, move over a UI element and release, which would lead to not lower the isFiring Flag.
247 if (EventSystem.current.IsPointerOverGameObject())
248 {
249 // return;
250 }
251
252 if (!this.IsFiring)
253 {
254 this.IsFiring = true;
255 }
256 }
257
258 if (Input.GetButtonUp("Fire1"))
259 {
260 if (this.IsFiring)
261 {
262 this.IsFiring = false;
263 }
264 }
265 }

271 public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
272 {
273 if (stream.IsWriting)
274 {
275 // We own this player: send the others our data
276 stream.SendNext(this.IsFiring);
277 stream.SendNext(this.Health);
278 }
279 else
280 {
281 // Network player, receive data
282 this.IsFiring = (bool)stream.ReceiveNext();
283 this.Health = (float)stream.ReceiveNext();
284 }
285 }
