136 protected void FixedUpdate()
137 {
138 #if PUN_DISPATCH_IN_FIXEDUPDATE
139 this.Dispatch();
140 #elif PUN_DISPATCH_IN_LATEUPDATE
141 // do not dispatch here
142 #else

151 protected void LateUpdate()
152 {
153 #if PUN_DISPATCH_IN_LATEUPDATE
154 this.Dispatch();
155 #elif PUN_DISPATCH_IN_FIXEDUPDATE
156 // do not dispatch here
157 #else
158 // see MinimalTimeScaleToDispatchInFixedUpdate and FixedUpdate for explanation:
159 if (Time.timeScale <= PhotonNetwork.MinimalTimeScaleToDispatchInFixedUpdate)
160 {
161 this.Dispatch();
162 }
163 #endif
164
165 int currentMsSinceStart = (int)(Time.realtimeSinceStartup * 1000); // avoiding Environment.TickCount, which could be negative on long-running platforms
166 if (PhotonNetwork.IsMessageQueueRunning && currentMsSinceStart > this.nextSendTickCountOnSerialize)
167 {
168 PhotonNetwork.RunViewUpdate();
169 this.nextSendTickCountOnSerialize = currentMsSinceStart + this.UpdateIntervalOnSerialize - SerializeRateFrameCorrection;
170 this.nextSendTickCount = 0; // immediately send when synchronization code was running
171 }
172
173 currentMsSinceStart = (int)(Time.realtimeSinceStartup * 1000);
174 if (SendAsap || currentMsSinceStart > this.nextSendTickCount)
175 {
176 SendAsap = false;
177 bool doSend = true;
178 int sendCounter = 0;
179 while (PhotonNetwork.IsMessageQueueRunning && doSend && sendCounter < MaxDatagrams)
180 {
181 // Send all outgoing commands
182 Profiler.BeginSample("SendOutgoingCommands");
183 doSend = PhotonNetwork.NetworkingClient.LoadBalancingPeer.SendOutgoingCommands();
184 sendCounter++;
185 Profiler.EndSample();
186 }
187
188 this.nextSendTickCount = currentMsSinceStart + this.UpdateInterval;
189 }
190 }

279 public void OnJoinedRoom()
280 {
281
282 if (PhotonNetwork.ViewCount == 0)
283 return;
284
285 var views = PhotonNetwork.PhotonViewCollection;
286
287 bool amMasterClient = PhotonNetwork.IsMasterClient;
288 bool amRejoiningMaster = amMasterClient && PhotonNetwork.CurrentRoom.PlayerCount > 1;
289
290 if (amRejoiningMaster)
291 reusableIntList.Clear();
292
293 // If this is the master rejoining, reassert ownership of non-creator owners
294 foreach (var view in views)
295 {
296 int viewOwnerId = view.OwnerActorNr;
297 int viewCreatorId = view.CreatorActorNr;
298
299 // on join / rejoin, assign control to either the Master Client (for room objects) or the owner (for anything else)
300 view.RebuildControllerCache();
301
302 // Rejoining master should enforce its world view, and override any changes that happened while it was soft disconnected
303 if (amRejoiningMaster)
304 if (viewOwnerId != viewCreatorId)
305 {
306 reusableIntList.Add(view.ViewID);
307 reusableIntList.Add(viewOwnerId);
308 }
309 }
310
311 if (amRejoiningMaster && reusableIntList.Count > 0)
312 {
313 PhotonNetwork.OwnershipUpdate(reusableIntList.ToArray());
314 }
315 }

324 public void OnPlayerEnteredRoom(Player newPlayer)
325 {
326 // note: if the master client becomes inactive, someone else becomes master. so there is no case where the active master client reconnects
327 // what may happen is that the Master Client disconnects locally and uses ReconnectAndRejoin before anyone (including the server) notices.
328
329 bool amMasterClient = PhotonNetwork.IsMasterClient;
330
331 var views = PhotonNetwork.PhotonViewCollection;
332 if (amMasterClient)
333 {
334 reusableIntList.Clear();
335 }
336
337 foreach (var view in views)
338 {
339 view.RebuildControllerCache(); // all clients will potentially have to clean up owner and controller, if someone re-joins
340
341 // the master client notifies joining players of any non-creator ownership
342 if (amMasterClient)
343 {
344 int viewOwnerId = view.OwnerActorNr;
345 if (viewOwnerId != view.CreatorActorNr)
346 {
347 reusableIntList.Add(view.ViewID);
348 reusableIntList.Add(viewOwnerId);
349 }
350 }
351 }
352
353 // update the joining player of non-creator ownership in the room
354 if (amMasterClient && reusableIntList.Count > 0)
355 {
356 PhotonNetwork.OwnershipUpdate(reusableIntList.ToArray(), newPlayer.ActorNumber);
357 }
358
359 }

361 public void OnPlayerLeftRoom(Player otherPlayer)
362 {
363 var views = PhotonNetwork.PhotonViewCollection;
364
365 int leavingPlayerId = otherPlayer.ActorNumber;
366 bool isInactive = otherPlayer.IsInactive;
367
368 // SOFT DISCONNECT: A player has timed out to the relay but has not yet exceeded PlayerTTL and may reconnect.
369 // Master will take control of this objects until the player hard disconnects, or returns.
370 if (isInactive)
371 {
372 foreach (var view in views)
373 {
374 // v2.27: changed from owner-check to controller-check
375 if (view.ControllerActorNr == leavingPlayerId)
376 view.ControllerActorNr = PhotonNetwork.MasterClient.ActorNumber;
377 }
378
379 }
380 // HARD DISCONNECT: Player permanently removed. Remove that actor as owner for all items they created (Unless AutoCleanUp is false)
381 else
382 {
383 bool autocleanup = PhotonNetwork.CurrentRoom.AutoCleanUp;
384
385 foreach (var view in views)
386 {
387 // Skip changing Owner/Controller for items that will be cleaned up.
388 if (autocleanup && view.CreatorActorNr == leavingPlayerId)
389 continue;
390
391 // Any views owned by the leaving player, default to null owner (which will become master controlled).
392 if (view.OwnerActorNr == leavingPlayerId || view.ControllerActorNr == leavingPlayerId)
393 {
394 view.OwnerActorNr = 0;
395 view.ControllerActorNr = PhotonNetwork.MasterClient.ActorNumber;
396 }
397 }
398 }
