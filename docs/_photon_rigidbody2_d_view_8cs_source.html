18 [AddComponentMenu("Photon Networking/Photon Rigidbody 2D View")]
19 public class PhotonRigidbody2DView : MonoBehaviourPun, IPunObservable

40 public void Awake()
41 {
42 this.m_Body = GetComponent<Rigidbody2D>();
43
44 this.m_NetworkPosition = new Vector2();
45 }
46
47 public void FixedUpdate()
48 {
49 if (!this.photonView.IsMine)
50 {
51 this.m_Body.position = Vector2.MoveTowards(this.m_Body.position, this.m_NetworkPosition, this.m_Distance * (1.0f / PhotonNetwork.SerializationRate));
52 this.m_Body.rotation = Mathf.MoveTowards(this.m_Body.rotation, this.m_NetworkRotation, this.m_Angle * (1.0f / PhotonNetwork.SerializationRate));
53 }
54 }
55
56 public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info)
57 {
58 if (stream.IsWriting)
59 {
60 stream.SendNext(this.m_Body.position);
61 stream.SendNext(this.m_Body.rotation);
62
63 if (this.m_SynchronizeVelocity)
64 {
65 stream.SendNext(this.m_Body.velocity);
66 }
67
68 if (this.m_SynchronizeAngularVelocity)
69 {
70 stream.SendNext(this.m_Body.angularVelocity);
71 }
72 }
73 else
74 {
75 this.m_NetworkPosition = (Vector2)stream.ReceiveNext();
76 this.m_NetworkRotation = (float)stream.ReceiveNext();
77
78 if (this.m_TeleportEnabled)
79 {
80 if (Vector3.Distance(this.m_Body.position, this.m_NetworkPosition) > this.m_TeleportIfDistanceGreaterThan)
81 {
82 this.m_Body.position = this.m_NetworkPosition;
83 }
84 }
85
86 if (this.m_SynchronizeVelocity || this.m_SynchronizeAngularVelocity)
87 {
88 float lag = Mathf.Abs((float)(PhotonNetwork.Time - info.SentServerTime));
89
90 if (m_SynchronizeVelocity)
91 {
92 this.m_Body.velocity = (Vector2)stream.ReceiveNext();
93
94 this.m_NetworkPosition += this.m_Body.velocity * lag;
95
96 this.m_Distance = Vector2.Distance(this.m_Body.position, this.m_NetworkPosition);
97 }
98
99 if (this.m_SynchronizeAngularVelocity)
100 {
101 this.m_Body.angularVelocity = (float)stream.ReceiveNext();
102
103 this.m_NetworkRotation += this.m_Body.angularVelocity * lag;
104
105 this.m_Angle = Mathf.Abs(this.m_Body.rotation - this.m_NetworkRotation);
106 }
107 }
108 }
109 }
110 }
