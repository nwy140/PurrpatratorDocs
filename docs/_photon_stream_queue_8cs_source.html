127 public object ReceiveNext()

146 public void Serialize(PhotonStream stream)
147 {
148 // TODO: find a better solution for this:
149 // the "if" is a workaround for packages which have only 1 sample/frame. in that case, SendNext didn't set the obj per sample.
150 if (this.m_Objects.Count > 0 && this.m_ObjectsPerSample < 0)

152 this.m_ObjectsPerSample = this.m_Objects.Count;

155 stream.SendNext(this.m_SampleCount);

173 public void Deserialize(PhotonStream stream)
174 {
175 this.m_Objects.Clear();
176
177 this.m_SampleCount = (int) stream.ReceiveNext();
178 this.m_ObjectsPerSample = (int) stream.ReceiveNext();
179
180 for (int i = 0; i < this.m_SampleCount * this.m_ObjectsPerSample; ++i)
181 {
182 this.m_Objects.Add(stream.ReceiveNext());
183 }
184
185 if (this.m_Objects.Count > 0)

187 this.m_NextObjectIndex = 0;
