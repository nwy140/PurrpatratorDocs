146 void IInRoomCallbacks.OnPlayerPropertiesUpdate(Player targetPlayer, Hashtable changedProps)
147 {
148 object temp;
149 if (changedProps.TryGetValue(TeamPlayerProp, out temp))
150 {
151 if (temp == null)
152 {
153 foreach (byte code in playersPerTeam.Keys)
154 {
155 if (playersPerTeam[code].Remove(targetPlayer))
156 {
157 if (PlayerLeftTeam != null)
158 {
159 PlayerLeftTeam(targetPlayer, teamsByCode[code]);
160 }
161 break;
162 }
163 }
164 }
165 else if (temp is byte)
166 {
167 byte teamCode = (byte) temp;
168 // check if player switched teams, remove from previous team
169 foreach (byte code in playersPerTeam.Keys)
170 {
171 if (code == teamCode)
172 {
173 continue;
174 }
175 if (playersPerTeam[code].Remove(targetPlayer))
176 {
177 if (PlayerLeftTeam != null)
178 {
179 PlayerLeftTeam(targetPlayer, teamsByCode[code]);
180 }
181 break;
182 }
183 }
184 PhotonTeam team = teamsByCode[teamCode];
185 if (!playersPerTeam[teamCode].Add(targetPlayer))
186 {
187 Debug.LogWarningFormat("Unexpected situation while setting team {0} for player {1}, updating teams for all", team, targetPlayer);
188 this.UpdateTeams();
189 }
190 if (PlayerJoinedTeam != null)
191 {
192 PlayerJoinedTeam(targetPlayer, team);
193 }
194 }
195 else
196 {
197 Debug.LogErrorFormat("Unexpected: custom property key {0} should have of type byte, instead we got {1} of type {2}. Player: {3}",
198 TeamPlayerProp, temp, temp.GetType(), targetPlayer);
199 }
200 }

203 void IInRoomCallbacks.OnPlayerLeftRoom(Player otherPlayer)
204 {
205 if (otherPlayer.IsInactive)
206 {
207 return;
208 }
209 PhotonTeam team = otherPlayer.GetPhotonTeam();
210 if (team != null && !playersPerTeam[team.Code].Remove(otherPlayer))
211 {
212 Debug.LogWarningFormat("Unexpected situation while removing player {0} who left from team {1}, updating teams for all", otherPlayer, team);
213 // revert to 'brute force' in case of unexpected situation
214 this.UpdateTeams();
215 }
216 }

218 void IInRoomCallbacks.OnPlayerEnteredRoom(Player newPlayer)
219 {
220 PhotonTeam team = newPlayer.GetPhotonTeam();
221 if (team == null)
222 {
223 return;
224 }
225 if (playersPerTeam[team.Code].Contains(newPlayer))
226 {
227 // player rejoined w/ same team
228 return;
229 }
230 // check if player rejoined w/ different team, remove from previous team
231 foreach (var key in teamsByCode.Keys)
232 {
233 if (playersPerTeam[key].Remove(newPlayer))
234 {
235 break;
236 }
237 }
238 if (!playersPerTeam[team.Code].Add(newPlayer))
239 {
240 Debug.LogWarningFormat("Unexpected situation while adding player {0} who joined to team {1}, updating teams for all", newPlayer, team);
241 // revert to 'brute force' in case of unexpected situation
242 this.UpdateTeams();
243 }

250 private void UpdateTeams()
251 {
252 this.ClearTeams();
253 for (int i = 0; i < PhotonNetwork.PlayerList.Length; i++)
254 {
255 Player player = PhotonNetwork.PlayerList[i];
256 PhotonTeam playerTeam = player.GetPhotonTeam();
257 if (playerTeam != null)
258 {
259 playersPerTeam[playerTeam.Code].Add(player);
260 }
261 }
262 }

264 private void ClearTeams()
265 {

268 playersPerTeam[key].Clear();
269 }

374 public bool TryGetTeamMatesOfPlayer(Player player, out Player[] teamMates)
375 {
376 teamMates = null;
377 if (player == null)
378 {
379 return false;
380 }
381 PhotonTeam team = player.GetPhotonTeam();
382 if (team == null)
383 {
384 return false;
385 }
386 HashSet<Player> players;
387 if (this.playersPerTeam.TryGetValue(team.Code, out players))
388 {
389 if (!players.Contains(player))
390 {
391 Debug.LogWarningFormat("Unexpected situation while getting team mates of player {0} who is joined to team {1}, updating teams for all", player, team);
392 // revert to 'brute force' in case of unexpected situation
393 this.UpdateTeams();
394 }
395 teamMates = new Player[players.Count - 1];
396 int i = 0;
397 foreach (var p in players)
398 {
399 if (p.Equals(player))
400 {
401 continue;
402 }
403 teamMates[i] = p;
404 i++;
405 }
406 return true;
407 }
408 return false;
409 }
