248 public Vector3 UpdatePosition(Vector3 currentPosition)
249 {
250 Vector3 targetPosition = GetNetworkPosition() + GetExtrapolatedPositionOffset();
251
252 switch (m_Model.InterpolateOption)
253 {
254 case PhotonTransformViewPositionModel.InterpolateOptions.Disabled:
255 if (m_UpdatedPositionAfterOnSerialize == false)
256 {
257 currentPosition = targetPosition;
258 m_UpdatedPositionAfterOnSerialize = true;
259 }
260
261 break;
262
263 case PhotonTransformViewPositionModel.InterpolateOptions.FixedSpeed:
264 currentPosition = Vector3.MoveTowards(currentPosition, targetPosition, Time.deltaTime * m_Model.InterpolateMoveTowardsSpeed);
265 break;
266
267 case PhotonTransformViewPositionModel.InterpolateOptions.EstimatedSpeed:
268 if (m_OldNetworkPositions.Count == 0)
269 {
270 // special case: we have no previous updates in memory, so we can't guess a speed!
271 break;
272 }
273
274 // knowing the last (incoming) position and the one before, we can guess a speed.
275 // note that the speed is times sendRateOnSerialize! we send X updates/sec, so our estimate has to factor that in.
276 float estimatedSpeed = (Vector3.Distance(m_NetworkPosition, GetOldestStoredNetworkPosition()) / m_OldNetworkPositions.Count) * PhotonNetwork.SerializationRate;
277
278 // move towards the targetPosition (including estimates, if that's active) with the speed calculated from the last updates.
279 currentPosition = Vector3.MoveTowards(currentPosition, targetPosition, Time.deltaTime * estimatedSpeed);
280 break;
281
282 case PhotonTransformViewPositionModel.InterpolateOptions.SynchronizeValues:
283 if (m_SynchronizedSpeed.magnitude == 0)
284 {
285 currentPosition = targetPosition;
286 }
287 else
288 {
289 currentPosition = Vector3.MoveTowards(currentPosition, targetPosition, Time.deltaTime * m_SynchronizedSpeed.magnitude);
290 }
291
292 break;
293
294 case PhotonTransformViewPositionModel.InterpolateOptions.Lerp:
295 currentPosition = Vector3.Lerp(currentPosition, targetPosition, Time.deltaTime * m_Model.InterpolateLerpSpeed);
296 break;
297 }
298
299 if (m_Model.TeleportEnabled == true)
300 {
301 if (Vector3.Distance(currentPosition, GetNetworkPosition()) > m_Model.TeleportIfDistanceGreaterThan)
302 {
303 currentPosition = GetNetworkPosition();
304 }
305 }
306
307 return currentPosition;
308 }

314 public Vector3 GetNetworkPosition()
315 {
316 return m_NetworkPosition;
317 }

324 public Vector3 GetExtrapolatedPositionOffset()
325 {
326 float timePassed = (float)(PhotonNetwork.Time - m_LastSerializeTime);
327
328 if (m_Model.ExtrapolateIncludingRoundTripTime == true)
329 {
330 timePassed += (float)PhotonNetwork.GetPing() / 1000f;
331 }
332
333 Vector3 extrapolatePosition = Vector3.zero;
334
335 switch (m_Model.ExtrapolateOption)
336 {
337 case PhotonTransformViewPositionModel.ExtrapolateOptions.SynchronizeValues:
338 Quaternion turnRotation = Quaternion.Euler(0, m_SynchronizedTurnSpeed * timePassed, 0);
339 extrapolatePosition = turnRotation * (m_SynchronizedSpeed * timePassed);
340 break;
341 case PhotonTransformViewPositionModel.ExtrapolateOptions.FixedSpeed:
342 Vector3 moveDirection = (m_NetworkPosition - GetOldestStoredNetworkPosition()).normalized;
343
344 extrapolatePosition = moveDirection * m_Model.ExtrapolateSpeed * timePassed;
345 break;
346 case PhotonTransformViewPositionModel.ExtrapolateOptions.EstimateSpeedAndTurn:
347 Vector3 moveDelta = (m_NetworkPosition - GetOldestStoredNetworkPosition()) * PhotonNetwork.SerializationRate;
348 extrapolatePosition = moveDelta * timePassed;
349 break;
350 }
351
352 return extrapolatePosition;
353 }

355 public void OnPhotonSerializeView(Vector3 currentPosition, PhotonStream stream, PhotonMessageInfo info)
356 {
357 if (m_Model.SynchronizeEnabled == false)
358 {
359 return;
360 }
361
362 if (stream.IsWriting == true)
363 {
364 SerializeData(currentPosition, stream, info);
365 }
366 else
367 {
368 DeserializeData(stream, info);
369 }
370
371 m_LastSerializeTime = PhotonNetwork.Time;
372 m_UpdatedPositionAfterOnSerialize = false;
373 }
374
375 void SerializeData(Vector3 currentPosition, PhotonStream stream, PhotonMessageInfo info)
376 {
377 stream.SendNext(currentPosition);
378 m_NetworkPosition = currentPosition;
379
380 if (m_Model.ExtrapolateOption == PhotonTransformViewPositionModel.ExtrapolateOptions.SynchronizeValues ||
381 m_Model.InterpolateOption == PhotonTransformViewPositionModel.InterpolateOptions.SynchronizeValues)
382 {
383 stream.SendNext(m_SynchronizedSpeed);
384 stream.SendNext(m_SynchronizedTurnSpeed);
385 }
386 }
387
388 void DeserializeData(PhotonStream stream, PhotonMessageInfo info)
389 {
390 Vector3 readPosition = (Vector3)stream.ReceiveNext();
391 if (m_Model.ExtrapolateOption == PhotonTransformViewPositionModel.ExtrapolateOptions.SynchronizeValues ||
392 m_Model.InterpolateOption == PhotonTransformViewPositionModel.InterpolateOptions.SynchronizeValues)
393 {
394 m_SynchronizedSpeed = (Vector3)stream.ReceiveNext();
395 m_SynchronizedTurnSpeed = (float)stream.ReceiveNext();
396 }
397
398 if (m_OldNetworkPositions.Count == 0)
399 {
400 // if we don't have old positions yet, this is the very first update this client reads. let's use this as current AND old position.
401 m_NetworkPosition = readPosition;
402 }
403
404 // the previously received position becomes the old(er) one and queued. the new one is the m_NetworkPosition
405 m_OldNetworkPositions.Enqueue(m_NetworkPosition);
406 m_NetworkPosition = readPosition;
407
408 // reduce items in queue to defined number of stored positions.
409 while (m_OldNetworkPositions.Count > m_Model.ExtrapolateNumberOfStoredPositions)
410 {
411 m_OldNetworkPositions.Dequeue();
412 }
413 }
