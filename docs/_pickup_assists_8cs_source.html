81 public static GameObject ConvertToPickup(GameObject selection, Space_XD space, Dynamics dynamics, params System.Type[] allowedSystems)
82 {
83 selection = NetObjectAssists.ConvertToBasicNetObject(selection, Photon.Pun.OwnershipOption.Takeover);
84
85 selection.EnsureComponentExists<ContactTrigger>();
86
87 selection.EnsureComponentExists<SyncContact>();
88 var ss = selection.EnsureComponentExists<SyncState>();
89 ss.mountableTo.mask = MountSettings.AllTrueMask;
90 selection.EnsureComponentExists<SyncOwner>();
91
92 if (dynamics != Dynamics.Static)
93 selection.EnsureComponentExists<OnStateChangeKinematic>();
94
96 selection.EnsureComponentOnNestedChildren<OnStateChangeToggle>(false);
97
99 var hga = selection.EnsureComponentExists<ContactGroupAssign>();
100 hga.contactGroups.Mask = 0;
101
102 if (dynamics != Dynamics.Static)
103 {
104 selection.AddRigidbody(space);
105 var st = selection.EnsureComponentExists<SyncTransform>();
106 st.transformCrusher.SclCrusher.Enabled = false;
107
108 }
109
110 selection.CreateChildStatePlaceholders(space, dynamics, 1.5f);
111
112 return selection;
113 }
