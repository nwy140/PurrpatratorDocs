126 public void RefreshData()
127 {
128 if (PhotonNetwork.CurrentRoom == null)
129 {
130 return;
131 }
132
133 if (PhotonNetwork.LocalPlayer.GetPlayerNumber() >= 0)
134 {
135 SortedPlayers = PhotonNetwork.CurrentRoom.Players.Values.OrderBy((p) => p.GetPlayerNumber()).ToArray();
136 if (OnPlayerNumberingChanged != null)
137 {
138 OnPlayerNumberingChanged();
139 }
140 return;
141 }
142
143
144 HashSet<int> usedInts = new HashSet<int>();
145 Player[] sorted = PhotonNetwork.PlayerList.OrderBy((p) => p.ActorNumber).ToArray();
146
147 string allPlayers = "all players: ";
148 foreach (Player player in sorted)
149 {
150 allPlayers += player.ActorNumber + "=pNr:"+player.GetPlayerNumber()+", ";
151
152 int number = player.GetPlayerNumber();
153
154 // if it's this user, select a number and break
155 // else:
156 // check if that user has a number
157 // if not, break!
158 // else remember used numbers
159
160 if (player.IsLocal)
161 {
162 Debug.Log ("PhotonNetwork.CurrentRoom.PlayerCount = " + PhotonNetwork.CurrentRoom.PlayerCount);
163
164 // select a number
165 for (int i = 0; i < PhotonNetwork.CurrentRoom.PlayerCount; i++)
166 {
167 if (!usedInts.Contains(i))
168 {
169 player.SetPlayerNumber(i);
170 break;
171 }
172 }
173 // then break
174 break;
175 }
176 else
177 {
178 if (number < 0)
179 {
180 break;
181 }
182 else
183 {
184 usedInts.Add(number);
185 }
186 }
187 }
188
189 //Debug.Log(allPlayers);
190 //Debug.Log(PhotonNetwork.LocalPlayer.ToStringFull() + " has PhotonNetwork.player.GetPlayerNumber(): " + PhotonNetwork.LocalPlayer.GetPlayerNumber());
191
192 SortedPlayers = PhotonNetwork.CurrentRoom.Players.Values.OrderBy((p) => p.GetPlayerNumber()).ToArray();
193 if (OnPlayerNumberingChanged != null)
194 {
195 OnPlayerNumberingChanged();
196 }
197 }

203 public static class PlayerNumberingExtensions
204 {
209 public static int GetPlayerNumber(this Player player)
210 {
211 if (player == null) {
212 return -1;
213 }
214
215 if (PhotonNetwork.OfflineMode)
216 {
217 return 0;
218 }
219 if (!PhotonNetwork.IsConnectedAndReady)
220 {
221 return -1;
222 }
223
224 object value;
225 if (player.CustomProperties.TryGetValue (PlayerNumbering.RoomPlayerIndexedProp, out value)) {
226 return (byte)value;
227 }
228 return -1;
229 }
230
237 public static void SetPlayerNumber(this Player player, int playerNumber)
238 {
239 if (player == null) {
240 return;
241 }
242
243 if (PhotonNetwork.OfflineMode)
244 {
245 return;
246 }
247
248 if (playerNumber < 0)
249 {
250 Debug.LogWarning("Setting invalid playerNumber: " + playerNumber + " for: " + player.ToStringFull());
251 }
252
253 if (!PhotonNetwork.IsConnectedAndReady)
254 {
255 Debug.LogWarning("SetPlayerNumber was called in state: " + PhotonNetwork.NetworkClientState + ". Not IsConnectedAndReady.");
256 return;
257 }
258
259 int current = player.GetPlayerNumber();
260 if (current != playerNumber)
261 {
262 Debug.Log("PlayerNumbering: Set number "+playerNumber);
263 player.SetCustomProperties(new Hashtable() { { PlayerNumbering.RoomPlayerIndexedProp, (byte)playerNumber } });
264 }
265 }
266 }
