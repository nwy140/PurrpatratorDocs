22 public class PlayerOverviewPanel : MonoBehaviourPunCallbacks

25
26 private Dictionary<int, GameObject> playerListEntries;
27
28 #region UNITY
29
30 public void Awake()
31 {
32 playerListEntries = new Dictionary<int, GameObject>();
33
34 foreach (Player p in PhotonNetwork.PlayerList)
35 {
36 GameObject entry = Instantiate(PlayerOverviewEntryPrefab);
37 entry.transform.SetParent(gameObject.transform);
38 entry.transform.localScale = Vector3.one;
39 entry.GetComponent<Text>().color = AsteroidsGame.GetColor(p.GetPlayerNumber());
40 entry.GetComponent<Text>().text = string.Format("{0}\nScore: {1}\nLives: {2}", p.NickName, p.GetScore(), AsteroidsGame.PLAYER_MAX_LIVES);
41
42 playerListEntries.Add(p.ActorNumber, entry);
43 }
44 }
45
46 #endregion
47
48 #region PUN CALLBACKS
49
50 public override void OnPlayerLeftRoom(Player otherPlayer)
51 {
52 GameObject go = null;
53 if (this.playerListEntries.TryGetValue(otherPlayer.ActorNumber, out go))
54 {
55 Destroy(playerListEntries[otherPlayer.ActorNumber]);
56 playerListEntries.Remove(otherPlayer.ActorNumber);
57 }
58 }
59
60 public override void OnPlayerPropertiesUpdate(Player targetPlayer, Hashtable changedProps)
61 {
62 GameObject entry;
63 if (playerListEntries.TryGetValue(targetPlayer.ActorNumber, out entry))
64 {
65 entry.GetComponent<Text>().text = string.Format("{0}\nScore: {1}\nLives: {2}", targetPlayer.NickName, targetPlayer.GetScore(), targetPlayer.CustomProperties[AsteroidsGame.PLAYER_LIVES]);
66 }
67 }
68
69 #endregion
70 }
