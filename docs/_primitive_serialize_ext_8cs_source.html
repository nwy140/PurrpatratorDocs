363 public static ulong Write(this ulong buffer, ulong value, ref int bitposition, int bits = 64)
364 {
365 System.Diagnostics.Debug.Assert(bitposition + bits <= 64, overrunerror);
366
367 ulong offsetvalue = value << bitposition;
368 ulong mask = ulong.MaxValue >> (64 - bits) << bitposition;
369
370 // Clear bits in buffer we need to write to, then write to them.
371 buffer &= ~mask;
372 buffer |= (mask & offsetvalue);
373
374 bitposition += bits;
375 return buffer;
376 }
377

388 {

390

392 uint mask = uint.MaxValue >> (32 - bits) << bitposition;
393
394 // Clear bits in buffer we need to write to, then write to them.
395 buffer &= ~mask;

398 bitposition += bits;

400 }

411 public static ushort Write(this ushort buffer, ulong value, ref int bitposition, int bits = 64)
412 {
413 System.Diagnostics.Debug.Assert(bitposition + bits <= 16, overrunerror);
414
415 uint offsetvalue = ((uint)value << bitposition);
416 uint mask = ((uint)ushort.MaxValue >> (16 - bits) << bitposition);
417
418 // Clear bits in buffer we need to write to, then write to them.
419 uint _target = buffer & ~mask;
420 _target |= (mask & offsetvalue);
421 buffer = (ushort)_target;
422
423 bitposition += bits;
