857 public class DefaultPool : IPunPrefabPool
858 {
860 public readonly Dictionary<string, GameObject> ResourceCache = new Dictionary<string, GameObject>();
861
867 public GameObject Instantiate(string prefabId, Vector3 position, Quaternion rotation)
868 {
869 GameObject res = null;
870 bool cached = this.ResourceCache.TryGetValue(prefabId, out res);
871 if (!cached)
872 {
873 res = Resources.Load<GameObject>(prefabId);
874 if (res == null)
875 {
876 Debug.LogError("DefaultPool failed to load \"" + prefabId + "\". Make sure it's in a \"Resources\" folder. Or use a custom IPunPrefabPool.");
877 }
878 else
879 {
880 this.ResourceCache.Add(prefabId, res);
881 }
882 }
883
884 bool wasActive = res.activeSelf;
885 if (wasActive) res.SetActive(false);
886
887 GameObject instance =GameObject.Instantiate(res, position, rotation) as GameObject;
888
889 if (wasActive) res.SetActive(true);
890 return instance;
891 }

895 public void Destroy(GameObject gameObject)
896 {
897 GameObject.Destroy(gameObject);
898 }
