167 public void SendMove(object move, bool finished)
168 {
169 if (IsFinishedByMe)
170 {
171 UnityEngine.Debug.LogWarning("Can't SendMove. Turn is finished by this player.");
172 return;
173 }
174
175 // along with the actual move, we have to send which turn this move belongs to
176 Hashtable moveHt = new Hashtable();
177 moveHt.Add("turn", Turn);
178 moveHt.Add("move", move);
179
180 byte evCode = (finished) ? EvFinalMove : EvMove;
181 PhotonNetwork.RaiseEvent(evCode, moveHt, new RaiseEventOptions() {CachingOption = EventCaching.AddToRoomCache}, SendOptions.SendReliable);
182 if (finished)
183 {
184 PhotonNetwork.LocalPlayer.SetFinishedTurn(Turn);
185 }
186
187 // the server won't send the event back to the origin (by default). to get the event, call it locally
188 // (note: the order of events might be mixed up as we do this locally)
189 ProcessOnEvent(evCode, moveHt, PhotonNetwork.LocalPlayer.ActorNumber);
190 }

207 #region Callbacks

209 // called internally
210 void ProcessOnEvent(byte eventCode, object content, int senderId)
211 {
212 if (senderId == -1)
213 {
214 return;
215 }
216
217 sender = PhotonNetwork.CurrentRoom.GetPlayer(senderId);
218
219 switch (eventCode)
220 {
221 case EvMove:
222 {
223 Hashtable evTable = content as Hashtable;
224 int turn = (int)evTable["turn"];
225 object move = evTable["move"];
226 this.TurnManagerListener.OnPlayerMove(sender, turn, move);
227
228 break;
229 }
230 case EvFinalMove:
231 {
232 Hashtable evTable = content as Hashtable;
233 int turn = (int)evTable["turn"];
234 object move = evTable["move"];
235
236 if (turn == this.Turn)
237 {
238 this.finishedPlayers.Add(sender);
239
240 this.TurnManagerListener.OnPlayerFinished(sender, turn, move);
241
242 }
243
244 if (IsCompletedByAll)
245 {
246 this.TurnManagerListener.OnTurnCompleted(this.Turn);
247 }
248 break;
249 }
250 }
251 }

398 public static int GetFinishedTurn(this Player player)
399 {

403 return 0;
