171 private void Move()
172 {
173 // set target speed based on move speed, sprint speed and if sprint is pressed
174 float targetSpeed = _input.sprint ? SprintSpeed : MoveSpeed;
175
176 // a simplistic acceleration and deceleration designed to be easy to remove, replace, or iterate upon
177
178 // note: Vector2's == operator uses approximation so is not floating point error prone, and is cheaper than magnitude
179 // if there is no input, set the target speed to 0
180 if (_input.move == Vector2.zero) targetSpeed = 0.0f;
181
182 // a reference to the players current horizontal velocity
183 float currentHorizontalSpeed = new Vector3(_controller.velocity.x, 0.0f, _controller.velocity.z).magnitude;
184
185 float speedOffset = 0.1f;
186 float inputMagnitude = _input.analogMovement ? _input.move.magnitude : 1f;
187
188 // accelerate or decelerate to target speed
189 if (currentHorizontalSpeed < targetSpeed - speedOffset || currentHorizontalSpeed > targetSpeed + speedOffset)
190 {
191 // creates curved result rather than a linear one giving a more organic speed change
192 // note T in Lerp is clamped, so we don't need to clamp our speed
193 _speed = Mathf.Lerp(currentHorizontalSpeed, targetSpeed * inputMagnitude, Time.deltaTime * SpeedChangeRate);
194
195 // round speed to 3 decimal places
196 _speed = Mathf.Round(_speed * 1000f) / 1000f;
197 }
198 else
199 {
200 _speed = targetSpeed;
201 }
202 _animationBlend = Mathf.Lerp(_animationBlend, targetSpeed, Time.deltaTime * SpeedChangeRate);
203
204 // normalise input direction
205 Vector3 inputDirection = new Vector3(_input.move.x, 0.0f, _input.move.y).normalized;
206
207 // note: Vector2's != operator uses approximation so is not floating point error prone, and is cheaper than magnitude
208 // if there is a move input rotate player when the player is moving
209 if (_input.move != Vector2.zero)
210 {
211 _targetRotation = Mathf.Atan2(inputDirection.x, inputDirection.z) * Mathf.Rad2Deg + _mainCamera.transform.eulerAngles.y;
212 float rotation = Mathf.SmoothDampAngle(transform.eulerAngles.y, _targetRotation, ref _rotationVelocity, RotationSmoothTime);
213
214 // rotate to face input direction relative to camera position
215 transform.rotation = Quaternion.Euler(0.0f, rotation, 0.0f);
216 }
217
218
219 Vector3 targetDirection = Quaternion.Euler(0.0f, _targetRotation, 0.0f) * Vector3.forward;
220
221 // move the player
222 _controller.Move(targetDirection.normalized * (_speed * Time.deltaTime) + new Vector3(0.0f, _verticalVelocity, 0.0f) * Time.deltaTime);
223
224 // update animator if using character
225 if (_hasAnimator)
226 {
227 _animator.SetFloat(_animIDSpeed, _animationBlend);
228 _animator.SetFloat(_animIDMotionSpeed, inputMagnitude);
229 }
230 }

232 private void JumpAndGravity()
233 {
234 if (Grounded)
235 {
236 // reset the fall timeout timer
237 _fallTimeoutDelta = FallTimeout;
238
239 // update animator if using character
240 if (_hasAnimator)
241 {
242 _animator.SetBool(_animIDJump, false);
243 _animator.SetBool(_animIDFreeFall, false);
244 }
245
246 // stop our velocity dropping infinitely when grounded
247 if (_verticalVelocity < 0.0f)
248 {
249 _verticalVelocity = -2f;
250 }
251
252 // Jump
253 if (_input.jump && _jumpTimeoutDelta <= 0.0f)
254 {
255 // the square root of H * -2 * G = how much velocity needed to reach desired height
256 _verticalVelocity = Mathf.Sqrt(JumpHeight * -2f * Gravity);
257
258 // update animator if using character
259 if (_hasAnimator)
260 {
261 _animator.SetBool(_animIDJump, true);
262 }
263 }
264
265 // jump timeout
266 if (_jumpTimeoutDelta >= 0.0f)
267 {
268 _jumpTimeoutDelta -= Time.deltaTime;
269 }
270 }
271 else
272 {
273 // reset the jump timeout timer
274 _jumpTimeoutDelta = JumpTimeout;
275
276 // fall timeout
277 if (_fallTimeoutDelta >= 0.0f)
278 {
279 _fallTimeoutDelta -= Time.deltaTime;
280 }
281 else
282 {
283 // update animator if using character
284 if (_hasAnimator)
285 {
286 _animator.SetBool(_animIDFreeFall, true);
287 }
288 }
289
290 // if we are not grounded, do not jump
291 _input.jump = false;
292 }
293
294 // apply gravity over time if under terminal (multiply by delta time twice to linearly speed up over time)
295 if (_verticalVelocity < _terminalVelocity)
296 {
297 _verticalVelocity += Gravity * Time.deltaTime;
298 }
299 }
