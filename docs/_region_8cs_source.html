52 private void SetCodeAndCluster(string codeAsString)
53 {
54 if (codeAsString == null)
55 {
56 this.Code = "";
57 this.Cluster = "";
58 return;
59 }
60
61 codeAsString = codeAsString.ToLower();
62 int slash = codeAsString.IndexOf('/');
63 this.Code = slash <= 0 ? codeAsString : codeAsString.Substring(0, slash);
64 this.Cluster = slash <= 0 ? "" : codeAsString.Substring(slash+1, codeAsString.Length-slash-1);
65 }
66

72 public string ToString(bool compact = false)
73 {
74 string regionCluster = this.Code;

76 {
77 regionCluster += "/" + this.Cluster;

79
80 if (compact)
81 {
82 return string.Format("{0}:{1}", regionCluster, this.Ping);
83 }
84 else
85 {
86 return string.Format("{0}[{2}]: {1}ms", regionCluster, this.Ping, this.HostAndPort);
