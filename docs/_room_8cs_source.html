347 protected internal override void InternalCacheProperties(Hashtable propertiesToCache)
348 {
349 int oldMasterId = this.masterClientId;
350
351 base.InternalCacheProperties(propertiesToCache); // important: updating the properties fields has no way to do callbacks on change
352
353 if (oldMasterId != 0 && this.masterClientId != oldMasterId)
354 {
355 this.LoadBalancingClient.InRoomCallbackTargets.OnMasterClientSwitched(this.GetPlayer(this.masterClientId));
356 }
357 }

405 public virtual bool SetCustomProperties(Hashtable propertiesToSet, Hashtable expectedProperties = null, WebFlags webFlags = null)
406 {
407 if (propertiesToSet == null || propertiesToSet.Count == 0)
408 {
409 return false;
410 }
411 Hashtable customProps = propertiesToSet.StripToStringKeys() as Hashtable;
412
413 if (this.isOffline)
414 {
415 if (customProps.Count == 0)
416 {
417 return false;
418 }
419 // Merge and delete values.
420 this.CustomProperties.Merge(customProps);
421 this.CustomProperties.StripKeysWithNullValues();
422
423 // invoking callbacks
424 this.LoadBalancingClient.InRoomCallbackTargets.OnRoomPropertiesUpdate(propertiesToSet);
425
426 }
427 else
428 {
429 // send (sync) these new values if in online room
430 return this.LoadBalancingClient.OpSetPropertiesOfRoom(customProps, expectedProperties, webFlags);
431 }
432
433 return true;
434 }
