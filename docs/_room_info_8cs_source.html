199 protected internal virtual void InternalCacheProperties(Hashtable propertiesToCache)
200 {
201 if (propertiesToCache == null || propertiesToCache.Count == 0 || this.customProperties.Equals(propertiesToCache))
202 {
203 return;
204 }
205
206 // check of this game was removed from the list. in that case, we don't
207 // need to read any further properties
208 // list updates will remove this game from the game listing
209 if (propertiesToCache.ContainsKey(GamePropertyKey.Removed))
210 {
211 this.RemovedFromList = (bool)propertiesToCache[GamePropertyKey.Removed];
212 if (this.RemovedFromList)
213 {
214 return;
215 }
216 }
217
218 // fetch the "well known" properties of the room, if available
219 if (propertiesToCache.ContainsKey(GamePropertyKey.MaxPlayers))
220 {
221 this.maxPlayers = (byte)propertiesToCache[GamePropertyKey.MaxPlayers];
222 }
223
224 if (propertiesToCache.ContainsKey(GamePropertyKey.IsOpen))
225 {
226 this.isOpen = (bool)propertiesToCache[GamePropertyKey.IsOpen];
227 }
228
229 if (propertiesToCache.ContainsKey(GamePropertyKey.IsVisible))
230 {
231 this.isVisible = (bool)propertiesToCache[GamePropertyKey.IsVisible];
232 }
233
234 if (propertiesToCache.ContainsKey(GamePropertyKey.PlayerCount))
235 {
236 this.PlayerCount = (int)((byte)propertiesToCache[GamePropertyKey.PlayerCount]);
237 }
238
239 if (propertiesToCache.ContainsKey(GamePropertyKey.CleanupCacheOnLeave))
240 {
241 this.autoCleanUp = (bool)propertiesToCache[GamePropertyKey.CleanupCacheOnLeave];
242 }
243
244 if (propertiesToCache.ContainsKey(GamePropertyKey.MasterClientId))
245 {
246 this.masterClientId = (int)propertiesToCache[GamePropertyKey.MasterClientId];
247 }
248
249 if (propertiesToCache.ContainsKey(GamePropertyKey.PropsListedInLobby))
250 {
251 this.propertiesListedInLobby = propertiesToCache[GamePropertyKey.PropsListedInLobby] as string[];
252 }
253
254 if (propertiesToCache.ContainsKey((byte)GamePropertyKey.ExpectedUsers))
255 {
256 this.expectedUsers = (string[])propertiesToCache[GamePropertyKey.ExpectedUsers];
257 }
258
259 if (propertiesToCache.ContainsKey((byte)GamePropertyKey.EmptyRoomTtl))
260 {
261 this.emptyRoomTtl = (int)propertiesToCache[GamePropertyKey.EmptyRoomTtl];
262 }
263
264 if (propertiesToCache.ContainsKey((byte)GamePropertyKey.PlayerTtl))
265 {
266 this.playerTtl = (int)propertiesToCache[GamePropertyKey.PlayerTtl];
267 }
268
269 // merge the custom properties (from your application) to the cache (only string-typed keys will be kept)
270 this.customProperties.MergeStringKeys(propertiesToCache);
271 this.customProperties.StripKeysWithNullValues();
272 }
