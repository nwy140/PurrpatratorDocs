60 public override void OnRoomListUpdate(List<RoomInfo> roomList)
61 {
62 UpdateStatusText.text = "Updated";
63
64 if (roomList.Count == 0 && !PhotonNetwork.InLobby) {
65 ContentFeedback.text = "No Room found in lobby "+LobbyNameInputField.text+" Matching: "+SqlQueryInputField.text;
66 }
67
68 foreach (RoomInfo entry in roomList)
69 {
70 if (roomCellList.ContainsKey(entry.Name))
71 {
72 if (entry.RemovedFromList)
73 {
74 // we delete the cell
75 roomCellList[entry.Name].RemoveFromList();
76 roomCellList.Remove(entry.Name);
77 }
78 else
79 {
80 // we update the cell
81 roomCellList[entry.Name].RefreshInfo(entry);
82 }
83
84 }
85 else
86 {
87 if (!entry.RemovedFromList)
88 {
89 // we create the cell
90 roomCellList[entry.Name] = Instantiate(CellPrototype);
91 roomCellList[entry.Name].gameObject.SetActive(true);
92 roomCellList[entry.Name].transform.SetParent(CellPrototype.transform.parent, false);
93 roomCellList[entry.Name].AddToList(entry, !_firstUpdate);
94 }
95 }
96 }

99

101 }
