268 private static Type GetType(string typeName)
269 {
270 var type = Type.GetType(typeName);
271 if (type != null) return type;
272 foreach (var a in AppDomain.CurrentDomain.GetAssemblies())
273 {
274 type = a.GetType(typeName);
275 if (type != null)
276 return type;
277 }
278 return null;
279 }

281 private int RpcListHashCode()
282 {
283 // this is a hashcode generated to (more) easily compare this Editor's RPC List with some other
284 int hashCode = PhotonNetwork.PhotonServerSettings.RpcList.Count + 1;
285 foreach (string s in PhotonNetwork.PhotonServerSettings.RpcList)
286 {
287 int h1 = s.GetHashCode();
288 hashCode = ((h1 << 5) + h1) ^ hashCode;
289 }
290 return hashCode;
291 }
