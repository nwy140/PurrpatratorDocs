354 public static void RenderContentsOfScriptableObject(SettingsScriptableObjectBase singleton, bool includeScriptField)
355 {
356 //EditorGUILayout.Space();
357
358 SerializedObject so = new SerializedObject(singleton);
359 SerializedProperty sp = so.GetIterator();
360 sp.Next(true);
361
362 // Skip drawing the script reference?
363 if (!includeScriptField)
364 sp.NextVisible(false);
365
366 EditorGUI.BeginChangeCheck();
367 while (sp.NextVisible(false))
368 {
369 RenderSP(sp);
370 }
371
372 EditorGUILayout.Space();
373
374 if (EditorGUI.EndChangeCheck())
375 {
376 singleton.Initialize();
377 so.ApplyModifiedProperties();
378 AssetDatabase.SaveAssets();
379 }
380 }
