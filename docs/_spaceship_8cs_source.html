63 public void Update()
64 {
65 if (!photonView.AmOwner || !controllable)
66 {
67 return;
68 }
69
70 // we don't want the master client to apply input to remote ships while the remote player is inactive
71 if (this.photonView.CreatorActorNr != PhotonNetwork.LocalPlayer.ActorNumber)
72 {
73 return;
74 }
75
76 rotation = Input.GetAxis("Horizontal");
77 acceleration = Input.GetAxis("Vertical");
78
79 if (Input.GetButton("Jump") && shootingTimer <= 0.0)
80 {
81 shootingTimer = 0.2f;
82
83 photonView.RPC("Fire", RpcTarget.AllViaServer, rigidbody.position, rigidbody.rotation);
84 }
85
86 if (shootingTimer > 0.0f)
87 {
88 shootingTimer -= Time.deltaTime;
89 }
90 }
91
92 public void FixedUpdate()
93 {
94 if (!photonView.IsMine)
95 {
96 return;
97 }
98
99 if (!controllable)
100 {
101 return;
102 }
103
104 Quaternion rot = rigidbody.rotation * Quaternion.Euler(0, rotation * RotationSpeed * Time.fixedDeltaTime, 0);
105 rigidbody.MoveRotation(rot);
106
107 Vector3 force = (rot * Vector3.forward) * acceleration * 1000.0f * MovementSpeed * Time.fixedDeltaTime;
108 rigidbody.AddForce(force);
109
110 if (rigidbody.velocity.magnitude > (MaxSpeed * 1000.0f))
111 {
112 rigidbody.velocity = rigidbody.velocity.normalized * MaxSpeed * 1000.0f;
113 }
114
115 CheckExitScreen();
116 }

200 private void CheckExitScreen()
201 {
202 if (Camera.main == null)
203 {
204 return;
205 }
206
207 if (Mathf.Abs(rigidbody.position.x) > (Camera.main.orthographicSize * Camera.main.aspect))
208 {
209 rigidbody.position = new Vector3(-Mathf.Sign(rigidbody.position.x) * Camera.main.orthographicSize * Camera.main.aspect, 0, rigidbody.position.z);
210 rigidbody.position -= rigidbody.position.normalized * 0.1f; // offset a little bit to avoid looping back & forth between the 2 edges
211 }
212
213 if (Mathf.Abs(rigidbody.position.z) > Camera.main.orthographicSize)
214 {
215 rigidbody.position = new Vector3(rigidbody.position.x, rigidbody.position.y, -Mathf.Sign(rigidbody.position.z) * Camera.main.orthographicSize);
216 rigidbody.position -= rigidbody.position.normalized * 0.1f; // offset a little bit to avoid looping back & forth between the 2 edges
217 }
218 }
