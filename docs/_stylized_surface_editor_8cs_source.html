235 private void HandleUrpSettings(Material material, MaterialEditor materialEditor) {
236 bool alphaClip = false;
237 if (material.HasProperty("_AlphaClip")) {
238 alphaClip = material.GetFloat("_AlphaClip") >= 0.5;
239 }
240
241 if (alphaClip) {
242 material.EnableKeyword("_ALPHATEST_ON");
243 } else {
244 material.DisableKeyword("_ALPHATEST_ON");
245 }
246
247 if (HasProperty("_Surface")) {
248 EditorGUI.BeginChangeCheck();
249 var surfaceProp = FindProperty("_Surface");
250 EditorGUI.showMixedValue = surfaceProp.hasMixedValue;
251 var surfaceType = (SurfaceType) surfaceProp.floatValue;
252 EditorGUILayout.Separator();
253 surfaceType = (SurfaceType) EditorGUILayout.EnumPopup("Surface Type", surfaceType);
254 if (EditorGUI.EndChangeCheck()) {
255 materialEditor.RegisterPropertyChangeUndo("Surface Type");
256 surfaceProp.floatValue = (float) surfaceType;
257 }
258
259 if (surfaceType == SurfaceType.Opaque) {
260 if (alphaClip) {
261 material.renderQueue = (int) UnityEngine.Rendering.RenderQueue.AlphaTest;
262 material.SetOverrideTag("RenderType", "TransparentCutout");
263 } else {
264 material.renderQueue = (int) UnityEngine.Rendering.RenderQueue.Geometry;
265 material.SetOverrideTag("RenderType", "Opaque");
266 }
267
268 material.renderQueue +=
269 material.HasProperty("_QueueOffset") ? (int) material.GetFloat("_QueueOffset") : 0;
270 material.SetInt("_SrcBlend", (int) UnityEngine.Rendering.BlendMode.One);
271 material.SetInt("_DstBlend", (int) UnityEngine.Rendering.BlendMode.Zero);
272 material.SetInt("_ZWrite", 1);
273 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
274 material.SetShaderPassEnabled("ShadowCaster", true);
275 } else // Transparent
276 {
277 BlendMode blendMode = (BlendMode) material.GetFloat("_Blend");
278
279 // Specific Transparent Mode Settings
280 switch (blendMode) {
281 case BlendMode.Alpha:
282 material.SetInt("_SrcBlend", (int) UnityEngine.Rendering.BlendMode.SrcAlpha);
283 material.SetInt("_DstBlend", (int) UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha);
284 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
285 break;
286 case BlendMode.Premultiply:
287 material.SetInt("_SrcBlend", (int) UnityEngine.Rendering.BlendMode.One);
288 material.SetInt("_DstBlend", (int) UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha);
289 material.EnableKeyword("_ALPHAPREMULTIPLY_ON");
290 break;
291 case BlendMode.Additive:
292 material.SetInt("_SrcBlend", (int) UnityEngine.Rendering.BlendMode.SrcAlpha);
293 material.SetInt("_DstBlend", (int) UnityEngine.Rendering.BlendMode.One);
294 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
295 break;
296 case BlendMode.Multiply:
297 material.SetInt("_SrcBlend", (int) UnityEngine.Rendering.BlendMode.DstColor);
298 material.SetInt("_DstBlend", (int) UnityEngine.Rendering.BlendMode.Zero);
299 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
300 material.EnableKeyword("_ALPHAMODULATE_ON");
301 break;
302 }
303
304 // General Transparent Material Settings
305 material.SetOverrideTag("RenderType", "Transparent");
306 material.SetInt("_ZWrite", 0);
307 material.renderQueue = (int) UnityEngine.Rendering.RenderQueue.Transparent;
308 material.renderQueue +=
309 material.HasProperty("_QueueOffset") ? (int) material.GetFloat("_QueueOffset") : 0;
310 material.SetShaderPassEnabled("ShadowCaster", false);
311 }
312
313 // DR: draw popup.
314 if (surfaceType == SurfaceType.Transparent && HasProperty("_Blend")) {
315 EditorGUI.BeginChangeCheck();

360 FilterMode filterMode) {

376 private Texture2D GenerateStepTexture() {
377 int numSteps = _celShadingNumSteps;
378 var t2d = new Texture2D(numSteps + 1, /*height=*/1, TextureFormat.R8, /*mipChain=*/false) {
379 filterMode = FilterMode.Point,
380 wrapMode = TextureWrapMode.Clamp
381 };
382 for (int i = 0; i < numSteps + 1; i++) {
383 var color = Color.white * i / numSteps;
384 t2d.SetPixel(i, 0, color);
385 }
386
387 t2d.Apply();
388 return t2d;
389 }

391 private Texture2D GenerateCurveTexture() {
392 const int width = 256;
393 const int height = 1;
394 var lut = new Texture2D(width, height, TextureFormat.R8, /*mipChain=*/false) {
395 alphaIsTransparency = false,
396 wrapMode = TextureWrapMode.Clamp,
397 hideFlags = HideFlags.HideAndDontSave,
398 filterMode = FilterMode.Trilinear
399 };
400
401 for (float x = 0; x < width; x++) {
402 float value = _gradient.Evaluate(x / width);
403 for (float y = 0; y < height; y++) {
404 var color = Color.white * value;
405 lut.SetPixel(Mathf.CeilToInt(x), Mathf.CeilToInt(y), color);
406 }
407 }
408
409 return lut;

412 private void SaveTextureAsPng(Texture2D texture, string fullPath, FilterMode filterMode) {
413 byte[] bytes = texture.EncodeToPNG();
414 File.WriteAllBytes(fullPath, bytes);
415 AssetDatabase.Refresh();
416 Debug.Log(string.Format("Texture saved as: {0}", fullPath));
417

419 TextureImporter importer = (TextureImporter) TextureImporter.GetAtPath(pathRelativeToAssets);
420 if (importer != null) {
421 importer.filterMode = filterMode;
422 importer.textureType = TextureImporterType.SingleChannel;
423 importer.textureCompression = TextureImporterCompression.Uncompressed;
424 importer.mipmapEnabled = false;
425 var textureSettings = new TextureImporterPlatformSettings {
426 format = TextureImporterFormat.R8
427 };
428 importer.SetPlatformTextureSettings(textureSettings);
429 EditorUtility.SetDirty(importer);
430 importer.SaveAndReimport();
431 }

440 private static Texture2D LoadTexture(string fullPath) {
441 string pathRelativeToAssets = ConvertFullPathToAssetPath(fullPath);
442 if (pathRelativeToAssets.Length == 0) {
443 return null;
444 }
445
446 var loadedTexture = AssetDatabase.LoadAssetAtPath(pathRelativeToAssets, typeof(Texture2D)) as Texture2D;
447 if (loadedTexture == null) {
448 Debug.LogError(string.Format("[FlatKit] Could not load texture from {0} [{1}].", fullPath,
449 pathRelativeToAssets));
450 return null;
451 }
452
453 loadedTexture.filterMode = FilterMode.Point;
454 loadedTexture.wrapMode = TextureWrapMode.Clamp;
455
456 return loadedTexture;
457 }
