86 private void AddVector()
87 {
89 float delta = DoubleTime.mixedDeltaTime;
90
92 transform.localScale += (sclDef.addVector * delta);
93
95 if (rotDef.local)
96 transform.localEulerAngles += rotDef.addVector * delta;
97 else
98 transform.eulerAngles += rotDef.addVector * delta;
99 //transform.Rotate(rotDef.addVector * mixedDelta, rotDef.local ? Space.Self : Space.World);
100
102 transform.Translate(posDef.addVector * delta, posDef.local ? Space.Self : Space.World);
103 }

124 public void AutoSetSyncTransformEnablesAdditive()
125 {
126 var syncTransform = GetComponent<SyncTransform>();
127
128 var st = syncTransform;
129
130 // Position
131 {
132 var def = posDef;
133 var c = st.transformCrusher.PosCrusher;
134 bool local = def.local;
135
136 c.local = local;
137
138 if (local)
139 {
140 bool iszero = def.addVector.magnitude == 0;
141 c.XCrusher.Enabled = !iszero;
142 c.YCrusher.Enabled = !iszero;
143 c.ZCrusher.Enabled = !iszero;
144 }
145 else
146 {
147 var addVector = def.addVector;
148 c.XCrusher.Enabled = addVector.x != 0;
149 c.YCrusher.Enabled = addVector.y != 0;
150 c.ZCrusher.Enabled = addVector.z != 0;
151 }
152 }
153
155 {
156 var def = rotDef;
157 var c = st.transformCrusher.RotCrusher;
158 bool local = def.local;
159
160 c.local = local;
161 bool iszero = def.addVector.magnitude == 0;
162
163 if (iszero || !rotDef.local)
164 {
165 c.TRSType = TRSType.Quaternion;
166 c.QCrusher.Enabled = !iszero;
167 }
168 else
169 {
170 c.TRSType = TRSType.Euler;
171 var addVector = def.addVector;
172
173 c.XCrusher.Enabled = local ? addVector.x != 0 : !iszero;
174 c.YCrusher.Enabled = local ? addVector.y != 0 : !iszero;
175 c.ZCrusher.Enabled = local ? addVector.z != 0 : !iszero;
176 }
177 }
178
179
181 {
182 var def = sclDef;
183 var c = st.transformCrusher.SclCrusher;
184
185 bool usescl = (sclDef.addVector.sqrMagnitude != 0);
186 if (usescl)
187 {
188 bool iszero = def.addVector.magnitude == 0;
189
190 bool local = def.local;
191 var addVector = sclDef.addVector;
192 c.uniformAxes = ElementCrusher.UniformAxes.NonUniform;
193 c.XCrusher.Enabled = local ? addVector.x != 0 : !iszero;
194 c.YCrusher.Enabled = local ? addVector.y != 0 : !iszero;
195 c.ZCrusher.Enabled = local ? addVector.z != 0 : !iszero;
196 c.local = def.local;
197 }
198 else
199 {
200 c.uniformAxes = ElementCrusher.UniformAxes.XYZ;
201 c.UCrusher.Enabled = false;
202 }
203 }
204 }
