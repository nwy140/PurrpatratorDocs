150 protected override bool Trigger(Frame frame, int subFrameId, float timeshift = 0)
151 {
152 Pool p = Pool.Spawn(projPrefab, origin);
153
154 var proxy = p.GetComponent<IContactTrigger>();
155 proxy.Proxy = contactTrigger;
156
157 var ssproj = p.GetComponent<IProjectile>();
158 ssproj.Initialize(this, frame.frameId, subFrameId, velocity, terminateOn, damageOn, lagCompensate * timeshift);
159 ssproj.Owner = this;
160
161 return true;
162 }
