87 protected override bool Trigger(Frame frame, int subFrameId, float timeshift = 0)
88 {
90 if (GetComponent<SyncContact>() && !photonView.IsMine)
91 {
92 hitscanDefinition.VisualizeHitscan(origin);
93 return true;
94 }
95
96 int nearest = -1;
97 RaycastHit[] rayhits;
98 Collider[] colhits;
99 int count = hitscanDefinition.GenericHitscanNonAlloc(origin, out rayhits, out colhits, ref nearest, visualizeHitscan);
100
101 if (count <=0)
102 return true;
103
104 //var contactTrigger = this.contactTrigger;// this.GetComponentInParent<ContactTrigger>();
105 for (int h = 0; h < count; ++h)
106 {
107 var col = colhits[h];
108
109 var otherCT = col.transform.GetNestedComponentInParents<IContactTrigger, NetObject>();
110
111 if (ReferenceEquals(otherCT, null))
112 continue;
113
114 if (otherCT.NetObj == contactTrigger.NetObj)
115 continue;
116
117 if (poke)
118 contactTrigger.OnContact(otherCT, ContactType.Hitscan);
119
120 if (grab)
121 otherCT.OnContact(contactTrigger, ContactType.Hitscan);
122 }
123
124 return true;
125 }
