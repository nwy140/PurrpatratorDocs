224 private void Oscillate(float lerpT)
225 {
226 var start = nodes[0];
227 var end = nodes[1];
228
229 var posLerped = (posDef.includeAxes == 0) ? new Vector3(0, 0, 0) : Vector3.Lerp(start.trs[0], end.trs[0], lerpT);// currentLerpT * position.oscillateRange + position.oscillateStart;
230 var rotLerped = (rotDef.includeAxes == 0) ? new Vector3(0, 0, 0) : Vector3.Lerp(start.trs[1], end.trs[1], lerpT); //currentLerpT * rotation.oscillateRange + rotation.oscillateStart;

233 ApplyOscillate(posLerped, rotLerped, sclLerped);

275 protected override void ApplySnapshot(Frame snapframe, Frame targframe, bool snapIsVaid, bool targIsValid)
276 {
277 if (snapIsVaid && snapframe.content == FrameContents.Complete)
278 {
279 if (movement == Movement.Oscillate)
280 {
281 if (predictWithRTT != 0)
282 {
283 //snapPhase = targPhase;
284
285 float nudge = (NetMaster.RTT + TickEngineSettings.targetBufferInterval) * predictWithRTT;
286 float nudgedSnapTime = snapframe.phase * oscillatePeriod + nudge;
287 float nudgedTargtime = targframe.phase * oscillatePeriod + nudge;
288
289 //Debug.Log(nudge + " time: " + targPhase * oscillatePeriod +
290 // " nudged: " + nudgedTargtime + " origphase: " + targPhase + " newphase: " + TimeToPhase(nudgedTargtime));
291
292 snapPhase = TimeToPhase(nudgedSnapTime);
293 targPhase = TimeToPhase(nudgedTargtime);
294 }
295 else
296 {
297 snapPhase = snapframe.phase;
298 targPhase = targframe.phase;
299 }
300
301 float lerpT = (float)OscillatePhaseToLerpT(snapPhase);
302 Oscillate(lerpT);
303 }
304 }

308 public override bool OnInterpolate(int snapFrameId, int targFrameId, float t)
309 {
310
311 if (!base.OnInterpolate(snapFrameId, targFrameId, t))
312 return false;
313
314 switch (movement)
315 {
316 case Movement.Oscillate:
317 {
318
319 if (targPhase < snapPhase)
320 targPhase += 1;
321
322 float phase = Mathf.Lerp(snapPhase, targPhase, t);
323
324 if (phase >= 1)
325 phase -= 1;
326
327 var lerpT = (float)OscillatePhaseToLerpT(phase);
328
329 Oscillate(lerpT);
330 return true;
331 }
332
333 case Movement.Trigger:
334 {
335 break;
336 }
337 default:
338 break;
339 }
340
341 return false;
342
343 }

347 protected override void ConstructMissingFrame(Frame prevFrame, Frame snapframe, Frame targframe)
348 {
349 //base.ConstructMissingFrame(frameId);
350 switch (movement)
351 {
352 // Oscilate is deterministic in nature. So for missing frames we can turn the previous t back into a moduls time value,
353 // and accurately predict t by adding a tick worth of time to that.
354 case Movement.Oscillate:
355 {
356 if (snapframe.content == FrameContents.Complete)
357 {
358 float restoredTime = snapframe.phase * oscillatePeriod;
359 float extrapTime = restoredTime + TickEngineSettings.netTickInterval;
360 targframe.phase = TimeToPhase(extrapTime);
361 targframe.content = FrameContents.Complete;
362 return;
363 }
364 break;
365 }
366
367 default:
368 break;
369 }

398 private void ApplyOscillate(Vector3 pos, Vector3 rot, Vector3 scl)
399 {
400 var posIncludeAxes = posDef.includeAxes;
401 var rotIncludeAxes = rotDef.includeAxes;
402 var sclIncludeAxes = sclDef.includeAxes;
403
404
405 // Scale
406 if (sclIncludeAxes != AxisMask.None)
407 {
408 transform.localScale = new Vector3(
409 ((sclIncludeAxes & AxisMask.X) != 0) ? scl.x : transform.localScale.x,
410 ((sclIncludeAxes & AxisMask.Y) != 0) ? scl.y : transform.localScale.y,
411 ((sclIncludeAxes & AxisMask.Z) != 0) ? scl.z : transform.localScale.z);
412 }
413
414 // Rotation
415 if (rotIncludeAxes != AxisMask.None)
416 {
417 if (rotDef.local)
418 {
419 transform.localEulerAngles = new Vector3(
420 ((rotIncludeAxes & AxisMask.X) != 0) ? rot.x : transform.localEulerAngles.x,
421 ((rotIncludeAxes & AxisMask.Y) != 0) ? rot.y : transform.localEulerAngles.y,
422 ((rotIncludeAxes & AxisMask.Z) != 0) ? rot.z : transform.localEulerAngles.z);
423 }
424 else
425 {
426 transform.eulerAngles = new Vector3(
427 ((rotIncludeAxes & AxisMask.X) != 0) ? rot.x : transform.eulerAngles.x,
428 ((rotIncludeAxes & AxisMask.Y) != 0) ? rot.y : transform.eulerAngles.y,
429 ((rotIncludeAxes & AxisMask.Z) != 0) ? rot.z : transform.eulerAngles.z);
430 }
431 }
432
433 // Position
434 if (posIncludeAxes != AxisMask.None)
435 {
436 if (posDef.local)
437 {
438 transform.localPosition = new Vector3(
439 ((posIncludeAxes & AxisMask.X) != 0) ? pos.x : transform.localPosition.x,
440 ((posIncludeAxes & AxisMask.Y) != 0) ? pos.y : transform.localPosition.y,
441 ((posIncludeAxes & AxisMask.Z) != 0) ? pos.z : transform.localPosition.z);
442 }
443 else
444 {
445 transform.position = new Vector3(
446 ((posIncludeAxes & AxisMask.X) != 0) ? pos.x : transform.position.x,
447 ((posIncludeAxes & AxisMask.Y) != 0) ? pos.y : transform.position.y,
448 ((posIncludeAxes & AxisMask.Z) != 0) ? pos.z : transform.position.z);
449 }
450 }
