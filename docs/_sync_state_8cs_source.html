664 public virtual bool ChangeMount(int newMountId)
665 {
666 if (ReferenceEquals(currentMount, null))
667 {
668 Debug.LogWarning("'" + name + "' is not currently mounted, so we cannot change to a different mount.");
669 return false;
670 }
671
672 if ((mountableTo & (1 << newMountId)) == 0)
673 {
674 Debug.LogWarning("'" + name + "' is trying to switch to a mount '" + MountSettings.GetName(newMountId) + "' , but mount is not set as valid in SyncState.");
675 return false;
676 }
677
678 var lookup = currentMount.mountsLookup.mountIdLookup;
679
680 if (!lookup.ContainsKey(newMountId))
681 {
682 Debug.LogWarning("'" + name + "' doesn't contain a mount for '" + MountSettings.GetName(newMountId) + "'.");
683 return false;
684 }
685 var attachTo = lookup[newMountId];
686
687 //Debug.Log("New Mount:" + attachTo + ":" + newMountId);
688
689 ChangeState(new StateChangeInfo(currentState.state, attachTo, false));
690
691 return true;
692 }

718 public SerializationFlags OnNetSerialize(int frameId, byte[] buffer, ref int bitposition, SerializationFlags writeFlags)
719 {
720
721 //if (false)
724 if (!enabled) // This was ActiveAndEnabled... but that is a problem if mounting immediately to a spawned object, as it is disabled for some reason.
725 {
726 // leading bool
727 buffer.WriteBool(false, ref bitposition);
728 return SerializationFlags.None;
729 }
730
731 Frame frame = frames[frameId];
732
733 bool iskeyframe = IsKeyframe(frameId);
734 bool isNewConnection = (writeFlags & SerializationFlags.NewConnection) != 0;
735 bool isFirstSend = ReferenceEquals(prevSerializedFrame, null);
736 bool stateHasChanged = isFirstSend || frame.state != prevSerializedFrame.state;
737 //bool parentHasChanged = isFirstSend || frame.attachedToViewID != prevSerializedFrame.attachedToViewID;
738 bool mountHasChanged = isFirstSend || frame.mountTypeId != prevSerializedFrame.mountTypeId || frame.mountToViewID != prevSerializedFrame.mountToViewID;
739 bool needsComplete = isNewConnection || iskeyframe || isFirstSend;
740 bool hasChanged = stateHasChanged || mountHasChanged;
741 bool hasContent = needsComplete || hasChanged;
742 bool forceReliable = isFirstSend || isNewConnection || (keyframeRate == 0 && hasContent) || (mountReliable && mountHasChanged);
743
744 // If we determined there is nothing to send, first bit 0 and exit
745 if (!hasContent && !forceReliable)
746 {
747 // leading bool
748 buffer.WriteBool(false, ref bitposition);
749 return SerializationFlags.None;
750 }
751
752 // Leading bool
753 buffer.WriteBool(true, ref bitposition);
754
755 var flags = SerializationFlags.HasContent;
756
757 // Determine if anything requires this be made a Reliable send
758 if (forceReliable)
759 {
760 //if (GetComponent<SyncPickup>())
761 // Debug.LogError(Time.time + " <b>" + frame.frameId + ":" + frameId + "</b> STATE " + isFirstSend + " " + isNewConnection + " " + (keyframeRate == 0 && hasContent) + " " + (mountReliable && mountHasChanged) + " " + frame.state);
762
763 flags |= SerializationFlags.ForceReliable;
764 }
765
767 buffer.Write((ulong)frame.state, ref bitposition, 6);
768
769 if ((frame.state & ObjState.Mounted) != 0)
770 {
771 if (mountHasChanged || needsComplete)
772 {
774 if (!iskeyframe)
775 buffer.Write(1, ref bitposition, 1);
776
777 //if (GetComponent<SyncPickup>())
