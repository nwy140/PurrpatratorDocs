52 public static SyncTransform AddDefaultDisabledSyncTransform(Transform t)
53 {
54 if (ReferenceEquals(t, null))
55 t = Selection.activeTransform;
56
57 if (ReferenceEquals(t, null))
58 return null;
59
60 SyncTransform st = t.GetComponent<SyncTransform>();
61 if (!st)
62 {
63 st = t.gameObject.AddComponent<SyncTransform>();
64 Debug.Log("Added SyncTransform to " + t.name);
65 var tc = st.transformCrusher;
66 var pc = tc.PosCrusher;
67 var rc = tc.RotCrusher;
68 var sc = tc.SclCrusher;
69
70 pc.XCrusher.Enabled = false;
71 pc.YCrusher.Enabled = false;
72 pc.ZCrusher.Enabled = false;
73
74 rc.TRSType = TRSType.Euler;
75 rc.XCrusher.Enabled = false;
76 rc.YCrusher.Enabled = false;
77 rc.ZCrusher.Enabled = false;
78
79 sc.uniformAxes = ElementCrusher.UniformAxes.NonUniform;
80 sc.XCrusher.Enabled = false;
81 sc.YCrusher.Enabled = false;
82 sc.ZCrusher.Enabled = false;
83 }
84
85 return st;
86 }
