344 public SerializationFlags Pack(ref String value, String prevValue, byte[] buffer, ref int bitposition, int frameId, SerializationFlags writeFlags)
345 {
346
347 if (!IsForcedClass(frameId, value, prevValue, writeFlags))
348 return SerializationFlags.None;
349
350 if (value == null)
351 {
352 buffer.Write(0, ref bitposition, 6);
353 return SerializationFlags.IsComplete;
354 }
355
357 int cnt = value.Length;
358 if (cnt > MAX_STR_LEN)
359 {
360 cnt = MAX_STR_LEN;
361#if UNITY_EDITOR
362 Debug.LogWarning("Default Pack attribute has a max string length of 63 characters. Use PackString for more options.");
363#endif
364 }
365
367 buffer.Write((uint)cnt, ref bitposition, 6);
368
370 for (int i = 0; i < cnt; ++i)
371 buffer.Write((ushort)value[i], ref bitposition, bitCount);
372 return SerializationFlags.IsComplete;
373 }
