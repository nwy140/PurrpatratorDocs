150 public virtual int GetMaxBits(Type fieldType)
151 {
152
153 if (fieldType.IsEnum)
154 {
155 fieldType = Enum.GetUnderlyingType(fieldType);
156 }
157
158 if (fieldType == typeof(Byte) || fieldType == typeof(SByte))
159 return 8;
160
161 if (fieldType == typeof(UInt16) || fieldType == typeof(Int16) || fieldType == typeof(Char))
162 return 16;
163
164 if (fieldType == typeof(UInt32) || fieldType == typeof(Int32) || fieldType == typeof(Single))
165 return 32;
166
167 if (fieldType == typeof(UInt64) || fieldType == typeof(Int64) || fieldType == typeof(Double))
168 return 64;
169
170 if (fieldType == typeof(Boolean))
171 return 1;
172
173 if (fieldType == typeof(Vector3))
174 return 96;
175
176 if (fieldType == typeof(Vector2))
177 return 64;
178
179 bool isList = fieldType.IsGenericType && fieldType.GetGenericTypeDefinition().IsAssignableFrom(typeof(List<>));
180
181 if (isList)
182 {
183 Debug.LogWarning("Can't get max bits needed for List<> types, as they are variable. " + fieldType.Name);
184 return 256 * 8;
185 }
186
187 Debug.LogWarning("Can't get bits needed for unsupported types. " + fieldType.Name);
188 return 256 * 8;
189 }

214 public bool IsForced(int frameId, SerializationFlags writeFlags)
215 {
216 if (syncAs == SyncAs.Trigger)
217 {

221
