54 int rbCount = reusableRBList.Count;
55 int rb2dCount = reusableRB2DList.Count;
56
57 bool isRigidBody = rbCount > 0 || rb2dCount > 0;

59
60 if (isRigidBody)
61 {
63 if (isOnRigidbody)
64 {
65 int colliderCount = sshealth.transform.CountChildCollider(false, true);
66 if (colliderCount < 1)
67 EditorGUILayout.HelpBox("Cannot locate any non-trigger Collider/Collider2D on this GameObject. Will not be able to detect RB collisions.", MessageType.Warning);
68 }
70 else
71 {
72 EditorGUILayout.HelpBox(sshealth.GetType().Name +
73 " must be on same child as a RigidBody to detect Rigidbody collisions.", MessageType.Warning);
74 }
75 }
76
77 if (!isOnRigidbody)
78 {
79 int triggerCount = sshealth.transform.CountChildCollider(true, true);
80
81 if (triggerCount < 1)

83
84 if (triggerCount > 1)

86 "More than one may cause multiple trigger events.", MessageType.Warning);
