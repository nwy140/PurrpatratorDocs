169 private void Move()
170 {
171 // set target speed based on move speed, sprint speed and if sprint is pressed
172 float targetSpeed = _input.sprint ? SprintSpeed : MoveSpeed;
173
174 // a simplistic acceleration and deceleration designed to be easy to remove, replace, or iterate upon
175
176 // note: Vector2's == operator uses approximation so is not floating point error prone, and is cheaper than magnitude
177 // if there is no input, set the target speed to 0
178 if (_input.move == Vector2.zero) targetSpeed = 0.0f;
179
180 // a reference to the players current horizontal velocity
181 float currentHorizontalSpeed = new Vector3(_controller.velocity.x, 0.0f, _controller.velocity.z).magnitude;
182
183 float speedOffset = 0.1f;
184 float inputMagnitude = _input.analogMovement ? _input.move.magnitude : 1f;
185
186 // accelerate or decelerate to target speed
187 if (currentHorizontalSpeed < targetSpeed - speedOffset || currentHorizontalSpeed > targetSpeed + speedOffset)
188 {
189 // creates curved result rather than a linear one giving a more organic speed change
190 // note T in Lerp is clamped, so we don't need to clamp our speed
191 _speed = Mathf.Lerp(currentHorizontalSpeed, targetSpeed * inputMagnitude, Time.deltaTime * SpeedChangeRate);
192
193 // round speed to 3 decimal places
194 _speed = Mathf.Round(_speed * 1000f) / 1000f;
195 }
196 else
197 {
198 _speed = targetSpeed;
199 }
200 _animationBlend = Mathf.Lerp(_animationBlend, targetSpeed, Time.deltaTime * SpeedChangeRate);
201
202 // normalise input direction
203 Vector3 inputDirection = new Vector3(_input.move.x, 0.0f, _input.move.y).normalized;
204
205 // note: Vector2's != operator uses approximation so is not floating point error prone, and is cheaper than magnitude
206 // if there is a move input rotate player when the player is moving
207 if (_input.move != Vector2.zero)
208 {
209 _targetRotation = Mathf.Atan2(inputDirection.x, inputDirection.z) * Mathf.Rad2Deg + _mainCamera.transform.eulerAngles.y;
210 float rotation = Mathf.SmoothDampAngle(transform.eulerAngles.y, _targetRotation, ref _rotationVelocity, RotationSmoothTime);
211
212 // rotate to face input direction relative to camera position
213 transform.rotation = Quaternion.Euler(0.0f, rotation, 0.0f);
214 }
215
216
217 Vector3 targetDirection = Quaternion.Euler(0.0f, _targetRotation, 0.0f) * Vector3.forward;
218
219 // move the player
220 _controller.Move(targetDirection.normalized * (_speed * Time.deltaTime) + new Vector3(0.0f, _verticalVelocity, 0.0f) * Time.deltaTime);
221
222 // update animator if using character
223 if (_hasAnimator)
224 {
225 _animator.SetFloat(_animIDSpeed, _animationBlend);
226 _animator.SetFloat(_animIDMotionSpeed, inputMagnitude);
227 }
228 }

230 private void JumpAndGravity()
231 {
232 if (Grounded)
233 {
234 // reset the fall timeout timer
235 _fallTimeoutDelta = FallTimeout;
236
237 // update animator if using character
238 if (_hasAnimator)
239 {
240 _animator.SetBool(_animIDJump, false);
241 _animator.SetBool(_animIDFreeFall, false);
242 }
243
244 // stop our velocity dropping infinitely when grounded
245 if (_verticalVelocity < 0.0f)
246 {
247 _verticalVelocity = -2f;
248 }
249
250 // Jump
251 if (_input.jump && _jumpTimeoutDelta <= 0.0f)
252 {
253 // the square root of H * -2 * G = how much velocity needed to reach desired height
254 _verticalVelocity = Mathf.Sqrt(JumpHeight * -2f * Gravity);
255
256 // update animator if using character
257 if (_hasAnimator)
258 {
259 _animator.SetBool(_animIDJump, true);
260 }
261 }
262
263 // jump timeout
264 if (_jumpTimeoutDelta >= 0.0f)
265 {
266 _jumpTimeoutDelta -= Time.deltaTime;
267 }
268 }
269 else
270 {
271 // reset the jump timeout timer
272 _jumpTimeoutDelta = JumpTimeout;
273
274 // fall timeout
275 if (_fallTimeoutDelta >= 0.0f)
276 {
277 _fallTimeoutDelta -= Time.deltaTime;
278 }
279 else
280 {
281 // update animator if using character
282 if (_hasAnimator)
283 {
284 _animator.SetBool(_animIDFreeFall, true);
285 }
286 }
287
288 // if we are not grounded, do not jump
289 _input.jump = false;
290 }
291
292 // apply gravity over time if under terminal (multiply by delta time twice to linearly speed up over time)
293 if (_verticalVelocity < _terminalVelocity)
294 {
295 _verticalVelocity += Gravity * Time.deltaTime;
296 }
297 }
