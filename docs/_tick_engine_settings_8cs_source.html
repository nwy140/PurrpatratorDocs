130 private void CalculateBufferValues()
131 {
133 if (_bufferCorrection == BufferCorrection.Manual)
134 {
135 //targetBufferSize = single._targetBufferSize;
136 minBufferSize = _minBufferSize;
137 maxBufferSize = _maxBufferSize;
138 }
139 else
140 {
141 minBufferSize = _minBufferSize;
142 maxBufferSize = _maxBufferSize;
143 }
144
145 frameCount = (int)_frameCount;
146 halfFrameCount = frameCount / 2;
147 thirdFrameCount = frameCount / 3;
148 quaterFrameCount = frameCount / 4;
149
150 frameCountBits = frameCount.GetBitsForMaxValue();
151
152 netTickInterval = Time.fixedDeltaTime * _sendEveryXTick;
153 netTickIntervalInv = 1f / (Time.fixedDeltaTime * _sendEveryXTick);
154 targetBufferInterval = Time.fixedDeltaTime * _sendEveryXTick * _targetBufferSize;
155
156 sendEveryXTick = _sendEveryXTick;
157
158 targetBufferSize = _targetBufferSize;
159 ticksBeforeGrow = _ticksBeforeGrow;
160 ticksBeforeShrink = _ticksBeforeShrink;
161
162 secondsOfBuffer = Time.fixedDeltaTime * _sendEveryXTick * frameCount;
163 secondsOfHalfBuffer = secondsOfBuffer * .5f;
164 bufferTargSecs = netTickInterval * _targetBufferSize;
165 }

258 private void AutoSetBuffer(SerializedProperty frameCount, SerializedProperty sendEveryX)
259 {
260 float secondsPerTick = Time.fixedDeltaTime * sendEveryX.intValue;
261
262 float framesNeeded = 1 / secondsPerTick;
263
264
265 if (framesNeeded < (float)FrameCountEnum.FrameCount12)
266 frameCount.intValue = (int)FrameCountEnum.FrameCount12;
267
268 else if (framesNeeded < (float)FrameCountEnum.FrameCount30)
269 frameCount.intValue = (int)FrameCountEnum.FrameCount30;
270
271 else if (framesNeeded < (float)FrameCountEnum.FrameCount60)
272 frameCount.intValue = (int)FrameCountEnum.FrameCount60;
273
274 else
275 frameCount.intValue = (int)FrameCountEnum.FrameCount120;
276
277 //Debug.Log("Frames needed for 1 sec " + framesNeeded + " : " + secondsPerTick + " frameCount: " + FrameCount + " fixed: " + Time.fixedDeltaTime + " " + sendEveryX.intValue);
