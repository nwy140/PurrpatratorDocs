186 public VitalData Deserialize(byte[] buffer, ref int bitposition, bool keyframe = true)
187 {
188 double val = (keyframe || buffer.ReadBool(ref bitposition)) ? (double)buffer.Read(ref bitposition, bitsForValue) : double.NegativeInfinity;
189 VitalData data = new VitalData(
190 val,
191 (int)(buffer.ReadBool(ref bitposition) ? buffer.Read(ref bitposition, bitsForDecayDelay) : 0),
192 (int)(buffer.ReadBool(ref bitposition) ? buffer.Read(ref bitposition, bitsForRegenDelay) : 0)
193 );
194
195 return data;
196 }

198 public VitalData Extrapolate(VitalData prev)
199 {
200 int ticksuntilregen = prev.ticksUntilRegen > 0 ? prev.ticksUntilRegen - 1 : 0;
201 int ticksuntildecay = prev.ticksUntilDecay > 0 ? prev.ticksUntilDecay - 1 : 0;
202 double preval = prev.Value;
203 double val =
204 (preval > FullValue && ticksuntildecay == 0) ? preval - _decayPerTick :
205 (preval < FullValue && ticksuntilregen == 0) ? preval + _regenPerTick :
206 preval;
207
208 return new VitalData(val, ticksuntildecay, ticksuntilregen);
209 }
