112 public bool FindUIElements()

124 //Debug.Log(name + " Find Rends " + resuableFindSpriteRend.Count);

126 if (resuableFindSpriteRend.Count > 0 && barSprite == null)

129 backSprite = resuableFindSpriteRend[1];
130
131 if (UIText == null)
132 UIText = (searchChildren) ? GetComponentInChildren<Text>() : GetComponent<Text>();
133
134 if (UIImage == null)
135 UIImage = (searchChildren) ? GetComponentInChildren<Image>() : GetComponent<Image>();

141 public override void UpdateGraphics(Vital vital)
142 {
143 if (vital == null)
144 return;
145
146 var vitalDef = vital.VitalDef;
147
148 double val = /*(vital == null) ? 1 :*/
149 (targetField == TargetField.Value) ? vital.VitalData.Value :
150 (targetField == TargetField.Max) ? vitalDef.FullValue :
151 vitalDef.MaxValue;
152
154 if (val == float.NegativeInfinity)
155 return;
156
157 if (textMesh)
158 textMesh.text = ((int)val).ToString();
159
160 if (barSprite)
161 barSprite.size = new Vector2((float)((val / vitalDef.MaxValue) * widthMultiplier), barSprite.size.y);
162
163 if (UIText != null)
164 {
165 UIText.text = ((int)val).ToString();
166 }
167
168 // Resize the healthbar
169 if (UIImage != null)
170 {
171 double fullval = vitalDef.MaxValue;
172
173 if (UIImage.type == Image.Type.Filled && UIImage.sprite != null)
174 UIImage.fillAmount = (float)((val / fullval) * widthMultiplier);
175 else
176 UIImage.rectTransform.localScale = new Vector3(
177 (float)((val / fullval) * widthMultiplier),
178 UIImage.rectTransform.localScale.y,
179 UIImage.rectTransform.localScale.z);
180 }
181 }
