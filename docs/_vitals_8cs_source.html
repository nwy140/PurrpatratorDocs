61 public void Initialize()
62 {
63 if (initialized)
64 return;
65
66
68 vitalCount = vitalDefs.Count;
69 vitalArray = new Vital[vitalCount];
70
71 for (int i = 0; i < vitalCount; ++i)
72 {
73 VitalDefinition vdef = vitalDefs[i];
74 Vital vital = new Vital(vdef);
75 vitalArray[i] = vital;
76
77 vital.Initialize(TickEngineSettings.netTickInterval);
78 vital.AddIOnVitalChange(this);
79
80 // add all of the vital types to the lookup. Customs use hash, enumerated are added both as hash and by their index
81 int hash = vdef.VitalName.hash;
82 if (vdef.VitalName.type != VitalType.None)
83 {
84 if (!vitalLookup.ContainsKey(hash))
85 vitalLookup.Add(hash, vital);
86 else
87 Debug.LogWarning("VitalNameType hash collision! Vitals cannot have more than one of each Vital Type in its list.");
88 }
89 }
90
91 initialized = true;

149 public SerializationFlags Deserialize(VitalsData vdata, byte[] buffer, ref int bitposition, bool keyframe)
150 {
151 var datas = vdata.datas;
152
153 bool isComplete = true;
154 bool hasChanged = false;
155
156 for (int i = 0; i < vitalCount; ++i)
157 {
158 datas[i] = vitalDefs[i].Deserialize(buffer, ref bitposition, keyframe);
159
161 if (datas[i].Value == double.NegativeInfinity)
162 isComplete = false;
163 else
164 hasChanged |= true;
165 }
166
167 return
168 isComplete ? SerializationFlags.IsComplete | SerializationFlags.HasContent :
169 hasChanged ? SerializationFlags.HasContent :
170 SerializationFlags.None;
171 }

208 private double ApplyCharges(int vitalIdx, double discharge, bool allowOverload, bool propagate)
209 {
210
211 double remaining = discharge;
212 double consumed = 0;
213
215 for (int i = vitalIdx; i >= 0; --i)
216 {
217 consumed += vitalArray[i].ApplyCharges(remaining, allowOverload, !propagate);
218
219 if (propagate)
220 {
221 remaining -= consumed;
222
223 if (remaining == 0)
224 break;
225 }
226 else
227 return consumed;
228 }
229
230 return consumed;
231 }

238 public double ApplyCharges(VitalNameType vitalNameType, double discharge, bool allowOverload, bool propagate)
239 {
240 var vtype = vitalNameType.type;
241
242 int vitalIdx;
243
244 // Type None indicates no specific Vital, so this value should be applied to any complete stack.
245 if (vtype == VitalType.None)
246 vitalIdx = vitalCount - 1;
247 else
248 {
249 vitalIdx = GetVitalIndex(vitalNameType);
250
251 if (vitalIdx == -1)
252 return 0;
253 }
254
255 double consumed = ApplyCharges(vitalIdx, discharge, allowOverload, propagate);
256
257 CheckForDisrupt(consumed);
258
259 return consumed;
260 }
