65 public WebRpcResponse(OperationResponse response)
66 {
67 object value;
68 if (response.Parameters.TryGetValue(ParameterCode.UriPath, out value))
69 {
70 this.Name = value as string;
71 }
72
73 this.ResultCode = -1;
74 if (response.Parameters.TryGetValue(ParameterCode.WebRpcReturnCode, out value))
75 {
76 this.ResultCode = (byte)value;
77 }
78
79 if (response.Parameters.TryGetValue(ParameterCode.WebRpcParameters, out value))
80 {
81 this.Parameters = value as Dictionary<string, object>;
82 }
83
84 if (response.Parameters.TryGetValue(ParameterCode.WebRpcReturnMessage, out value))
85 {
86 this.Message = value as string;
87 }
88 }
89
92 public string ToStringFull()
93 {
94 return string.Format("{0}={2}: {1} \"{3}\"", this.Name, SupportClass.DictionaryToString(this.Parameters), this.ResultCode, this.Message);
95 }
