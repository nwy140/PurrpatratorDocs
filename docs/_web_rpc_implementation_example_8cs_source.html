28 public void OnWebRpcResponse(OperationResponse response)
29 {
30 Debug.LogFormat("WebRPC operation response {0}", response.ToStringFull());
31 switch (response.ReturnCode)
32 {
33 case ErrorCode.Ok:
34 WebRpcResponse webRpcResponse = new WebRpcResponse(response);
35 Debug.LogFormat("Parsed WebRPC response {0}", response.ToStringFull());
36 if (string.IsNullOrEmpty(webRpcResponse.Name))
37 {
38 Debug.LogError("Unexpected: WebRPC response did not contain WebRPC method name");
39 }
40 if (webRpcResponse.ResultCode == 0) // success
41 {
42 switch (webRpcResponse.Name)
43 {
44 // todo: add your code here
45 case GetGameListWebRpcMethodName: // example
46 // ...
47 break;
48 }
49 }
50 else if (webRpcResponse.ResultCode == -1)
51 {
52 Debug.LogErrorFormat("Web server did not return ResultCode for WebRPC method=\"{0}\", Message={1}", webRpcResponse.Name, webRpcResponse.Message);
53 }
54 else
55 {
56 Debug.LogErrorFormat("Web server returned ResultCode={0} for WebRPC method=\"{1}\", Message={2}", webRpcResponse.ResultCode, webRpcResponse.Name, webRpcResponse.Message);
57 }
58 break;
59 case ErrorCode.ExternalHttpCallFailed: // web service unreachable
60 Debug.LogErrorFormat("WebRPC call failed as request could not be sent to the server. {0}", response.DebugMessage);
61 break;
62 case ErrorCode.HttpLimitReached: // too many WebRPCs in a short period of time
63 // the debug message should contain the limit exceeded
64 Debug.LogErrorFormat("WebRPCs rate limit exceeded: {0}", response.DebugMessage);
65 break;
66 case ErrorCode.InvalidOperation: // WebRPC not configured at all OR not configured properly OR trying to send on name server
67 if (PhotonNetwork.Server == ServerConnection.NameServer)
68 {
69 Debug.LogErrorFormat("WebRPC not supported on NameServer. {0}", response.DebugMessage);
70 }
71 else
72 {
73 Debug.LogErrorFormat("WebRPC not properly configured or not configured at all. {0}", response.DebugMessage);
74 }
75 break;
76 default:
77 // other unknown error, unexpected
78 Debug.LogErrorFormat("Unexpected error, {0} {1}", response.ReturnCode, response.DebugMessage);
79 break;
80 }

88 public static void WebRpcCall(string methodName, object parameters = null, bool sendAuthCookieIfAny = false)
89 {
90 if (string.IsNullOrEmpty(methodName))
91 {
92 Debug.LogError("WebRpc method name must not be null nor empty");
93 return;
94 }
95 if (!PhotonNetwork.WebRpc(methodName, parameters, sendAuthCookieIfAny))
96 {
97 Debug.LogErrorFormat("Error sending WebRPC \"{0}\" (\"{1}\") request, check the previous error logs for more details", methodName, parameters);
98 }
