186 private IEnumerator GenerateWorld()
187 {
188 Debug.Log(string.Format("<b>Procedural Demo</b>: Creating world using Seed: {0}, World Size: {1}, Cluster Size: {2} and World Type: {3}", Seed, WorldSize, ClusterSize, WorldType));
189
190 Simplex.Noise.Seed = Seed;
191
192 int clusterId = 0;
193
194 // Instantiating all necessary clusters at their target position
195 for (int x = 0; x < (int) WorldSize; x += (int) Mathf.Sqrt((int) ClusterSize))
196 {
197 for (int z = 0; z < (int) WorldSize; z += (int) Mathf.Sqrt((int) ClusterSize))
198 {
199 GameObject cluster = new GameObject();
200 cluster.name = "Cluster " + clusterId;
201
202 cluster.transform.SetParent(transform);
203 cluster.transform.position = new Vector3(x, 0.0f, z);
204
205 Cluster clusterComponent = cluster.AddComponent<Cluster>();
206 clusterComponent.ClusterId = clusterId;
207
208 clusterList.Add(clusterId++, cluster);
209 }
210 }
211
212 yield return new WaitForEndOfFrame();
213
214 // Instantiating all necessary blocks as a child of a certain cluster
215 foreach (GameObject cluster in clusterList.Values)
216 {
217 Vector3 clusterPosition = cluster.transform.position;
218
219 int blockId = 0;
220
221 for (int x = 0; x < (int) Mathf.Sqrt((int) ClusterSize); ++x)
222 {
223 for (int z = 0; z < (int) Mathf.Sqrt((int) ClusterSize); ++z)
224 {
225 float noiseValue = Simplex.Noise.CalcPixel2D((int) clusterPosition.x + x, (int) clusterPosition.z + z, 0.02f);
226
227 int height = (int) noiseValue / (int) (256.0f / (float) WorldType);
228 int materialIndex = (int) noiseValue / (int) (256.0f / WorldMaterials.Length);
229
230 GameObject block = GameObject.CreatePrimitive(PrimitiveType.Cube);
231 block.name = "Block " + blockId;
232
233 block.transform.SetParent(cluster.transform);
234 block.transform.localScale = new Vector3(1.0f, height, 1.0f);
235 block.transform.position = new Vector3(clusterPosition.x + x, height / 2.0f, clusterPosition.z + z);
236 block.GetComponent<MeshRenderer>().material = WorldMaterials[materialIndex];
237
238 Block blockComponent = block.AddComponent<Block>();
239 blockComponent.BlockId = blockId;
240 blockComponent.ClusterId = cluster.GetComponent<Cluster>().ClusterId;
241
242 cluster.GetComponent<Cluster>().AddBlock(blockId++, block);
243 }
244 }
245
246 yield return new WaitForEndOfFrame();
247 }
248
249 // Applying modifications made to the world when joining the room later or while it is created
250 foreach (DictionaryEntry entry in PhotonNetwork.CurrentRoom.CustomProperties)
251 {
252 if (entry.Value == null)
253 {
254 continue;
255 }
256
257 string key = entry.Key.ToString();
258
259 if ((key == SeedPropertiesKey) || (key == WorldSizePropertiesKey) || (key == ClusterSizePropertiesKey) || (key == WorldTypePropertiesKey))
260 {
261 continue;
262 }
263
264 int indexOfBlank = key.IndexOf(' ');
265 key = key.Substring(indexOfBlank + 1, (key.Length - (indexOfBlank + 1)));
266
267 int.TryParse(key, out clusterId);
268
269 GameObject cluster;
270 if (clusterList.TryGetValue(clusterId, out cluster))
271 {
272 Cluster c = cluster.GetComponent<Cluster>();
273
274 if (c != null)
275 {
276 Dictionary<int, float> clusterModifications = (Dictionary<int, float>) entry.Value;
277
278 foreach (KeyValuePair<int, float> pair in clusterModifications)
279 {
280 c.SetBlockHeightRemote(pair.Key, pair.Value);
281 }
282 }
283 }
